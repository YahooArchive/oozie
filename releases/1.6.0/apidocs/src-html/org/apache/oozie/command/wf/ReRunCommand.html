056    public class ReRunCommand extends Command<Void> {
057    
058        private String jobId;
059        private Configuration conf;
060        private String authToken;
061        private Set<String> nodesToSkip = new HashSet<String>();
062        public static final String TO_SKIP = "TO_SKIP";
063    
064        public ReRunCommand(String jobId, Configuration conf, String authToken) {
065            super("rerun", "rerun", 0, XLog.STD);
066            this.jobId = ParamChecker.notEmpty(jobId, "jobId");
067            this.conf = ParamChecker.notNull(conf, "conf");
068            this.authToken = ParamChecker.notEmpty(authToken, "authToken");
069        }
070    
071        /**
072         * Checks the pre-conditions that are required for workflow to recover -
073         * Last run of Workflow should be completed - The nodes that are to be
074         * skipped are to be completed successfully in the base run.
075         * 
076         * @param wfBean Workflow bean
077         * @param actions List of actions of Workflow
078         * @throws org.apache.oozie.command.CommandException On failure of pre-conditions
079         */
080        private void checkPreConditions(WorkflowJobBean wfBean, List<WorkflowActionBean> actions) throws CommandException {
081            if (!(wfBean.getStatus().equals(WorkflowJob.Status.FAILED) || wfBean.getStatus().equals(WorkflowJob.Status.KILLED) ||
082                  wfBean.getStatus().equals(WorkflowJob.Status.SUCCEEDED))) {
083                throw new CommandException(ErrorCode.E0805, wfBean.getStatus());
084            }
085            Set<String> unmachedNodes = new HashSet<String>(nodesToSkip);
086            for (WorkflowActionBean action : actions) {
087                if (nodesToSkip.contains(action.getName())) {
088                    if (!action.getStatus().equals(WorkflowAction.Status.OK) &&
089                        !action.getStatus().equals(WorkflowAction.Status.ERROR)) {
090                        throw new CommandException(ErrorCode.E0806, action.getName());
091                    }
092                    unmachedNodes.remove(action.getName());
093                }
094            }
095            if (unmachedNodes.size() > 0) {
096                StringBuilder sb = new StringBuilder();
097                String separator = "";
098                for (String s : unmachedNodes) {
099                    sb.append(separator).append(s);
100                    separator = ",";
101                }
102                throw new CommandException(ErrorCode.E0807, sb);
103            }
104        }
105    
106        /**
107         * Parses the config and adds the nodes that are to be skipped to the
108         * skipped node list
109         */
110        private void parseSkippedNodeConf() {
111            if (conf != null) {
112                Collection<String> skipNodes = conf.getStringCollection(OozieClient.RERUN_SKIP_NODES);
113                for (String str : skipNodes) {
114                    // trimming is required
115                    nodesToSkip.add(str.trim());
116                }
117            }
118        }
119    
120        protected Void call(WorkflowStore store) throws StoreException, CommandException {
121            incrJobCounter(1);
122            WorkflowJobBean wfBean = store.getWorkflow(jobId, true);
123            setLogInfo(wfBean);
124            List<WorkflowActionBean> actions = store.getActionsForWorkflow(jobId, true);
125            WorkflowInstance oldWfInstance = wfBean.getWorkflowInstance();
126            WorkflowInstance newWfInstance;
127            XLog log = XLog.getLog(getClass());
128            parseSkippedNodeConf();
129            checkPreConditions(wfBean, actions);
130    
131            WorkflowAppService wps = Services.get().get(WorkflowAppService.class);
132            try {
133                XLog.Info.get().setParameter(DagXLogInfoService.TOKEN, conf.get(OozieClient.LOG_TOKEN));
134                WorkflowApp app = wps.parseDef(conf, authToken);
135                XConfiguration protoActionConf = wps.createProtoActionConf(conf, authToken);
136                WorkflowLib workflowLib = Services.get().get(WorkflowStoreService.class).getWorkflowLibWithNoDB();
137    
138                Path configDefault = new Path(conf.get(OozieClient.APP_PATH), SubmitCommand.CONFIG_DEFAULT);
139    
140                FileSystem fs = Services.get().get(HadoopAccessorService.class).
141                        createFileSystem(wfBean.getUser(), wfBean.getGroup(), configDefault.toUri(), new Configuration());
142    
143                if (fs.exists(configDefault)) {
144                    Configuration defaultConf = new XConfiguration(fs.open(configDefault));
145                    SubmitCommand.validateDefaultConfiguration(defaultConf);
146                    XConfiguration.injectDefaults(defaultConf, conf);
147                }
148                try {
149                    newWfInstance = workflowLib.createInstance(app, conf, jobId);
150                }
151                catch (WorkflowException e) {
152                    throw new StoreException(e);
153                }
154                wfBean.setAppName(app.getName());
155                wfBean.setProtoActionConf(protoActionConf.toXmlString());
156            }
157            catch (WorkflowException ex) {
158                throw new CommandException(ex);
159            }
160            catch (IOException ex) {
161                throw new CommandException(ErrorCode.E0803, ex);
162            }
163    
164            for (int i = 0; i < actions.size(); i++) {
165                if (!nodesToSkip.contains(actions.get(i).getName())) {
166                    store.deleteAction(actions.get(i).getId());
167                    log.info("Deleting Action[{0}] for re-run", actions.get(i).getId());
168                }
169                else {
170                    copyActionData(newWfInstance, oldWfInstance);
171                }
172            }
173    
174            wfBean.setAppPath(conf.get(OozieClient.APP_PATH));
175            wfBean.setConf(XmlUtils.prettyPrint(conf).toString());
176            wfBean.setLogToken(conf.get(OozieClient.LOG_TOKEN, ""));
177            wfBean.setUser(conf.get(OozieClient.USER_NAME));
178            wfBean.setGroup(conf.get(OozieClient.GROUP_NAME));
179            wfBean.setExternalId(conf.get(OozieClient.EXTERNAL_ID));
180            wfBean.setEndTime(null);
181            wfBean.setRun(wfBean.getRun() + 1);
182            wfBean.setStatus(WorkflowJob.Status.PREP);
183            wfBean.setWorkflowInstance(newWfInstance);
184            store.updateWorkflow(wfBean);
185            return null;
186        }
187    
188        /**
189         * Copys the variables for skipped nodes from the old wfInstance to new one.
190         * 
191         * @param newWfInstance
192         * @param oldWfInstance
193         */
194        private void copyActionData(WorkflowInstance newWfInstance, WorkflowInstance oldWfInstance) {
195            Map<String, String> oldVars = new HashMap<String, String>();
196            Map<String, String> newVars = new HashMap<String, String>();
197            oldVars = oldWfInstance.getAllVars();
198            for (String var : oldVars.keySet()) {
199                String actionName = var.split(WorkflowInstance.NODE_VAR_SEPARATOR)[0];
200                if (nodesToSkip.contains(actionName)) {
201                    newVars.put(var, oldVars.get(var));
202                }
203            }
204            for (String node : nodesToSkip) {
205                // Setting the TO_SKIP variable to true. This will be used by
206                // SignalCommand and LiteNodeHandler to skip the action.
207                newVars.put(node + WorkflowInstance.NODE_VAR_SEPARATOR + TO_SKIP, "true");
208                String visitedFlag = NodeHandler.getLoopFlag(node);
209                // Removing the visited flag so that the action won't be considered
210                // a loop.
211                if (newVars.containsKey(visitedFlag)) {
212                    newVars.remove(visitedFlag);
213                }
214            }
215            newWfInstance.setAllVars(newVars);
216        }
