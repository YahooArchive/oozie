018    package org.apache.oozie.command.wf;
019    
020    import org.apache.oozie.client.WorkflowJob;
021    import org.apache.oozie.WorkflowJobBean;
022    import org.apache.oozie.command.Command;
023    import org.apache.oozie.command.CommandException;
024    import org.apache.oozie.store.StoreException;
025    import org.apache.oozie.store.WorkflowStore;
026    import org.apache.oozie.workflow.WorkflowException;
027    import org.apache.oozie.util.ParamChecker;
028    import org.apache.oozie.util.XLog;
029    
030    public class SuspendCommand extends Command<Void> {
031    
032        private String id;
033    
034        public SuspendCommand(String id) {
035            super("suspend", "suspend", 0, XLog.STD);
036            this.id = ParamChecker.notEmpty(id, "id");
037        }
038    
039        @Override
040        protected Void call(WorkflowStore store) throws StoreException, CommandException {
041            try {
042                WorkflowJobBean workflow = store.getWorkflow(id, true);
043                setLogInfo(workflow);
044                if (workflow.getStatus() == WorkflowJob.Status.RUNNING) {
045                    incrJobCounter(1);
046                    suspendJob(workflow, id);
047                    store.updateWorkflow(workflow);
048                    queueCallable(new NotificationCommand(workflow));
049                }
050                return null;
051            }
052            catch (WorkflowException ex) {
053                throw new CommandException(ex);
054            }
055        }
056    
057        public static void suspendJob(WorkflowJobBean workflow, String id) throws WorkflowException {
058            if (workflow.getStatus() == WorkflowJob.Status.RUNNING) {
059                workflow.getWorkflowInstance().suspend();
060                workflow.setStatus(WorkflowJob.Status.SUSPENDED);
061            }
062        }
063    }
