041    public class JobServlet extends JsonRestServlet {
042        private static final String INSTRUMENTATION_NAME = "job";
043    
044        private static final ResourceInfo RESOURCES_INFO[] = new ResourceInfo[1];
045    
046        static {
047            RESOURCES_INFO[0] = new ResourceInfo("*", Arrays.asList("PUT", "GET"), Arrays.asList(
048                    new ParameterInfo(RestConstants.ACTION_PARAM, String.class, true, Arrays.asList("PUT")),
049                    new ParameterInfo(RestConstants.JOB_SHOW_PARAM, String.class, false, Arrays.asList("GET"))));
050        }
051    
052        public JobServlet() {
053            super(INSTRUMENTATION_NAME, RESOURCES_INFO);
054        }
055    
056        /**
057         * Perform various job related actions - start, suspend, resume, kill, etc.
058         */
059        protected void doPut(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
060            String jobId = getResourceName(request);
061            request.setAttribute(AUDIT_PARAM, jobId);
062            request.setAttribute(AUDIT_OPERATION, request.getParameter(RestConstants.ACTION_PARAM));
063            try {
064                AuthorizationService auth = Services.get().get(AuthorizationService.class);
065                auth.authorizeForJob(getUser(request), jobId, true);
066            }
067            catch (AuthorizationException ex) {
068                throw new XServletException(HttpServletResponse.SC_UNAUTHORIZED, ex);
069            }
070    
071            DagEngine dagEngine =
072                    Services.get().get(DagEngineService.class).getDagEngine(getUser(request), getAuthToken(request));
073            try {
074                String action = request.getParameter(RestConstants.ACTION_PARAM);
075                if (action.equals(RestConstants.JOB_ACTION_START)) {
076                    stopCron();
077                    dagEngine.start(jobId);
078                    startCron();
079                    response.setStatus(HttpServletResponse.SC_OK);
080                }
081                else if (action.equals(RestConstants.JOB_ACTION_RESUME)) {
082                    stopCron();
083                    dagEngine.resume(jobId);
084                    startCron();
085                    response.setStatus(HttpServletResponse.SC_OK);
086                }
087                else if (action.equals(RestConstants.JOB_ACTION_SUSPEND)) {
088                    stopCron();
089                    dagEngine.suspend(jobId);
090                    startCron();
091                    response.setStatus(HttpServletResponse.SC_OK);
092                }
093                else if (action.equals(RestConstants.JOB_ACTION_KILL)) {
094                    stopCron();
095                    dagEngine.kill(jobId);
096                    startCron();
097                    response.setStatus(HttpServletResponse.SC_OK);
098                }
099                else if (action.equals(RestConstants.JOB_ACTION_RERUN)) {
100                    validateContentType(request, RestConstants.XML_CONTENT_TYPE);
101                    Configuration conf = new XConfiguration(request.getInputStream());
102    
103                    stopCron();
104    
105                    conf = XConfiguration.trim(conf);
106    
107                    JobsServlet.validateJobConfiguration(conf);
108    
109                    checkAuthorizationForApp(getUser(request),  conf);
110    
111                    dagEngine.reRun(jobId, conf);
112                    startCron();
113                    response.setStatus(HttpServletResponse.SC_OK);
114                }
115                else {
116                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0303,
117                                                RestConstants.ACTION_PARAM, action);
118                }
119            }
120            catch (DagEngineException ex) {
121                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
122            }
123        }
124    
125        /**
126         * Validate the configuration user/group.
127         * <p/>
128         *
129         * @param requestUser  user in request.
130         * @param conf         configuration.
131         * @throws XServletException thrown if the configuration does not have a property {@link org.apache.oozie.client.OozieClient#USER_NAME}.
132         */
133        static void checkAuthorizationForApp(String requestUser, Configuration conf) throws XServletException {
134            String user = conf.get(OozieClient.USER_NAME);
135            String group = conf.get(OozieClient.GROUP_NAME);
136            try {
137                if (user == null) {
138                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0401,
139                                                OozieClient.USER_NAME);
140                }
141                if (!requestUser.equals(UNDEF) && !user.equals(requestUser)) {
142                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0400, requestUser, user);
143                }
144                AuthorizationService auth = Services.get().get(AuthorizationService.class);
145                if (group == null) {
146                    group = auth.getDefaultGroup(user);
147                    conf.set(OozieClient.GROUP_NAME, group);
148                }
149                else {
150                    auth.authorizeForGroup(user, group);
151                }
152                XLog.Info.get().setParameter(XLogService.GROUP, group);
153                auth.authorizeForApp(user, group, conf.get(OozieClient.APP_PATH), conf);
154            }
155            catch (AuthorizationException ex) {
156                throw new XServletException(HttpServletResponse.SC_UNAUTHORIZED, ex);
157            }
158        }
159    
160        /**
161         * Return information about jobs.
162         */
163        public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
164            String jobId = getResourceName(request);
165            String show = request.getParameter(RestConstants.JOB_SHOW_PARAM);
166    
167            try {
168                AuthorizationService auth = Services.get().get(AuthorizationService.class);
169                auth.authorizeForJob(getUser(request), jobId, false);
170            }
171            catch (AuthorizationException ex) {
172                throw new XServletException(HttpServletResponse.SC_UNAUTHORIZED, ex);
173            }
174    
175            DagEngine dagEngine =
176                    Services.get().get(DagEngineService.class).getDagEngine(getUser(request), getAuthToken(request));
177            try {
178                if (show == null || show.equals(RestConstants.JOB_SHOW_INFO)) {
179                    stopCron();
180                    JsonWorkflowJob job = (JsonWorkflowJob) dagEngine.getJob(jobId);
181                    startCron();
182                    sendJsonResponse(response, HttpServletResponse.SC_OK, job);
183                }
184                else if (show.equals(RestConstants.JOB_SHOW_LOG)) {
185                    response.setContentType(TEXT_UTF8);
186                    dagEngine.streamLog(jobId, response.getWriter());
187                }
188                else if (show.equals(RestConstants.JOB_SHOW_DEFINITION)) {
189                    stopCron();
190                    response.setContentType(XML_UTF8);
191                    String wfDefinition = dagEngine.getDefinition(jobId);
192                    startCron();
193                    response.setStatus(HttpServletResponse.SC_OK);
194                    response.getWriter().write(wfDefinition);
195                }
196                else {
197                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0303,
198                                                RestConstants.JOB_SHOW_PARAM, show);
199                }
200            }
201            catch (DagEngineException ex) {
202                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
203            }
204        }
