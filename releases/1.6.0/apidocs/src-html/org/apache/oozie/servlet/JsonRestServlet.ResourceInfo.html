067        /**
068         * This bean defines a query string parameter.
069         */
070        public static class ParameterInfo {
071            private String name;
072            private Class type;
073            private List<String> methods;
074            private boolean required;
075    
076            /**
077             * Creates a ParameterInfo with querystring parameter definition.
078             *
079             * @param name     querystring parameter name.
080             * @param type     type for the parameter value, valid types are: <code>Integer, Boolean and String</code>
081             * @param required indicates if the parameter is required.
082             * @param methods  HTTP methods the parameter is used by.
083             */
084            public ParameterInfo(String name, Class type, boolean required, List<String> methods) {
085                this.name = ParamChecker.notEmpty(name, "name");
086                if (type != Integer.class && type != Boolean.class && type != String.class) {
087                    throw new IllegalArgumentException("Type must be integer, boolean or string");
088                }
089                this.type = ParamChecker.notNull(type, "type");
090                this.required = required;
091                this.methods = ParamChecker.notNull(methods, "methods");
092            }
093    
094        }

096        /**
097         * This bean defines a REST resource.
098         */
099        public static class ResourceInfo {
100            private String name;
101            private boolean wildcard;
102            private List<String> methods;
103            private Map<String, ParameterInfo> parameters = new HashMap<String, ParameterInfo>();
104    
105            /**
106             * Creates a ResourceInfo with a REST resource definition.
107             *
108             * @param name       name of the REST resource, it can be an fixed resource name, empty or a wildcard ('*').
109             * @param methods    HTTP methods supported by the resource.
110             * @param parameters parameters supported by the resource.
111             */
112            public ResourceInfo(String name, List<String> methods, List<ParameterInfo> parameters) {
113                this.name = name;
114                wildcard = name.equals("*");
115                for (ParameterInfo parameter : parameters) {
116                    this.parameters.put(parameter.name, parameter);
117                }
118                this.methods = ParamChecker.notNull(methods, "methods");
119            }
120        }

139        /**
140         * Creates a servlet with a specified instrumentation sampler name for its requests.
141         *
142         * @param instrumentationName instrumentation name for timer and samplers for the servlet.
143         * @param resourcesInfo       list of resource definitions supported by the servlet, empty and wildcard resources
144         *                            must be the last ones, in that order, first empty and the wildcard.
145         */
146        public JsonRestServlet(String instrumentationName, ResourceInfo... resourcesInfo) {
147            this.instrumentationName = ParamChecker.notEmpty(instrumentationName, "instrumentationName");
148            if (resourcesInfo.length == 0) {
149                throw new IllegalArgumentException("There must be at least one ResourceInfo");
150            }
151            this.resourcesInfo = Arrays.asList(resourcesInfo);
152            auditLog = XLog.getLog("oozieaudit");
153            auditLog.setMsgPrefix("");
154        }

165        /**
166         * Define an instrumentation sampler.
167         * <p/>
168         * Sampling period is 60 seconds, the sampling frequency is 1 second.
169         * <p/>
170         * The instrumentation group used is {@link #INSTRUMENTATION_GROUP}.
171         *
172         * @param samplerName    sampler name.
173         * @param samplerCounter sampler counter.
174         */
175        private void defineSampler(String samplerName, final AtomicLong samplerCounter) {
176            instrumentation.addSampler(INSTRUMENTATION_GROUP, samplerName, 60, 1, new Instrumentation.Variable<Long>() {
177                public Long getValue() {
178                    return samplerCounter.get();
179                }
180            });
181        }

207        /**
208         * Initializes total request and servlet request samplers.
209         */
210        public void init(ServletConfig servletConfig) throws ServletException {
211            super.init(servletConfig);
212            instrumentation = Services.get().get(InstrumentationService.class).get();
213            synchronized (JsonRestServlet.class) {
214                if (TOTAL_REQUESTS_SAMPLER_COUNTER == null) {
215                    TOTAL_REQUESTS_SAMPLER_COUNTER = new AtomicLong();
216                    defineSampler(INSTR_TOTAL_REQUESTS_SAMPLER, TOTAL_REQUESTS_SAMPLER_COUNTER);
217                }
218            }
219            defineSampler(instrumentationName, samplerCounter);
220        }

228        private void incrCounter(String name, int count) {
229            if (instrumentation != null) {
230                instrumentation.incr(INSTRUMENTATION_GROUP, name, count);

234        /**
235         * Logs audit information for write requests to the audit log.
236         * @param request the http request.
237         */
238        private void logAuditInfo(HttpServletRequest request) {
239            if (request.getAttribute(AUDIT_OPERATION) != null) {
240                Integer httpStatusCode = (Integer) request.getAttribute(AUDIT_HTTP_STATUS_CODE);
241                httpStatusCode = (httpStatusCode != null) ? httpStatusCode : HttpServletResponse.SC_OK;
242                String status = (httpStatusCode == HttpServletResponse.SC_OK) ? "SUCCESS" : "FAILED";
243                String operation = (String) request.getAttribute(AUDIT_OPERATION);
244                String param = (String) request.getAttribute(AUDIT_PARAM);
245                String user = XLog.Info.get().getParameter(XLogService.USER);
246                String group = XLog.Info.get().getParameter(XLogService.GROUP);
247                String jobId = XLog.Info.get().getParameter(DagXLogInfoService.JOB);
248                String app = XLog.Info.get().getParameter(DagXLogInfoService.APP);
249    
250                String errorCode = (String) request.getAttribute(AUDIT_ERROR_CODE);
251                String errorMessage = (String) request.getAttribute(AUDIT_ERROR_MESSAGE);
252    
253                auditLog.info(
254                        "USER [{0}], GROUP [{1}], APP [{2}], JOBID [{3}], OPERATION [{4}], PARAMETER [{5}], STATUS [{6}], HTTPCODE [{7}], ERRORCODE [{8}], ERRORMESSAGE [{9}]",
255                        user, group, app, jobId, operation, param, status, httpStatusCode, errorCode, errorMessage);
256            }
257        }

259        /**
260         * Dispatches to super after loginfo and intrumentation handling. In case of errors dispatches error
261         * response codes and does error logging.
262         */
263        @SuppressWarnings("unchecked")
264        protected final void service(HttpServletRequest request, HttpServletResponse response)
265                throws ServletException, IOException {
266            if (Services.get().isSafeMode() && !request.getMethod().equals("GET") && !allowSafeModeChanges) {
267                sendErrorResponse(response, HttpServletResponse.SC_SERVICE_UNAVAILABLE,
268                                  ErrorCode.E0002.toString(),
269                                  ErrorCode.E0002.getTemplate());
270            }
271            Instrumentation.Cron cron = new Instrumentation.Cron();
272            requestCron.set(cron);
273            try {
274                cron.start();
275                validateRestUrl(request.getMethod(), getResourceName(request), request.getParameterMap());
276                XLog.Info.get().clear();
277                String user = getUser(request);
278                XLog.Info.get().setParameter(XLogService.USER, user);
279                TOTAL_REQUESTS_SAMPLER_COUNTER.incrementAndGet();
280                samplerCounter.incrementAndGet();
281                super.service(request, response);
282            }
283            catch (XServletException ex) {
284                XLog log = XLog.getLog(getClass());
285                log.warn("URL[{0} {1}] error[{2}], {3}", request.getMethod(), getRequestUrl(request), ex.getErrorCode(),
286                         ex.getMessage(), ex);
287                request.setAttribute(AUDIT_ERROR_MESSAGE, ex.getMessage());
288                request.setAttribute(AUDIT_ERROR_CODE, ex.getErrorCode().toString());
289                request.setAttribute(AUDIT_HTTP_STATUS_CODE, ex.getHttpStatusCode());
290                incrCounter(INSTR_TOTAL_FAILED_REQUESTS_COUNTER, 1);
291                sendErrorResponse(response, ex.getHttpStatusCode(), ex.getErrorCode().toString(), ex.getMessage());
292            }
293            catch (RuntimeException ex) {
294                XLog log = XLog.getLog(getClass());
295                log.error("URL[{0} {1}] error, {2}", request.getMethod(), getRequestUrl(request), ex.getMessage(), ex);
296                incrCounter(INSTR_TOTAL_FAILED_REQUESTS_COUNTER, 1);
297                throw ex;
298            }
299            finally {
300                logAuditInfo(request);
301                TOTAL_REQUESTS_SAMPLER_COUNTER.decrementAndGet();
302                incrCounter(INSTR_TOTAL_REQUESTS_COUNTER, 1);
303                samplerCounter.decrementAndGet();
304                XLog.Info.remove();
305                cron.stop();
306                //TODO
307                incrCounter(instrumentationName, 1);
308                incrCounter(instrumentationName + "-" + request.getMethod(), 1);
309                addCron(instrumentationName, cron);
310                addCron(instrumentationName + "-" + request.getMethod(), cron);
311                requestCron.remove();
312            }
313        }

315        private String getRequestUrl(HttpServletRequest request) {
316            StringBuffer url = request.getRequestURL();
317            if (request.getQueryString() != null) {
318                url.append("?").append(request.getQueryString());
319            }
320            return url.toString();
321        }

326         * @param response   servlet response.
327         * @param statusCode HTTP status code.
328         * @param bean       bean to send as JSON response.
329         * @throws java.io.IOException thrown if the bean could not be serialized to the response output stream.

332            response.setStatus(statusCode);

334            bean.toJSONObject().writeJSONString(response.getWriter());

339         *
340         * @param response   servlet response.
341         * @param statusCode HTTP status code.
342         * @param error  error code.
343         * @param message error message.
344         * @throws java.io.IOException thrown if the error response could not be set.

347                throws IOException {
348            response.setHeader(RestConstants.OOZIE_ERROR_CODE, error);
349            response.setHeader(RestConstants.OOZIE_ERROR_MESSAGE, message);
350            response.sendError(statusCode);
351        }

354        protected void sendJsonResponse(HttpServletResponse response, int statusCode, JSONStreamAware json)
355                throws IOException {
356            if (statusCode == HttpServletResponse.SC_OK || statusCode == HttpServletResponse.SC_CREATED) {
357                response.setStatus(statusCode);
358            }
359            else {
360                response.sendError(statusCode);
361            }
362            response.setStatus(statusCode);
363            response.setContentType(JSTON_UTF8);
364            json.writeJSONString(response.getWriter());
365        }

367        /**
368         * Validates REST URL using the ResourceInfos of the servlet.
369         *
370         * @param method            HTTP method.
371         * @param resourceName      resource name.
372         * @param queryStringParams query string parameters.
373         * @throws javax.servlet.ServletException thrown if the resource name or parameters are incorrect.
374         */
375        @SuppressWarnings("unchecked")
376        protected void validateRestUrl(String method, String resourceName, Map<String, String[]> queryStringParams)
377                throws ServletException {
378    
379            if (resourceName.contains("/")) {
380                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0301, resourceName);
381            }
382    
383            boolean valid = false;
384            for (int i = 0; !valid && i < resourcesInfo.size(); i++) {
385                ResourceInfo resourceInfo = resourcesInfo.get(i);
386                if (resourceInfo.name.equals(resourceName) || resourceInfo.wildcard) {
387                    if (!resourceInfo.methods.contains(method)) {
388                        throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0301, resourceName);
389                    }
390                    for (Map.Entry<String, String[]> entry : queryStringParams.entrySet()) {
391                        String name = entry.getKey();
392                        ParameterInfo parameterInfo = resourceInfo.parameters.get(name);
393                        if (parameterInfo != null) {
394                            if (!parameterInfo.methods.contains(method)) {
395                                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0302, name);
396                            }
397                            String value = entry.getValue()[0].trim();
398                            if (parameterInfo.type.equals(Boolean.class)) {
399                                value = value.toLowerCase();
400                                if (!value.equals("true") && !value.equals("false")) {
401                                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0304, name,
402                                                                "boolean");
403                                }
404                            }
405                            if (parameterInfo.type.equals(Integer.class)) {
406                                try {
407                                    Integer.parseInt(value);
408                                }
409                                catch (NumberFormatException ex) {
410                                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0304, name,
411                                                                "integer");
412                                }
413                            }
414                        }
415                    }
416                    for (ParameterInfo parameterInfo : resourceInfo.parameters.values()) {
417                        if (parameterInfo.methods.contains(method) && parameterInfo.required &&
418                            queryStringParams.get(parameterInfo.name) == null) {
419                            throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0305,
420                                                        parameterInfo.name);
421                        }
422                    }
423                    valid = true;
424                }
425            }
426            if (!valid) {
427                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0301, resourceName);
428            }
429        }

431        /**
432         * Return the resource name of the request.
433         * <p/>
434         * The resource name is the whole extra path. If the extra path starts with '/', the first '/' is trimmed.
435         *
436         * @param request request instance
437         * @return the resource name, <code>null</code> if none.
438         */
439        protected String getResourceName(HttpServletRequest request) {
440            String requestPath = request.getPathInfo();
441            if (requestPath != null) {
442                while (requestPath.startsWith("/")) {
443                    requestPath = requestPath.substring(1);
444                }
445                requestPath = requestPath.trim();
446            }
447            else {
448                requestPath = "";
449            }
450            return requestPath;
451        }

453        /**
454         * Return the request content type, lowercase and without attributes.
455         *
456         * @param request servlet request.
457         * @return the request content type, <code>null</code> if none.
458         */
459        protected String getContentType(HttpServletRequest request) {
460            String contentType = request.getContentType();
461            if (contentType != null) {
462                int index = contentType.indexOf(";");
463                if (index > -1) {
464                    contentType = contentType.substring(0, index);
465                }
466                contentType = contentType.toLowerCase();
467            }
468            return contentType;
469        }
470    
471        /**
472         * Validate and return the content type of the request.
473         *
474         * @param request  servlet request.
475         * @param expected expected contentType.
476         * @return the normalized content type (lowercase and without modifiers).
477         * @throws XServletException thrown if the content type is invalid.
478         */
479        protected String validateContentType(HttpServletRequest request, String expected) throws XServletException {
480            String contentType = getContentType(request);
481            if (contentType == null || contentType.trim().length() == 0) {
482                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0300, contentType);
483            }
484            if (!contentType.equals(expected)) {
485                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0300, contentType);
486            }
487            return contentType;
488        }

504         *
505         * @param request request.
506         * @return the authentication token, <code>null</code> if there is none.
507         */
508        protected String getAuthToken(HttpServletRequest request) {
509            String authToken = (String) request.getAttribute(AUTH_TOKEN);
510            return (authToken != null) ? authToken : UNDEF;
511        }

520            String userName = (String) request.getAttribute(USER_NAME);
