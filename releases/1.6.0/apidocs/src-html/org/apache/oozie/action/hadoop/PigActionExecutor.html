018    package org.apache.oozie.action.hadoop;
019    
020    import org.apache.hadoop.conf.Configuration;
021    import org.apache.hadoop.fs.Path;
022    import org.apache.oozie.action.ActionExecutorException;
023    import org.apache.oozie.client.WorkflowAction;
024    import org.apache.oozie.util.XmlUtils;
025    import org.jdom.Element;
026    import org.jdom.Namespace;
027    import org.jdom.JDOMException;
028    
029    import java.util.List;
030    
031    public class PigActionExecutor extends JavaActionExecutor {
032    
033        public PigActionExecutor() {
034            super("pig");
035        }
036    
037        protected List<Class> getLauncherClasses() {
038            List<Class> classes = super.getLauncherClasses();
039            classes.add(LauncherMain.class);
040            classes.add(MapReduceMain.class);
041            classes.add(PigMain.class);
042            return classes;
043        }
044    
045        protected String getLauncherMain(Configuration launcherConf, Element actionXml) {
046            return launcherConf.get(LauncherMapper.CONF_OOZIE_ACTION_MAIN_CLASS, PigMain.class.getName());
047        }
048    
049        void injectActionCallback(Context context, Configuration launcherConf) {
050        }
051    
052        Configuration setupLauncherConf(Configuration conf, Element actionXml, Path appPath)
053                throws ActionExecutorException {
054            super.setupLauncherConf(conf, actionXml, appPath);
055            Namespace ns = actionXml.getNamespace();
056            String script = actionXml.getChild("script", ns).getTextTrim();
057            String pigName = new Path(script).getName();
058            addToCache(conf, appPath, script + "#" + pigName, false);
059            return conf;
060        }
061    
062        @SuppressWarnings("unchecked")
063        Configuration setupActionConf(Configuration actionConf, Context context, Element actionXml, Path appPath)
064                throws ActionExecutorException {
065            super.setupActionConf(actionConf, context, actionXml, appPath);
066            Namespace ns = actionXml.getNamespace();
067    
068            String script = actionXml.getChild("script", ns).getTextTrim();
069            String pigName = new Path(script).getName();
070    
071            List<Element> params = (List<Element>) actionXml.getChildren("param", ns);
072            String[] strParams = new String[params.size()];
073            for (int i = 0; i < params.size(); i++) {
074                strParams[i] = params.get(i).getTextTrim();
075            }
076    
077            PigMain.setPigScript(actionConf, pigName, strParams);
078            return actionConf;
079        }
080    
081        protected boolean getCaptureOutput(WorkflowAction action) throws JDOMException {
082            return true;
083        }
084    
085    }
