018    package org.apache.oozie.action.hadoop;
019    
020    import org.apache.hadoop.conf.Configuration;
021    import org.apache.hadoop.fs.Path;
022    import org.apache.hadoop.fs.FileSystem;
023    import org.apache.hadoop.mapred.JobClient;
024    import org.apache.hadoop.mapred.JobConf;
025    import org.apache.hadoop.mapred.RunningJob;
026    import org.apache.hadoop.security.UserGroupInformation;
027    
028    import java.util.Map;
029    import java.util.Properties;
030    import java.io.ByteArrayOutputStream;
031    import java.io.IOException;
032    import java.io.FileOutputStream;
033    import java.io.OutputStream;
034    import java.io.File;
035    
036    public class MapReduceMain extends LauncherMain {
037    
038        public static void main(String[] args) throws Exception {
039            run(MapReduceMain.class, args);
040        }
041    
042        protected void run(String[] args) throws Exception {
043            System.out.println();
044            System.out.println("Oozie Map-Reduce action configuration");
045            System.out.println("=======================");
046    
047            //loading action conf prepared by Oozie
048            Configuration actionConf = new Configuration(false);
049            actionConf.addResource(new Path("file:///", System.getProperty("oozie.action.conf.xml")));
050    
051            System.out.println("Oozie Map-Reduce Configuration: ");
052            System.out.println("------------------------");
053            ByteArrayOutputStream baos = new ByteArrayOutputStream();
054            actionConf.writeXml(baos);
055            baos.close();
056            System.out.println(new String(baos.toByteArray()));
057            System.out.println("------------------------");
058            System.out.println();
059    
060    
061    
062            System.out.println("Submitting Oozie action Map-Reduce job");
063            System.out.println();
064            //submitting job
065            RunningJob runningJob = submitJob(actionConf);
066    
067            //propagating job id back to Oozie
068            String jobId = runningJob.getID().toString();
069            Properties props = new Properties();
070            props.setProperty("id", jobId);
071            File idFile = new File(System.getProperty("oozie.action.newId.properties"));
072            OutputStream os = new FileOutputStream(idFile);
073            props.store(os, "");
074            os.close();
075    
076            System.out.println("=======================");
077            System.out.println();
078        }
079    
080        protected void addActionConf(JobConf jobConf, Configuration actionConf) {
081            for (Map.Entry<String, String> entry : actionConf) {
082                jobConf.set(entry.getKey(), entry.getValue());
083            }
084        }
085    
086        protected RunningJob submitJob(Configuration actionConf) throws Exception {
087            JobConf jobConf = new JobConf();
088            addActionConf(jobConf, actionConf);
089            
090            //propagate delegation related props from launcher job to MR job
091            if (System.getenv("HADOOP_TOKEN_FILE_LOCATION") != null) {
092                jobConf.set("mapreduce.job.credentials.binary", System.getenv("HADOOP_TOKEN_FILE_LOCATION"));
093            }
094            
095            JobClient jobClient = createJobClient(jobConf);
096            return jobClient.submitJob(jobConf);
097        }
098    
099        @SuppressWarnings("unchecked")
100        protected JobClient createJobClient(JobConf jobConf) throws IOException {
101            return new JobClient(jobConf);
102        }
103    
104        // allows any character in the value, the conf.setStrings() does not allow commas
105        public static void setStrings(Configuration conf, String key, String[] values) {
106            if (values != null) {
107                conf.setInt(key + ".size", values.length);
108                for (int i = 0; i < values.length; i++) {
109                    conf.set(key + "." + i, values[i]);
110                }
111            }
112        }
113    
114        public static String[] getStrings(Configuration conf, String key) {
115            String[] values = new String[conf.getInt(key + ".size", 0)];
116            for (int i = 0; i < values.length; i++) {
117                values[i] = conf.get(key + "." + i);
118            }
119            return values;
120        }
121    
122    }
