018    package org.apache.oozie.action.hadoop;
019    
020    import java.util.List;
021    
022    import org.apache.hadoop.conf.Configuration;

037    
038    public class MapReduceActionExecutor extends JavaActionExecutor {
039    
040        public static final String HADOOP_COUNTERS = "hadoop.counters";
041    
042        public MapReduceActionExecutor() {
043            super("map-reduce");
044        }
045    
046        protected List<Class> getLauncherClasses() {
047            List<Class> classes = super.getLauncherClasses();
048            classes.add(LauncherMain.class);
049            classes.add(MapReduceMain.class);
050            classes.add(StreamingMain.class);
051            classes.add(PipesMain.class);
052            return classes;
053        }
054    
055        protected String getLauncherMain(Configuration launcherConf, Element actionXml) {
056            String mainClass;
057            Namespace ns = actionXml.getNamespace();
058            if (actionXml.getChild("streaming", ns) != null) {
059                mainClass = launcherConf.get(LauncherMapper.CONF_OOZIE_ACTION_MAIN_CLASS, StreamingMain.class.getName());
060            }
061            else if (actionXml.getChild("pipes", ns) != null) {
062                mainClass = launcherConf.get(LauncherMapper.CONF_OOZIE_ACTION_MAIN_CLASS, PipesMain.class.getName());
063            }
064            else {
065                mainClass = launcherConf.get(LauncherMapper.CONF_OOZIE_ACTION_MAIN_CLASS, MapReduceMain.class.getName());
066            }
067            return mainClass;
068        }
069    
070        Configuration setupLauncherConf(Configuration conf, Element actionXml, Path appPath)
071                throws ActionExecutorException {
072            super.setupLauncherConf(conf, actionXml, appPath);
073            conf.setBoolean("mapreduce.job.complete.cancel.delegation.tokens", true);
074            return conf;
075        }
076    
077        @SuppressWarnings("unchecked")
078        Configuration setupActionConf(Configuration actionConf, Context context, Element actionXml, Path appPath)
079                throws ActionExecutorException {
080            Namespace ns = actionXml.getNamespace();
081            if (actionXml.getChild("streaming", ns) != null) {
082                Element streamingXml = actionXml.getChild("streaming", ns);
083                String mapper = streamingXml.getChildTextTrim("mapper", ns);
084                String reducer = streamingXml.getChildTextTrim("reducer", ns);
085                String recordReader = streamingXml.getChildTextTrim("record-reader", ns);
086                List<Element> list = (List<Element>) streamingXml.getChildren("record-reader-mapping", ns);
087                String[] recordReaderMapping = new String[list.size()];
088                for (int i = 0; i < list.size(); i++) {
089                    recordReaderMapping[i] = list.get(i).getTextTrim();
090                }
091                list = (List<Element>) streamingXml.getChildren("env", ns);
092                String[] env = new String[list.size()];
093                for (int i = 0; i < list.size(); i++) {
094                    env[i] = list.get(i).getTextTrim();
095                }
096                StreamingMain.setStreaming(actionConf, mapper, reducer, recordReader, recordReaderMapping, env);
097            }
098            else if (actionXml.getChild("pipes", ns) != null) {
099                Element pipesXml = actionXml.getChild("pipes", ns);
100                String map = pipesXml.getChildTextTrim("map", ns);
101                String reduce = pipesXml.getChildTextTrim("reduce", ns);
102                String inputFormat = pipesXml.getChildTextTrim("inputformat", ns);
103                String partitioner = pipesXml.getChildTextTrim("partitioner", ns);
104                String writer = pipesXml.getChildTextTrim("writer", ns);
105                String program = pipesXml.getChildTextTrim("program", ns);
106                PipesMain.setPipes(actionConf, map, reduce, inputFormat, partitioner, writer, program);
107            }
108            actionConf = super.setupActionConf(actionConf, context, actionXml, appPath);
109            return actionConf;
110        }
111    
112        @Override
113        public void end(Context context, WorkflowAction action) throws ActionExecutorException {
114            super.end(context, action);
115            try {
116                if (action.getStatus() == WorkflowAction.Status.OK) {
117                    Element actionXml = XmlUtils.parseXml(action.getConf());
118                    Configuration conf = createBaseHadoopConf(context, actionXml);
119                    JobConf jobConf = new JobConf();
120                    XConfiguration.copy(conf, jobConf);
121                    JobClient jobClient = createJobClient(context, jobConf);
122                    RunningJob runningJob = jobClient.getJob(JobID.forName(action.getExternalId()));
123                    // TODO this has to be done in a better way
124                    if (!runningJob.getJobName().startsWith("oozie:action:")) {
125                        throw new ActionExecutorException(ActionExecutorException.ErrorType.FAILED, "MR001",
126                                "ID swap should have happened in launcher job [{0}]", action.getExternalId());
127                    }
128                    Counters counters = runningJob.getCounters();
129                    if (counters != null) {
130                        JSONObject json = counterstoJson(counters);
131                        context.setVar(HADOOP_COUNTERS, json.toJSONString());
132                    }
133                    else {
134    
135                        context.setVar(HADOOP_COUNTERS, "");
136    
137                        XLog.getLog(getClass()).warn("Could not find Hadoop Counters for: [{0}]", action.getExternalId());
138                    }
139                }
140            }
141            catch (Exception ex) {
142                throw convertException(ex);
143            }
144        }
145    
146        @SuppressWarnings("unchecked")
147        private JSONObject counterstoJson(Counters counters) {
148    
149            if (counters == null) {
150                return null;
151            }
152    
153            JSONObject groups = new JSONObject();
154            for (String gName : counters.getGroupNames()) {
155                JSONObject group = new JSONObject();
156                for (Counters.Counter counter : counters.getGroup(gName)) {
157                    String cName = counter.getName();
158                    Long cValue = counter.getCounter();
159                    group.put(cName, cValue);
160                }
161                groups.put(gName, group);
162            }
163            return groups;
164        }
