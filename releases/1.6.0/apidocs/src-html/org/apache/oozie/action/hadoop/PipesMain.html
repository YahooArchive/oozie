018    package org.apache.oozie.action.hadoop;
019    
020    import org.apache.hadoop.conf.Configuration;
021    import org.apache.hadoop.mapred.JobConf;
022    import org.apache.hadoop.mapred.RunningJob;
023    import org.apache.hadoop.mapred.pipes.Submitter;
024    import org.apache.hadoop.filecache.DistributedCache;
025    
026    public class PipesMain extends MapReduceMain {
027    
028        public static void main(String[] args) throws Exception {
029            run(PipesMain.class, args);
030        }
031    
032        protected RunningJob submitJob(Configuration actionConf) throws Exception {
033            JobConf jobConf = new JobConf();
034    
035            String value = actionConf.get("oozie.pipes.map");
036            if (value != null) {
037                jobConf.setBoolean("hadoop.pipes.java.mapper", true);
038                jobConf.set("mapred.mapper.class", value);
039            }
040            value = actionConf.get("oozie.pipes.reduce");
041            if (value != null) {
042                jobConf.setBoolean("hadoop.pipes.java.reducer", true);
043                jobConf.set("mapred.reducer.class", value);
044            }
045            value = actionConf.get("oozie.pipes.inputformat");
046            if (value != null) {
047                jobConf.setBoolean("hadoop.pipes.java.recordreader", true);
048                jobConf.set("mapred.input.format.class", value);
049            }
050            value = actionConf.get("oozie.pipes.partitioner");
051            if (value != null) {
052                jobConf.set("mapred.partitioner.class", value);
053            }
054            value = actionConf.get("oozie.pipes.writer");
055            if (value != null) {
056                jobConf.setBoolean("hadoop.pipes.java.recordwriter", true);
057                 jobConf.set("mapred.output.format.class", value);
058            }
059            value = actionConf.get("oozie.pipes.program");
060            if (value != null) {
061                jobConf.set("hadoop.pipes.executable", value);
062                if (value.contains("#")) {
063                    DistributedCache.createSymlink(jobConf);
064                }
065            }
066    
067            addActionConf(jobConf, actionConf);
068    
069            //propagate delegation related props from launcher job to MR job
070            if (System.getenv("HADOOP_TOKEN_FILE_LOCATION") != null) {
071                jobConf.set("mapreduce.job.credentials.binary", System.getenv("HADOOP_TOKEN_FILE_LOCATION"));
072            }
073            
074            return Submitter.jobSubmit(jobConf);
075        }
076    
077        public static void setPipes(Configuration conf, String map, String reduce, String inputFormat, String partitioner,
078                                    String writer, String program) {
079            if (map != null) {
080                conf.set("oozie.pipes.map", map);
081            }
082            if (reduce != null) {
083                conf.set("oozie.pipes.reduce", reduce);
084            }
085            if (inputFormat != null) {
086                conf.set("oozie.pipes.inputformat", inputFormat);
087            }
088            if (partitioner != null) {
089                conf.set("oozie.pipes.partitioner", partitioner);
090            }
091            if (writer != null) {
092                conf.set("oozie.pipes.writer", writer);
093            }
094            if (program != null) {
095                conf.set("oozie.pipes.program", program);
096            }
097        }
098    
099    }
