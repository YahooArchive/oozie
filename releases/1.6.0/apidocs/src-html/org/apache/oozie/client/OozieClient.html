051    /**
052     * Client API to submit and manage Oozie workflow jobs against an Oozie intance.
053     * <p/>
054     * This class is thread safe.
055     * <p/>
056     * Syntax for filter for the {@link #getJobsInfo(String)}  {@link #getJobsInfo(String, int, int)}  methods:
057     * <code>[NAME=VALUE][;NAME=VALUE]*</code>.
058     * <p/>
059     * Valid filter names are:
060     * <p/>
061     * <ul/>
062     * <li>name: the workflow application name from the workflow definition.</li>
063     * <li>user: the user that submitted the job.</li>
064     * <li>group: the group for the job.</li>
065     * <li>status: the status of the job.</li>
066     * </ul>
067     * <p/>
068     * The query will do an AND among all the filter names.
069     * The query will do an OR among all the filter values for the same name. Multiple values must be specified as
070     * different name value pairs.
071     */

169        /**
170         * Validate that the Oozie client and server instances are protocol compatible.
171         *
172         * @throws OozieClientException thrown in the client and the server are not protocol compatible.
173         */
174        public synchronized void validateWSVersion() throws OozieClientException {
175            if (!validatedVersion) {
176                try {
177                    URL url = new URL(baseUrl + RestConstants.VERSIONS);
178                    HttpURLConnection conn = createConnection(url, "GET");
179                    if (conn.getResponseCode() == HttpURLConnection.HTTP_OK) {
180                        JSONArray array = (JSONArray) JSONValue.parse(new InputStreamReader(conn.getInputStream()));
181                        if (!array.contains(WS_PROTOCOL_VERSION)) {
182                            StringBuilder msg = new StringBuilder();
183                            msg.append("Unsupported version [").append(WS_PROTOCOL_VERSION)
184                                    .append("], supported versions[");
185                            String separator = "";
186                            for (Object version : array) {
187                                msg.append(separator).append(version);
188                            }
189                            msg.append("]");
190                            throw new OozieClientException(OozieClientException.UNSUPPORTED_VERSION, msg.toString());
191                        }
192                    }
193                    else {
194                        handleError(conn);
195                    }
196                }
197                catch (IOException ex) {
198                    throw new OozieClientException(OozieClientException.IO_ERROR, ex);
199                }
200                protocolUrl =  baseUrl + "v" + WS_PROTOCOL_VERSION + "/";
201                validatedVersion = true;
202            }
203        }

254        private URL createURL(String collection, String resource, Map<String, String> parameters)
255                throws IOException, OozieClientException {
256            validateWSVersion();
257            StringBuilder sb = new StringBuilder();
258            sb.append(protocolUrl).append(collection);
259            if (resource != null && resource.length() > 0) {
260                sb.append("/").append(resource);
261            }
262            if (parameters.size() > 0) {
263                String separator = "?";
264                for (Map.Entry<String, String> param : parameters.entrySet()) {
265                    if (param.getValue() != null) {
266                        sb.append(separator).append(URLEncoder.encode(param.getKey(), "UTF-8")).append("=")
267                                .append(URLEncoder.encode(param.getValue(), "UTF-8"));
268                        separator = "&";
269                    }
270                }
271            }
272            return new URL(sb.toString());
273        }

275        private HttpURLConnection createConnection(URL url, String method) throws IOException, OozieClientException {
276            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
277            conn.setRequestMethod(method);
278            if (method.equals("POST") || method.equals("PUT")) {
279                conn.setDoOutput(true);
280            }
281            for (Map.Entry<String, String> header : headers.entrySet()) {
282                conn.setRequestProperty(header.getKey(), header.getValue());
283            }
284            return conn;
285        }

287        private abstract class ClientCallable<T> implements Callable<T> {
288            private String method;
289            private String collection;
290            private String resource;
291            private Map<String, String> params;
292    
293            public ClientCallable(String method, String collection, String resource, Map<String, String> params) {
294                this.method = method;
295                this.collection = collection;
296                this.resource = resource;
297                this.params = params;
298            }
299    
300            public T call() throws OozieClientException {
301                try {
302                    URL url = createURL(collection, resource, params);
303                    HttpURLConnection conn = createConnection(url, method);
304                    return call(conn);
305                }
306                catch (IOException ex) {
307                    throw new OozieClientException(OozieClientException.IO_ERROR, ex);
308                }
309    
310            }
311    
312            protected abstract T call(HttpURLConnection conn) throws IOException, OozieClientException;
313        }
314    
315        private static void handleError(HttpURLConnection conn) throws IOException, OozieClientException {
316            int status = conn.getResponseCode();
317            String error = conn.getHeaderField(RestConstants.OOZIE_ERROR_CODE);
318            String message = conn.getHeaderField(RestConstants.OOZIE_ERROR_MESSAGE);
319    
320            if(error == null) {
321                error = "HTTP error code: " + status;
322            }
323    
324            if(message == null){
325                message = conn.getResponseMessage();
326            }
327            throw new OozieClientException(error, message);
328        }

330        private static Map<String, String> prepareParams(String... params) {
331            Map<String, String> map = new HashMap<String, String>();
332            for (int i = 0; i < params.length; i = i + 2) {
333                map.put(params[i], params[i + 1]);
334            }
335            return map;
336        }

338        public void writeToXml(Properties props, OutputStream out) throws IOException {
339            try {
340                Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder().newDocument();
341                Element conf = doc.createElement("configuration");
342                doc.appendChild(conf);
343                conf.appendChild(doc.createTextNode("\n"));
344                for (Enumeration e = props.keys(); e.hasMoreElements();) {
345                    String name = (String) e.nextElement();
346                    Object object = props.get(name);
347                    String value;
348                    if (object instanceof String) {
349                        value = (String) object;
350                    }
351                    else {
352                        continue;
353                    }
354                    Element propNode = doc.createElement("property");
355                    conf.appendChild(propNode);
356    
357                    Element nameNode = doc.createElement("name");
358                    nameNode.appendChild(doc.createTextNode(name));
359                    propNode.appendChild(nameNode);
360    
361                    Element valueNode = doc.createElement("value");
362                    valueNode.appendChild(doc.createTextNode(value));
363                    propNode.appendChild(valueNode);
364    
365                    conf.appendChild(doc.createTextNode("\n"));
366                }
367    
368                DOMSource source = new DOMSource(doc);
369                StreamResult result = new StreamResult(out);
370                TransformerFactory transFactory = TransformerFactory.newInstance();
371                Transformer transformer = transFactory.newTransformer();
372                transformer.transform(source, result);
373            }
374            catch (Exception e) {
375                throw new IOException(e);
376            }
377        }

379        private class JobSubmit extends ClientCallable<String> {
380            private Properties conf;
381    
382            JobSubmit(Properties conf, boolean start) {
383                super("POST", RestConstants.JOBS, "", (start) ? prepareParams(RestConstants.ACTION_PARAM,
384                                                                              RestConstants.JOB_ACTION_START)
385                                                              : prepareParams());
386                this.conf = notNull(conf, "conf");
387            }
388    
389            JobSubmit(String jobId, Properties conf) {
390                super("PUT", RestConstants.JOB, notEmpty(jobId, "jobId"),
391                      prepareParams(RestConstants.ACTION_PARAM, RestConstants.JOB_ACTION_RERUN));
392                this.conf = notNull(conf, "conf");
393            }
394    
395            protected String call(HttpURLConnection conn) throws IOException, OozieClientException {
396                conn.setRequestProperty("content-type", RestConstants.XML_CONTENT_TYPE);
397                writeToXml(conf, conn.getOutputStream());
398                if (conn.getResponseCode() == HttpURLConnection.HTTP_CREATED) {
399                    JSONObject json = (JSONObject) JSONValue.parse(new InputStreamReader(conn.getInputStream()));
400                    return (String) json.get(JsonTags.JOB_ID);
401                }
402                if (conn.getResponseCode() != HttpURLConnection.HTTP_OK) {
403                    handleError(conn);
404                }
405                return null;
406            }
407        }
