018    package org.apache.oozie;
019    
020    import org.apache.oozie.client.OozieClientException;
021    import org.apache.oozie.client.WorkflowJob;
022    import org.apache.oozie.client.OozieClient;
023    import org.apache.oozie.util.XConfiguration;
024    import org.apache.oozie.service.Services;
025    
026    import java.util.Collections;
027    import java.util.Iterator;
028    import java.util.List;
029    import java.util.Properties;
030    
031    /**
032     * Client API to submit and manage Oozie workflow jobs against an Oozie intance.
033     * <p/>
034     * This class is thread safe.
035     * <p/>
036     * Syntax for filter for the {@link #getJobsInfo(String)}  {@link #getJobsInfo(String, int, int)}  methods:
037     * <code>[NAME=VALUE][;NAME=VALUE]*</code>.
038     * <p/>
039     * Valid filter names are:
040     * <p/>
041     * <ul/>
042     * <li>name: the workflow application name from the workflow definition.</li>
043     * <li>user: the user that submitted the job.</li>
044     * <li>group: the group for the job.</li>
045     * <li>status: the status of the job.</li>
046     * </ul>
047     * <p/>
048     * The query will do an AND among all the filter names.
049     * The query will do an OR among all the filter values for the same name. Multiple values must be specified as
050     * different name value pairs.
051     */
052    public class LocalOozieClient extends OozieClient {
053    
054        private DagEngine dagEngine;
055    
056        /**
057         * Create a workflow client for Oozie local use.
058         * <p/>
059         *
060         * @param dagEngine the dag engine instance to use.
061         */
062        public LocalOozieClient(DagEngine dagEngine) {
063            this.dagEngine = dagEngine;
064        }
065    
066        /**
067         * Return the Oozie URL of the workflow client instance.
068         * <p/>
069         * This URL is the base URL fo the Oozie system, with not protocol versioning.
070         *
071         * @return the Oozie URL of the workflow client instance.
072         */
073        @Override
074        public String getOozieUrl() {
075            return "localoozie";
076        }
077    
078        /**
079         * Return the Oozie URL used by the client and server for WS communications.
080         * <p/>
081         * This URL is the original URL plus the versioning element path.
082         *
083         * @return the Oozie URL used by the client and server for communication.
084         * @throws org.apache.oozie.client.OozieClientException thrown in the client and the server are not protocol compatible.
085         */
086        @Override
087        public String getProtocolUrl() throws OozieClientException {
088            return "localoozie";
089        }
090    
091        /**
092         * Validate that the Oozie client and server instances are protocol compatible.
093         *
094         * @throws org.apache.oozie.client.OozieClientException thrown in the client and the server are not protocol compatible.
095         */
096        @Override
097        public synchronized void validateWSVersion() throws OozieClientException {
098        }
099    
100        /**
101         * Create an empty configuration with just the {@link #USER_NAME} set to the JVM user name and the
102         * {@link #GROUP_NAME} set to 'other'.
103         *
104         * @return an empty configuration.
105         */
106        @Override
107        public Properties createConfiguration() {
108            Properties conf = new Properties();
109            if (dagEngine != null) {
110                conf.setProperty(USER_NAME, dagEngine.getUser());
111            }
112            conf.setProperty(GROUP_NAME, "users");
113            return conf;
114        }
115    
116        /**
117         * Set a HTTP header to be used in the WS requests by the workflow instance.
118         *
119         * @param name header name.
120         * @param value header value.
121         */
122        @Override
123        public void setHeader(String name, String value) {
124        }
125    
126        /**
127         * Get the value of a set HTTP header from the workflow instance.
128         *
129         * @param name header name.
130         * @return header value, <code>null</code> if not set.
131         */
132        @Override
133        public String getHeader(String name) {
134            return null;
135        }
136    
137        /**
138         * Remove a HTTP header from the workflow client instance.
139         *
140         * @param name header name.
141         */
142        @Override
143        public void removeHeader(String name) {
144        }
145    
146        /**
147         * Return an iterator with all the header names set in the workflow instance.
148         *
149         * @return header names.
150         */
151        @Override
152        @SuppressWarnings("unchecked")
153        public Iterator<String> getHeaderNames() {
154            return Collections.EMPTY_SET.iterator();
155        }
156    
157    
158        /**
159         * Submit a workflow job.
160         *
161         * @param conf job configuration.
162         * @return the job Id.
163         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be submitted.
164         */
165        @Override
166        public String submit(Properties conf) throws OozieClientException {
167            try {
168                return dagEngine.submitJob(new XConfiguration(conf), false);
169            }
170            catch (DagEngineException ex) {
171                throw new OozieClientException(ex.getErrorCode().toString(), ex);
172            }
173        }
174    
175        /**
176         * Start a workflow job.
177         *
178         * @param jobId job Id.
179         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be started.
180         */
181        @Override
182        public void start(String jobId) throws OozieClientException {
183            try {
184                dagEngine.start(jobId);
185            }
186            catch (DagEngineException ex) {
187                throw new OozieClientException(ex.getErrorCode().toString(), ex);
188            }
189        }
190    
191        /**
192         * Submit and start a workflow job.
193         *
194         * @param conf job configuration.
195         * @return the job Id.
196         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be submitted.
197         */
198        @Override
199        public String run(Properties conf) throws OozieClientException {
200            try {
201                return dagEngine.submitJob(new XConfiguration(conf), true);
202            }
203            catch (DagEngineException ex) {
204                throw new OozieClientException(ex.getErrorCode().toString(), ex);
205            }
206        }
207    
208        /**
209         * Rerun a workflow job.
210         *
211         * @param jobId job Id to rerun.
212         * @param conf configuration information for the rerun.
213         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be started.
214         */
215        @Override
216        public void reRun(String jobId, Properties conf) throws OozieClientException {
217            try {
218                dagEngine.reRun(jobId, new XConfiguration(conf));
219            }
220            catch (DagEngineException ex) {
221                throw new OozieClientException(ex.getErrorCode().toString(), ex);
222            }
223        }
224    
225        /**
226         * Suspend a workflow job.
227         *
228         * @param jobId job Id.
229         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be suspended.
230         */
231        @Override
232        public void suspend(String jobId) throws OozieClientException {
233            try {
234                dagEngine.suspend(jobId);
235            }
236            catch (DagEngineException ex) {
237                throw new OozieClientException(ex.getErrorCode().toString(), ex);
238            }
239        }
240    
241        /**
242         * Resume a workflow job.
243         *
244         * @param jobId job Id.
245         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be resume.
246         */
247        @Override
248        public void resume(String jobId) throws OozieClientException {
249            try {
250                dagEngine.resume(jobId);
251            }
252            catch (DagEngineException ex) {
253                throw new OozieClientException(ex.getErrorCode().toString(), ex);
254            }
255        }
256    
257        /**
258         * Kill a workflow job.
259         *
260         * @param jobId job Id.
261         * @throws org.apache.oozie.client.OozieClientException thrown if the job could not be killed.
262         */
263        @Override
264        public void kill(String jobId) throws OozieClientException {
265            try {
266                dagEngine.kill(jobId);
267            }
268            catch (DagEngineException ex) {
269                throw new OozieClientException(ex.getErrorCode().toString(), ex);
270            }
271        }
272    
273        /**
274         * Get the info of a workflow job.
275         *
276         * @param jobId job Id.
277         * @return the job info.
278         * @throws org.apache.oozie.client.OozieClientException thrown if the job info could not be retrieved.
279         */
280        @Override
281        public WorkflowJob getJobInfo(String jobId) throws OozieClientException {
282            try {
283                return dagEngine.getJob(jobId);
284            }
285            catch (DagEngineException ex) {
286                throw new OozieClientException(ex.getErrorCode().toString(), ex);
287            }
288        }
289    
290        /**
291         * Return the info of the workflow jobs that match the filter.
292         *
293         * @param filter job filter. Refer to the {@link LocalOozieClient} for the filter syntax.
294         * @param start jobs offset, base 1.
295         * @param len number of jobs to return.
296         * @return a list with the workflow jobs info, without node details.
297         * @throws org.apache.oozie.client.OozieClientException thrown if the jobs info could not be retrieved.
298         */
299        @Override
300        public List<WorkflowJob> getJobsInfo(String filter, int start, int len) throws OozieClientException {
301            try {
302                return (List<WorkflowJob>) (List) dagEngine.getJobs(filter, start, len).getWorkflows();
303            }
304            catch (DagEngineException ex) {
305                throw new OozieClientException(ex.getErrorCode().toString(), ex);
306            }
307        }
308    
309        /**
310         * Return the info of the workflow jobs that match the filter.
311         * <p/>
312         * It returns the first 100 jobs that match the filter.
313         *
314         * @param filter job filter. Refer to the {@link LocalOozieClient} for the filter syntax.
315         * @return a list with the workflow jobs info, without node details.
316         * @throws org.apache.oozie.client.OozieClientException thrown if the jobs info could not be retrieved.
317         */
318        public List<WorkflowJob> getJobsInfo(String filter) throws OozieClientException {
319            return getJobsInfo(filter, 1, 100);
320        }
321    
322        /**
323         * Return the workflow job Id for an external Id.
324         * <p/>
325         * The external Id must have provided at job creation time.
326         *
327         * @param externalId external Id given at job creation time.
328         * @return the workflow job Id for an external Id, <code>null</code> if none.
329         * @throws org.apache.oozie.client.OozieClientException thrown if the operation could not be done.
330         */
331        @Override
332        public String getJobId(String externalId) throws OozieClientException {
333            try {
334                return dagEngine.getJobIdForExternalId(externalId);
335            }
336            catch (DagEngineException ex) {
337                throw new OozieClientException(ex.getErrorCode().toString(), ex);
338            }
339        }
340    
341        /**
342         * Returns if Oozie is in safe mode or not.
343         *
344         * @return true if safe mode is ON<br> false if safe mode is OFF
345         * @throws org.apache.oozie.client.OozieClientException throw if it could not obtain the safe mode status.
346         */
347        public boolean isInSafeMode() throws OozieClientException {
348            return Services.get().isSafeMode();
349        }
