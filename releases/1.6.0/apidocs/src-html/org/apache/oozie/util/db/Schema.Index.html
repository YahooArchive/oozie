031        public static interface Table {

033             * Name of the Table
034             * 
035             * @return

037            String name();
038        }

040        /**
041         * Interface or table columns
042         */
043        public static interface Column {
044            /**
045             * Table to which the column belongs
046             * 
047             * @return table name
048             */
049            Table table();
050    
051            /**
052             * Alias to be used by the select statement for this column
053             * 
054             * @return alias for column
055             */
056            String asLabel();
057    
058            /**
059             * Name of the column
060             * 
061             * @return column name
062             */
063            String columnName();
064    
065            /**
066             * Returns the datatype of the column
067             * 
068             * @return column type
069             */
070            Class<?> getType();
071    
072            /**
073             * Returns the length of the column
074             * 
075             * @return
076             */
077            int getLength();
078    
079            /**
080             * Returns if the field is a primary key or not
081             * 
082             * @return true if field is a primary key
083             */
084            boolean isPrimaryKey();
085        }

094             * @return

096            Column column();
097        }

099        /**
100         * DB types
101         */
102        public static enum DBType {
103            HSQL, MySQL;
104        }
105    
106        //TODO Add the SQL Change catalog for different DBMS.
107        /**
108         * Returns the appropriate DB type for given column according to the DB Type
109         * 
110         * @param column
111         * @param dbType
112         * @return column type
113         */
114        public static String getDbDataType(Column column, DBType dbType) {
115            String retVal = null;
116            if (String.class.equals(column.getType())) {
117                if (column.getLength() < 0) {
118                    retVal = (dbType.equals(DBType.HSQL) ? "VARCHAR" : "TEXT");
119                }
120                else {
121                    retVal = "VARCHAR(" + column.getLength() + ")";
122                }
123            }
124            else if (Timestamp.class.equals(column.getType())) {
125                retVal = "DATETIME";
126            }
127            else if (Boolean.class.equals(column.getType())) {
128                retVal = "BOOLEAN";
129            }
130            else if (Long.class.equals(column.getType())) {
131                retVal = "BIGINT";
132            }
133            else if (Blob.class.equals(column.getType())) {
134                retVal = (dbType.equals(DBType.MySQL) ? "MEDIUMBLOB" : "LONGVARBINARY");
135            }
136            else {
137                throw new RuntimeException("Column Type[" + column.getType() + "] not mapped to any DB Data Type !!");
138            }
139            return retVal;
140        }
141    
142        /**
143         * Generates the SQL Statement for creating the table
144         * 
145         * @param table
146         * @param dbType
147         * @param tableColumns
148         * @return CREATE TABLE SQL Statement
149         */
150        public static String generateCreateTableScript(Table table, DBType dbType, List<Column> tableColumns) {
151            StringBuilder sb = new StringBuilder("CREATE TABLE ").append(table).append(" ( ");
152            StringBuilder pk = new StringBuilder(", PRIMARY KEY ( ");
153            boolean pkFlag = false;
154            String sep = "";
155            String psep = "";
156            for (Column column : tableColumns) {
157                sb.append(sep).append(column.columnName() + " ").append(Schema.getDbDataType(column, dbType));
158                if (column.isPrimaryKey()) {
159                    pkFlag = true;
160                    pk.append(psep).append(column.columnName());
161                    psep = ", ";
162                }
163                sep = ", ";
164            }
165            if (pkFlag) {
166                pk.append(" )");
167                sb.append(pk.toString());
168            }
169            sb.append(" )");
170            if(dbType == DBType.MySQL) {
171                sb.append(" ENGINE=InnoDB");
172            }
173            return sb.toString();
174        }
175    
176        /**
177         * Generates the SQL statement for creating the Index
178         * 
179         * @param index
180         * @param dbType
181         * @return CREATE INDEX SQL Statement
182         */
183        public static String generateCreateIndexScript(Index index, DBType dbType) {
184            StringBuilder sb = new StringBuilder("CREATE INDEX ").append(index).append(" ON ").append(
185                    index.column().table()).append("( " + index.column().columnName() + " )");
186            return sb.toString();
187        }
188    
189        /**
190         * Checks if the given connection's driver is HSQL Database Driver
191         * @param conn
192         * @return true if the driver is HSQL
193         * @throws SQLException
194         */
195        public static boolean isHsqlConnection(Connection conn) throws SQLException {
196            if(conn.getMetaData().getDriverName().toLowerCase().contains(DBType.HSQL.name().toLowerCase())) {
197                return true;
198            }
199            return false;
200        }
