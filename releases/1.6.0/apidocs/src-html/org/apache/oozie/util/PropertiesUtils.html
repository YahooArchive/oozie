018    package org.apache.oozie.util;
019    
020    import java.util.Properties;
021    import java.io.StringWriter;
022    import java.io.IOException;
023    import java.io.StringReader;
024    import java.io.Reader;
025    
026    public class PropertiesUtils {
027    
028        public static String propertiesToString(Properties props) {
029            ParamChecker.notNull(props, "props");
030            try {
031                StringWriter sw = new StringWriter();
032                props.store(sw, "");
033                sw.close();
034                return sw.toString();
035            }
036            catch (IOException ex) {
037                throw new RuntimeException(ex);
038            }
039        }
040    
041        public static Properties stringToProperties(String str) {
042            ParamChecker.notNull(str, "str");
043            try {
044                StringReader sr = new StringReader(str);
045                Properties props = new Properties();
046                props.load(sr);
047                sr.close();
048                return props;
049            }
050            catch (IOException ex) {
051                throw new RuntimeException(ex);
052            }
053        }
054    
055        public static Properties readProperties(Reader reader, int maxDataLen) throws IOException {
056            String data = IOUtils.getReaderAsString(reader, maxDataLen);
057            return stringToProperties(data);
058        }
059    
060    }
