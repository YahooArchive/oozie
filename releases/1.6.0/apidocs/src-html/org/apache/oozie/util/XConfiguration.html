041    /**
042     * Extends Hadoop Configuration providing a new constructor which reads an XML configuration from an InputStream.
043     * <p/>
044     * OConfiguration(InputStream is).
045     */
046    public class XConfiguration extends Configuration {
047    
048        /**
049         * Create an empty configuration.
050         * <p/>
051         * Default values are not loaded.
052         */
053        public XConfiguration() {
054            super(false);
055        }
056    
057        /**
058         * Create a configuration from an InputStream.
059         * <p/>
060         * Code canibalized from <code>Configuration.loadResource()</code>.
061         *
062         * @param is inputstream to read the configuration from.
063         * @throws IOException thrown if the configuration could not be read.
064         */
065        public XConfiguration(InputStream is) throws IOException {
066            this();
067            parse(is);
068        }
069    
070        /**
071         * Create a configuration from an Reader.
072         * <p/>
073         * Code canibalized from <code>Configuration.loadResource()</code>.
074         *
075         * @param reader reader to read the configuration from.
076         * @throws IOException thrown if the configuration could not be read.
077         */
078        public XConfiguration(Reader reader) throws IOException {
079            this();
080            parse(reader);
081        }
082    
083        /**
084         * Create an configuration from a Properties instance.
085         *
086         * @param props Properties instance to get all properties from.
087         */
088        public XConfiguration(Properties props) {
089            this();
090            for (Map.Entry entry : props.entrySet()) {
091                set((String) entry.getKey(), (String) entry.getValue());
092            }
093    
094        }
095    
096        /**
097         * Return a Properties instance with the configuration properties.
098         *
099         * @return a Properties instance with the configuration properties.
100         */
101        public Properties toProperties() {
102            Properties props = new Properties();
103            for (Map.Entry<String, String> entry : this) {
104                    props.setProperty(entry.getKey(), entry.getValue());
105            }
106            return props;
107        }
108    
109        /**
110         * This is a stop gap fix for <link href="https://issues.apache.org/jira/browse/HADOOP-4416">HADOOP-4416</link>.
111         */
112        public Class<?> getClassByName(String name) throws ClassNotFoundException {
113            return super.getClassByName(name.trim());
114        }
115    
116        /**
117         * Copy configuration key/value pairs from one configuration to another if a property exists in the target, it
118         * gets replaced.
119         *
120         * @param source source configuration.
121         * @param target target configuration.
122         */
123        public static void copy(Configuration source, Configuration target) {
124            for (Map.Entry<String, String> entry : source) {
125                target.set(entry.getKey(), entry.getValue());
126            }
127        }
128    
129        /**
130         * Injects configuration key/value pairs from one configuration to another if the key does not exist
131         * in the target configuration.
132         *
133         * @param source    source configuration.
134         * @param target target configuration.
135         */
136        public static void injectDefaults(Configuration source, Configuration target) {
137            for (Map.Entry<String, String> entry : source) {
138                if (target.get(entry.getKey()) == null) {
139                    target.set(entry.getKey(), entry.getValue());
140                }
141            }
142        }
143    
144        /**
145         * Returns a new Configuration with all values trimmed.
146         *
147         * @param conf the configuraton to trim.
148         * @return a new Configuration with all values trimmed.
149         */
150        public static Configuration trim(Configuration conf) {
151            XConfiguration trimmed = new XConfiguration();
152            for (Map.Entry<String, String> entry : conf) {
153                trimmed.set(entry.getKey(), entry.getValue().trim());
154            }
155            return trimmed;
156        }
157    
158        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
159        private void parse(InputStream is) throws IOException {
160            try {
161                DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
162                // ignore all comments inside the xml file
163                docBuilderFactory.setIgnoringComments(true);
164                DocumentBuilder builder = docBuilderFactory.newDocumentBuilder();
165                Document doc = builder.parse(is);
166                parseDocument(doc);
167            }
168            catch (SAXException e) {
169                throw new IOException(e);
170            }
171            catch (ParserConfigurationException e) {
172                throw new IOException(e);
173            }
174        }
175    
176        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
177        private void parse(Reader reader) throws IOException {
178            try {
179                DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
180                // ignore all comments inside the xml file
181                docBuilderFactory.setIgnoringComments(true);
182                DocumentBuilder builder = docBuilderFactory.newDocumentBuilder();
183                Document doc = builder.parse(new InputSource(reader));
184                parseDocument(doc);
185            }
186            catch (SAXException e) {
187                throw new IOException(e);
188            }
189            catch (ParserConfigurationException e) {
190                throw new IOException(e);
191            }
192        }
193    
194        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
195        private void parseDocument(Document doc) throws IOException {
196            try {
197                Element root = doc.getDocumentElement();
198                if (!"configuration".equals(root.getTagName())) {
199                    throw new IOException("bad conf file: top-level element not <configuration>");
200                }
201                NodeList props = root.getChildNodes();
202                for (int i = 0; i < props.getLength(); i++) {
203                    Node propNode = props.item(i);
204                    if (!(propNode instanceof Element)) {
205                        continue;
206                    }
207                    Element prop = (Element) propNode;
208                    if (!"property".equals(prop.getTagName())) {
209                        throw new IOException("bad conf file: element not <property>");
210                    }
211                    NodeList fields = prop.getChildNodes();
212                    String attr = null;
213                    String value = null;
214                    for (int j = 0; j < fields.getLength(); j++) {
215                        Node fieldNode = fields.item(j);
216                        if (!(fieldNode instanceof Element)) {
217                            continue;
218                        }
219                        Element field = (Element) fieldNode;
220                        if ("name".equals(field.getTagName()) && field.hasChildNodes()) {
221                            attr = ((Text) field.getFirstChild()).getData().trim();
222                        }
223                        if ("value".equals(field.getTagName()) && field.hasChildNodes()) {
224                            value = ((Text) field.getFirstChild()).getData();
225                        }
226                    }
227    
228                    if (attr != null && value != null) {
229                        set(attr, value);
230                    }
231                }
232    
233            }
234            catch (DOMException e) {
235                throw new IOException(e);
236            }
237        }

239        /**
240         * Return a string with the configuration in XML format.
241         *
242         * @return a string with the configuration in XML format.
243         */
244        public String toXmlString() {
245            return toXmlString(true);
246        }
247    
248        public String toXmlString(boolean prolog) {
249            String xml;
250            try {
251                ByteArrayOutputStream baos = new ByteArrayOutputStream();
252                this.writeXml(baos);
253                baos.close();
254                xml =  new String(baos.toByteArray());
255            }
256            catch (IOException ex) {
257                throw new RuntimeException("It should not happen, " + ex.getMessage(), ex);
258            }
259            if (!prolog) {
260                xml = xml.substring(xml.indexOf("<configuration>"));
261            }
262            return xml;
263        }
