021    import java.util.List;
022    import java.util.Map;
023    import java.util.regex.Matcher;
024    import java.util.regex.Pattern;
025    
026    import java.util.Date;
027    import java.io.File;
028    import java.io.FileInputStream;
029    import java.io.IOException;
030    import java.io.InputStream;
031    import java.io.Writer;
032    import java.util.ArrayList;
033    import java.util.Collections;

037    /**
038     * XLogStreamer streams the given log file to logWriter after applying the given
039     * filter.
040     */
041    public class XLogStreamer {
042    
043        /**
044         * Filter that will construct the regular expression that will be used to
045         * filter the log statement. And also checks if the given log message go
046         * through the filter. Filters that can be used are logLevel(Multi values
047         * separated by "|") jobId appName actionId token
048         */
049        public static class Filter {
050            private Map<String, Integer> logLevels;
051            private Map<String, String> filterParams;
052            private static List<String> parameters = new ArrayList<String>();
053            private boolean noFilter;
054            private Pattern filterPattern;
055    
056            //TODO Patterns to be read from config file
057            private static final String DEFAULT_REGEX = "[^\\]]*";
058    
059            public static final String ALLOW_ALL_REGEX = "(.*)";
060            private static final String TIMESTAMP_REGEX = "(\\d\\d\\d\\d-\\d\\d-\\d\\d \\d\\d:\\d\\d:\\d\\d,\\d\\d\\d)";
061            private static final String WHITE_SPACE_REGEX = "\\s+";
062            private static final String LOG_LEVEL_REGEX = "(\\w+)";
063            private static final String PREFIX_REGEX = TIMESTAMP_REGEX + WHITE_SPACE_REGEX + LOG_LEVEL_REGEX
064                    + WHITE_SPACE_REGEX;
065            private static final Pattern SPLITTER_PATTERN = Pattern.compile(PREFIX_REGEX + ALLOW_ALL_REGEX);
066    
067            public Filter() {
068                filterParams = new HashMap<String, String>();
069                for (int i = 0; i < parameters.size(); i++) {
070                    filterParams.put(parameters.get(i), DEFAULT_REGEX);
071                }
072                logLevels = null;
073                noFilter = true;
074                filterPattern = null;
075            }
076    
077            public void setLogLevel(String logLevel) {
078                if (logLevel != null && logLevel.trim().length() > 0) {
079                    this.logLevels = new HashMap<String, Integer>();
080                    String[] levels = logLevel.split("\\|");
081                    for (int i = 0; i < levels.length; i++) {
082                        String s = levels[i].trim().toUpperCase();
083                        try {
084                            XLog.Level.valueOf(s);
085                        }
086                        catch (Exception ex) {
087                            continue;
088                        }
089                        this.logLevels.put(levels[i].toUpperCase(), 1);
090                    }
091                }
092            }
093    
094            public void setParameter(String filterParam, String value) {
095                if (filterParams.containsKey(filterParam)) {
096                    noFilter = false;
097                    filterParams.put(filterParam, value);
098                }
099            }
100    
101            public static void defineParameter(String filterParam) {
102                parameters.add(filterParam);
103            }
104    
105            public boolean isFilterPresent() {
106                if (noFilter && logLevels == null) {
107                    return false;
108                }
109                return true;
110            }
111    
112            /**
113             * Checks if the logLevel and logMessage goes through the logFilter.
114             * 
115             * @param logParts
116             * @return
117             */
118            public boolean matches(ArrayList<String> logParts) {
119                String logLevel = logParts.get(0);
120                String logMessage = logParts.get(1);
121                if (this.logLevels == null || this.logLevels.containsKey(logLevel.toUpperCase())) {
122                    Matcher logMatcher = filterPattern.matcher(logMessage);
123                    return logMatcher.matches();
124                }
125                else {
126                    return false;
127                }
128            }
129    
130            /**
131             * Splits the log line into timestamp, logLevel and remaining log
132             * message. Returns array containing logLevel and logMessage if the
133             * pattern matches i.e A new log statement, else returns null.
134             * 
135             * @param logLine
136             * @return Array containing log level and log message
137             */
138            public ArrayList<String> splitLogMessage(String logLine) {
139                Matcher splitter = SPLITTER_PATTERN.matcher(logLine);
140                if (splitter.matches()) {
141                    ArrayList<String> logParts = new ArrayList<String>();
142                    logParts.add(splitter.group(2));// log level
143                    logParts.add(splitter.group(3));// Log Message
144                    return logParts;
145                }
146                else {
147                    return null;
148                }
149            }
150    
151            /**
152             * Constructs the regular expression according to the filter and assigns
153             * it to fileterPattarn. ".*" will be assigned if no filters are set.
154             */
155            public void constructPattern() {
156                if (noFilter && logLevels == null) {
157                    filterPattern = Pattern.compile(ALLOW_ALL_REGEX);
158                    return;
159                }
160                StringBuilder sb = new StringBuilder();
161                if (noFilter) {
162                    sb.append("(.*)");
163                }
164                else {
165                    sb.append("(.* - ");
166                    for (int i = 0; i < parameters.size(); i++) {
167                        sb.append(parameters.get(i) + "\\[");
168                        sb.append(filterParams.get(parameters.get(i)) + "\\] ");
169                    }
170                    sb.append(".*)");
171                }
172                filterPattern = Pattern.compile(sb.toString());
173            }
174    
175            public static void reset() {
176                parameters.clear();
177            }
178        }
179    
180        private String logFile;
181        private String logPath;
182        private Filter logFilter;
183        private Writer logWriter;
184        private long logRotation;
185    
186        public XLogStreamer(Filter logFilter, Writer logWriter, String logPath, String logFile, long logRotationSecs) {
187            this.logWriter = logWriter;
188            this.logFilter = logFilter;
189            if (logFile == null) {
190                logFile = "oozie-app.log";
191            }
192            this.logFile = logFile;
193            this.logPath = logPath;
194            this.logRotation = logRotationSecs * 1000l;
195        }
196    
197        /**
198         * Gets the files that are modified between startTime and endTime in the
199         * given logPath and streams the log after applying the filters.
200         * 
201         * @param startTime
202         * @param endTime
203         * @throws IOException
204         */
205        public void streamLog(Date startTime, Date endTime) throws IOException {
206            long startTimeMillis = 0;
207            long endTimeMillis;
208            if (startTime != null) {
209                startTimeMillis = startTime.getTime();
210            }
211            if (endTime == null) {
212                endTimeMillis = System.currentTimeMillis();
213            }
214            else {
215                endTimeMillis = endTime.getTime();
216            }
217            File dir = new File(logPath);
218            ArrayList<FileInfo> fileList = getFileList(dir, startTimeMillis, endTimeMillis, logRotation, logFile);
219            for (int i = 0; i < fileList.size(); i++) {
220                InputStream ifs;
221                ifs = new FileInputStream(fileList.get(i).getFileName());
222                XLogReader logReader = new XLogReader(ifs, logFilter, logWriter);
223                logReader.processLog();
224            }
225        }
226    
227        /**
228         * File name along with the modified time which will be used to sort later.
229         */
230        class FileInfo implements Comparable<FileInfo> {
231            String fileName;
232            long modTime;
233    
234            public FileInfo(String fileName, long modTime) {
235                this.fileName = fileName;
236                this.modTime = modTime;
237            }
238    
239            public String getFileName() {
240                return fileName;
241            }
242    
243            public long getModTime() {
244                return modTime;
245            }
246    
247            public int compareTo(FileInfo fileInfo) {
248                return (int) (this.modTime - fileInfo.modTime);
249            }
250        }
251    
252        /**
253         * Gets the file list that will have the logs between startTime and endTime.
254         * 
255         * @param dir
256         * @param startTime
257         * @param endTime
258         * @param logRotationTime
259         * @param logFile
260         * @return List of files to be streamed
261         */
262        private ArrayList<FileInfo> getFileList(File dir, long startTime, long endTime, long logRotationTime,
263                                                String logFile) {
264            String[] children = dir.list();
265            ArrayList<FileInfo> fileList = new ArrayList<FileInfo>();
266            if (children == null) {
267                return fileList;
268            }
269            else {
270                for (int i = 0; i < children.length; i++) {
271                    String filename = children[i];
272                    if (!filename.startsWith(logFile) && !filename.equals(logFile)) {
273                        continue;
274                    }
275                    File file = new File(dir.getAbsolutePath(), filename);
276                    long modTime = file.lastModified();
277                    if (modTime < startTime) {
278                        continue;
279                    }
280                    if (modTime / logRotationTime > (endTime / logRotationTime + 1)) {
281                        continue;
282                    }
283                    fileList.add(new FileInfo(file.getAbsolutePath(), modTime));
284                }
285            }
286            Collections.sort(fileList);
287            return fileList;
288        }
