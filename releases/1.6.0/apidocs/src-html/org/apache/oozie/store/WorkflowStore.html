018    package org.apache.oozie.store;
019    
020    import org.apache.oozie.WorkflowActionBean;
021    import org.apache.oozie.WorkflowJobBean;
022    import org.apache.oozie.WorkflowsInfo;
023    
024    import java.util.List;
025    import java.util.Map;
026    
027    /**
028     * <code>WorkflowStore</code> Interface to persist the Jobs and Action
029     */
030    public interface WorkflowStore {
031    
032        /**
033         * Inserts the given workflow into the store
034         * 
035         * @param workflow workflow bean.
036         * @throws StoreException
037         */
038        public void insertWorkflow(WorkflowJobBean workflow) throws StoreException;
039    
040        /**
041         * Load the Workflow with given id to the WorkflowBean. Load the Process
042         * Instance also to the bean. Lock the Workflow if locking option is given.
043         * 
044         * @param id Workflow ID.
045         * @param locking flag to lock the job.
046         * @return WorkflowBean Workflow bean.
047         * @throws StoreException If the job cannot be laoded.
048         */
049        public WorkflowJobBean getWorkflow(String id, boolean locking) throws StoreException;
050    
051        /**
052         * Load the Workflow Info. Load the List of actions of the workflow also.
053         * process instance is not loaded.
054         * 
055         * @param id Workflow ID.
056         * @return Workflow bean.
057         * @throws StoreException
058         */
059        public WorkflowJobBean getWorkflowInfo(String id) throws StoreException;
060    
061        /**
062         * Get the Workflow ID that belongs to the given external ID.
063         *
064         * @param extId External ID.
065         * @return Workflow ID.
066         * @throws StoreException If there are no jobs with given external id.
067         */
068        public String getWorkflowIdForExternalId(String extId) throws StoreException;
069    
070        /**
071         * Update the data from the given WorkflowBean to Store.
072         * 
073         * @param workflow Workflow bean to be updated.
074         * @throws StoreException if the bean cannot be updated.
075         */
076        public void updateWorkflow(WorkflowJobBean workflow) throws StoreException;
077    
078        /**
079         * Get the number of Workflows with the given status.
080         * 
081         * @param status Workflow Status.
082         * @return number of Workflows with given status.
083         * @throws StoreException
084         */
085        public int getWorkflowCountWithStatus(String status) throws StoreException;
086    
087        /**
088         * Get the number of Workflows with the given status which was modified in given time limit.
089         * 
090         * @param status Workflow Status.
091         * @param secs No. of seconds within which the workflow got modified.
092         * @return number of Workflows modified within given time with given status.
093         * @throws StoreException
094         */
095        public int getWorkflowCountWithStatusInLastNSeconds(String status, int secs) throws StoreException;
096    
097        /**
098         * Create a New Action record from the given ActionBean.
099         * 
100         * @param action Action bean.
101         * @throws StoreException if the action cannot be inserted.
102         */
103        public void insertAction(WorkflowActionBean action) throws StoreException;
104    
105        /**
106         * Load the action record with the given id to bean. Lock the action if
107         * locking option is given.
108         * 
109         * @param id Action ID.
110         * @param locking flag to lock the action.
111         * @return Action bean.
112         * @throws StoreException if the action with given id is not present.
113         */
114        public WorkflowActionBean getAction(String id, boolean locking) throws StoreException;
115    
116        /**
117         * Update the data from the given ActionBean to the Store.
118         * @param action Action bean.
119         * @throws StoreException if the action cannot be updated.
120         */
121        public void updateAction(WorkflowActionBean action) throws StoreException;
122    
123        /**
124         * Delete the action with given action id.
125         * @param id Action id.
126         * @throws StoreException if the action is not present.
127         */
128        public void deleteAction(String id) throws StoreException;
129    
130        /**
131         * Load all the actions for the given Workflow id.
132         * @param id Workflow ID.
133         * @param locking flag to lock the actions.
134         * @return List of action beans.
135         * @throws StoreException if there is an error while loading the actions.
136         */
137        public List<WorkflowActionBean> getActionsForWorkflow(String id, boolean locking) throws StoreException;
138    
139        /**
140         * Load the actions that are pending for more than given time.
141         * @param minimumPendingAgeSecs minimum pending age in seconds.
142         * @return list of action beans.
143         * @throws StoreException if there is an error while loading the actions.
144         */
145        public List<WorkflowActionBean> getPendingActions(long minimumPendingAgeSecs) throws StoreException;
146    
147        /**
148         * Load running actions which were last checked before the specified time.
149         *
150         * @param checkAgeSecs the check age in seconds.
151         * @return A list of running actions which were last checked after now - checkAgeSecs.
152         * @throws StoreException
153         */
154        public List<WorkflowActionBean> getRunningActions(long checkAgeSecs) throws StoreException;
155    
156        /**
157         * Load the Workflows according to the given filter information.
158         * @param filter Can be name, status, user, group and combination of these.
159         * @param start returned result set offset.
160         * @param len returned result set rows.
161         * @return List of Workflows Satisfying the given filter.
162         * @throws StoreException thrown if the workflow result could not be queried.
163         */
164        public WorkflowsInfo getWorkflowsInfo(Map<String, List<String>> filter, int start, int len)
165                throws StoreException;
166    
167        /**
168         * Purge the Jobs older than given days.
169         * @param olderThanDays
170         * @throws StoreException
171         */
172        public void purge(long olderThanDays) throws StoreException;
173    
174        /**
175         * Commit the transaction.
176         * @throws StoreException
177         */
178        public void commit() throws StoreException;
179    
180        /**
181         * Close the connection.
182         * @throws StoreException
183         */
184        public void close() throws StoreException;
185    }
