167        public synchronized boolean start() throws WorkflowException {
168            if (status != Status.PREP) {
169                throw new WorkflowException(ErrorCode.E0719);
170            }
171            log.debug(XLog.STD, "Starting job");
172            status = Status.RUNNING;
173            executionPaths.put(ROOT, new NodeInstance(StartNodeDef.START));
174            return signal(ROOT, StartNodeDef.START);
175        }

179        public synchronized boolean signal(String executionPath, String signalValue) throws WorkflowException {
180            ParamChecker.notEmpty(executionPath, "executionPath");
181            ParamChecker.notNull(signalValue, "signalValue");
182            log.debug(XLog.STD, "Signaling job execution path [{0}] signal value [{1}]", executionPath, signalValue);
183            if (status != Status.RUNNING) {
184                throw new WorkflowException(ErrorCode.E0716);
185            }
186            NodeInstance nodeJob = executionPaths.get(executionPath);
187            if (nodeJob == null) {
188                status = Status.FAILED;
189                log.error("invalid execution path [{0}]", executionPath);
190            }
191            NodeDef nodeDef = null;
192            if (!status.isEndState()) {
193                nodeDef = def.getNode(nodeJob.nodeName);
194                if (nodeDef == null) {
195                    status = Status.FAILED;
196                    log.error("invalid transition [{0}]", nodeJob.nodeName);
197                }
198            }
199            if (!status.isEndState()) {
200                NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
201                boolean exiting = true;
202    
203                Context context = new Context(nodeDef, executionPath, signalValue);
204                if (!nodeJob.started) {
205                    try {
206                        nodeHandler.loopDetection(context);
207                        exiting = nodeHandler.enter(context);
208                        nodeJob.started = true;
209                    }
210                    catch (WorkflowException ex) {
211                        status = Status.FAILED;
212                        throw ex;
213                    }
214                }
215    
216                if (exiting) {
217                    List<String> pathsToStart = new ArrayList<String>();
218                    List<String> fullTransitions;
219                    try {
220                        fullTransitions = nodeHandler.multiExit(context);
221                        int last = fullTransitions.size() - 1;
222                        // TEST THIS
223                        if (last >= 0) {
224                            String transitionTo = getTransitionNode(fullTransitions.get(last));
225    
226                            persistentVars.put(nodeDef.getName() + WorkflowInstance.NODE_VAR_SEPARATOR + TRANSITION_TO,
227                                    transitionTo);
228                        }
229                    }
230                    catch (WorkflowException ex) {
231                        status = Status.FAILED;
232                        throw ex;
233                    }
234    
235                    if (context.status == Status.KILLED) {
236                        status = Status.KILLED;
237                        log.debug(XLog.STD, "Completing job, kill node [{0}]", nodeJob.nodeName);
238                    }
239                    else if (context.status == Status.FAILED) {
240                        status = Status.FAILED;
241                        log.debug(XLog.STD, "Completing job, fail node [{0}]", nodeJob.nodeName);
242                    }
243                    else if (context.status == Status.SUCCEEDED) {
244                        status = Status.SUCCEEDED;
245                        log.debug(XLog.STD, "Completing job, end node [{0}]", nodeJob.nodeName);
246                    }
247                    else {
248                        for (String fullTransition : fullTransitions) {
249                            // this is the whole trick for forking, we need the
250                            // executionpath and the transition
251                            // in the case of no forking last element of
252                            // executionpath is different from transition
253                            // in the case of forking they are the same
254    
255                            log.debug(XLog.STD, "Exiting node [{0}] with transition[{1}]", nodeJob.nodeName,
256                                            fullTransition);
257    
258                            String execPathFromTransition = getExecutionPath(fullTransition);
259                            String transition = getTransitionNode(fullTransition);
260                            def.validateTransition(nodeJob.nodeName, transition);
261    
262                            NodeInstance nodeJobInPath = executionPaths.get(execPathFromTransition);
263                            if ((nodeJobInPath == null) || (!transition.equals(nodeJobInPath.nodeName))) {
264                                // TODO explain this IF better
265                                // If the WfJob is signaled with the parent
266                                // execution executionPath again
267                                // The Fork node will execute again.. and replace
268                                // the Node WorkflowJobBean
269                                // so this is required to prevent that..
270                                // Question : Should we throw an error in this case
271                                // ??
272                                executionPaths.put(execPathFromTransition, new NodeInstance(transition));
273                                pathsToStart.add(execPathFromTransition);
274                            }
275    
276                        }
277                        // signal all new synch transitions
278                        for (String pathToStart : pathsToStart) {
279                            signal(pathToStart, "::synch::");
280                        }
281                    }
282                }
283            }
284            if (status.isEndState()) {
285                if (status == Status.FAILED) {
286                    List<String> failedNodes = terminateNodes(status);
287                    log.warn(XLog.STD, "Workflow completed [{0}], failing [{1}] running nodes", status, failedNodes
288                            .size());
289                } else {
290                    List<String> killedNodes = terminateNodes(Status.KILLED);
291                    if (killedNodes.size() > 1) {
292                        log.warn(XLog.STD, "Workflow completed [{0}], killing [{1}] running nodes", status, killedNodes
293                                .size());
294                    }
295                }
296            }
297            return status.isEndState();
298        }

300        public synchronized void fail(String nodeName) throws WorkflowException {
301            if (status.isEndState()) {
302                throw new WorkflowException(ErrorCode.E0718);
303            }
304            String failedNode = failNode(nodeName);
305            if (failedNode != null) {
306                log.warn(XLog.STD, "Workflow Failed. Failing node [{0}]", failedNode);
307            } else {
308                //TODO failed attempting to fail the action. EXCEPTION
309            }
310            List<String> killedNodes = killNodes();
311            if (killedNodes.size() > 1) {
312                log.warn(XLog.STD, "Workflow Failed, killing [{0}] nodes", killedNodes.size());
313            }
314            status = Status.FAILED;
315        }
316    
317        public synchronized void kill() throws WorkflowException {
318            if (status.isEndState()) {
319                throw new WorkflowException(ErrorCode.E0718);
320            }
321            log.debug(XLog.STD, "Killing job");
322            List<String> killedNodes = killNodes();
323            if (killedNodes.size() > 1) {
324                log.warn(XLog.STD, "workflow killed, killing [{0}] nodes", killedNodes.size());
325            }
326            status = Status.KILLED;
327        }

330            if (status != Status.RUNNING) {
331                throw new WorkflowException(ErrorCode.E0716);
332            }
333            log.debug(XLog.STD, "Suspending job");
334            status = Status.SUSPENDED;
335        }

394        private List<String> terminateNodes(Status endStatus) {
395            List<String> endNodes = new ArrayList<String>();
396            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
397                if (entry.getValue().started) {
398                    NodeDef nodeDef = def.getNode(entry.getValue().nodeName);
399                    NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
400                    try {
401                        if (endStatus == Status.KILLED) {
402                            nodeHandler.kill(new Context(nodeDef, entry.getKey(), null));
403                        }
404                        else if (endStatus == Status.FAILED) {
405                            nodeHandler.fail(new Context(nodeDef, entry.getKey(), null));
406                        }
407                        endNodes.add(nodeDef.getName());
408                    }
409                    catch (Exception ex) {
410                        log.warn(XLog.STD, "Error Changing node state to [{0}] for Node [{1}]", endStatus.toString(),
411                                nodeDef.getName(), ex);
412                    }
413                }
414            }
415            return endNodes;
416        }
417    
418        private String failNode(String nodeName) {
419            String failedNode = null;
420            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
421                String node = entry.getKey();
422                NodeInstance nodeInstance = entry.getValue();
423                if (nodeInstance.started && nodeInstance.nodeName.equals(nodeName)) {
424                    NodeDef nodeDef = def.getNode(nodeInstance.nodeName);
425                    NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
426                    try {
427                        nodeHandler.fail(new Context(nodeDef, node, null));
428                        failedNode = nodeDef.getName();
429                        nodeInstance.started = false;
430                    }
431                    catch (Exception ex) {
432                        log.warn(XLog.STD, "Error failing node [{0}]", nodeDef.getName(), ex);
433                    }
434                    return failedNode;
435                }
436            }
437            return failedNode;
438        }

440        private List<String> killNodes() {
441            List<String> killedNodes = new ArrayList<String>();
442            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
443                String node = entry.getKey();
444                NodeInstance nodeInstance = entry.getValue();
445                if (nodeInstance.started) {
446                    NodeDef nodeDef = def.getNode(nodeInstance.nodeName);
447                    NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
448                    try {
449                        nodeHandler.kill(new Context(nodeDef, node, null));
450                        killedNodes.add(nodeDef.getName());
451                    }
452                    catch (Exception ex) {
453                        log.warn(XLog.STD, "Error killing node [{0}]", nodeDef.getName(), ex);
454                    }
455                }
456            }
457            return killedNodes;
458        }

464        private static String createChildPath(String path, String child) {
465            return path + child + PATH_SEPARATOR;
466        }

468        private static String getParentPath(String path) {
469            path = path.substring(0, path.length() - 1);
470            return (path.length() == 0) ? null : path.substring(0, path.lastIndexOf(PATH_SEPARATOR) + 1);
471        }
472    
473        private static String createFullTransition(String executionPath, String transition) {
474            return executionPath + TRANSITION_SEPARATOR + transition;
475        }
476    
477        private static String getExecutionPath(String fullTransition) {
478            int index = fullTransition.indexOf(TRANSITION_SEPARATOR);
479            if (index == -1) {
480                throw new IllegalArgumentException("Invalid fullTransition");
481            }
482            return fullTransition.substring(0, index);
483        }
484    
485        private static String getTransitionNode(String fullTransition) {
486            int index = fullTransition.indexOf(TRANSITION_SEPARATOR);
487            if (index == -1) {
488                throw new IllegalArgumentException("Invalid fullTransition");
489            }
490            return fullTransition.substring(index + 1);
491        }

497        private void refreshLog() {
498            XLog.Info.get().setParameter(XLogService.USER, conf.get(OozieClient.USER_NAME));
499            XLog.Info.get().setParameter(XLogService.GROUP, conf.get(OozieClient.GROUP_NAME));
500            XLog.Info.get().setParameter(DagXLogInfoService.APP, def.getName());
501            XLog.Info.get().setParameter(DagXLogInfoService.TOKEN, conf.get(OozieClient.LOG_TOKEN, ""));
502            XLog.Info.get().setParameter(DagXLogInfoService.JOB, instanceId);
503            log = XLog.getLog(getClass());
504        }

510        @Override
511        public void write(DataOutput dOut) throws IOException {
512            dOut.writeUTF(instanceId);
513    
514            //Hadoop Configuration has to get its act right
515            ByteArrayOutputStream baos = new ByteArrayOutputStream();
516            conf.writeXml(baos);
517            baos.close();
518            byte[] array = baos.toByteArray();
519            dOut.writeInt(array.length);
520            dOut.write(array);
521    
522            def.write(dOut);
523            dOut.writeUTF(status.toString());
524            dOut.writeInt(executionPaths.size());
525            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
526                dOut.writeUTF(entry.getKey());
527                dOut.writeUTF(entry.getValue().nodeName);
528                dOut.writeBoolean(entry.getValue().started);
529            }
530            dOut.writeInt(persistentVars.size());
531            for (Map.Entry<String, String> entry : persistentVars.entrySet()) {
532                dOut.writeUTF(entry.getKey());
533                dOut.writeUTF(entry.getValue());
534            }
535        }

537        @Override
538        public void readFields(DataInput dIn) throws IOException {
539            instanceId = dIn.readUTF();
540    
541            //Hadoop Configuration has to get its act right
542            int len = dIn.readInt();
543            byte[] array = new byte[len];
544            dIn.readFully(array);
545            ByteArrayInputStream bais = new ByteArrayInputStream(array);
546            conf = new XConfiguration(bais);
547    
548            def = new LiteWorkflowApp();
549            def.readFields(dIn);
550            status = Status.valueOf(dIn.readUTF());
551            int numExPaths = dIn.readInt();
552            for (int x = 0; x < numExPaths; x++) {
553                String path = dIn.readUTF();
554                String nodeName = dIn.readUTF();
555                boolean isStarted = dIn.readBoolean();
556                NodeInstance nodeInstance = new NodeInstance(nodeName);
557                nodeInstance.started = isStarted;
558                executionPaths.put(path, nodeInstance);
559            }
560            int numVars = dIn.readInt();
561            for (int x = 0; x < numVars; x++) {
562                String vName = dIn.readUTF();
563                String vVal = dIn.readUTF();
564                persistentVars.put(vName, vVal);
565            }
566            refreshLog();
567        }
