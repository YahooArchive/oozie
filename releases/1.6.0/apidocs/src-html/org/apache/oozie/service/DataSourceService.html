018    package org.apache.oozie.service;
019    
020    import org.apache.oozie.store.OozieSchema;
021    import org.apache.oozie.service.DBLiteWorkflowStoreService;
022    import org.apache.commons.dbcp.BasicDataSource;
023    import org.apache.commons.dbcp.BasicDataSourceFactory;
024    import org.apache.commons.logging.LogFactory;
025    import org.apache.hadoop.conf.Configuration;
026    import org.apache.oozie.util.Instrumentable;
027    import org.apache.oozie.util.Instrumentation;
028    import org.apache.oozie.util.XLog;
029    import org.apache.oozie.ErrorCode;
030    
031    import javax.sql.DataSource;
032    import java.lang.reflect.InvocationHandler;
033    import java.lang.reflect.InvocationTargetException;
034    import java.lang.reflect.Method;
035    import java.lang.reflect.Proxy;
036    import java.sql.Connection;
037    import java.sql.DriverManager;
038    import java.sql.SQLException;
039    import java.util.Properties;
040    import java.util.concurrent.atomic.AtomicLong;
041    
042    /**
043     * The datasource service provides access to the SQL datasource used by Oozie.
044     * <p/>
045     * The returned datasource and connections are instrumented.
046     * <p/>
047     * The datasource checks at startup that if the database support select for update or not.
048     * <p/>
049     * The following configuration parameters control the datasource service:
050     * <p/>
051     * Regardless JNDI being set or not the following properties must be set:
052     * <p/>
053     * {@link #CONF_DRIVER} driver class.
054     * <p/>
055     * {@link #CONF_URL} database JDBC URL.
056     * <p/>
057     * {@link #CONF_USERNAME} database user.
058     * <p/>
059     * {@link #CONF_PASSWORD} database password.
060     * <p/>
061     * If JNDI is not set, the following property must be set:
062     * <p/>
063     * {@link #CONF_MAX_ACTIVE_CONN} max number of action JDBC connections.
064     */
065    public class DataSourceService implements Service, Instrumentable {
066        private static final String INTRUMENTATION_GROUP = "jdbc";
067        private static final String INSTR_ACTIVE_CONNECTIONS_SAMPLER = "connections.active";
068    
069        public static final String CONF_PREFIX = Service.CONF_PREFIX + "DataSourceService.";
070        public static final String CONF_DRIVER = CONF_PREFIX + "jdbc.driver";
071        public static final String CONF_URL = CONF_PREFIX + "jdbc.url";
072        public static final String CONF_USERNAME = CONF_PREFIX + "jdbc.username";
073        public static final String CONF_PASSWORD = CONF_PREFIX + "jdbc.password";
074        public static final String CONF_MAX_ACTIVE_CONN = CONF_PREFIX + "pool.max.active.conn";
075    
076        private BasicDataSource ownDataSource;
077        private DataSourceProxy dataSourceProxy;
078        private AtomicLong connectionCount = new AtomicLong();
079    
080        /**
081         * Initialize the datasource service.
082         *
083         * @param services services instance.
084         * @throws ServiceException thrown if the service could not be initialized.
085         */
086        @SuppressWarnings({"ThrowFromFinallyBlock"})
087        public void init(Services services) throws ServiceException {
088            Configuration conf = services.getConf();
089            String dbName = conf.get(DBLiteWorkflowStoreService.CONF_SCHEMA_NAME, "oozie");
090            OozieSchema.setOozieDbName(dbName);
091            String validation_query = OozieSchema.getValidationQuery(dbName);
092            Properties props = new Properties();
093            props.put("driverClassName", conf.get(CONF_DRIVER, "org.hsqldb.jdbcDriver"));
094            props.put("url", conf.get(CONF_URL, "jdbc:hsqldb:mem:testdb"));
095            props.put("username", conf.get(CONF_USERNAME, "sa"));
096            props.put("password", conf.get(CONF_PASSWORD, "").trim());
097            props.put("maxActive", conf.get(CONF_MAX_ACTIVE_CONN, "10"));
098            if(validation_query != null) {
099                props.put("testOnBorrow", "true");
100                props.put("validationQuery", validation_query);
101            }
102            try {
103                ownDataSource = (BasicDataSource) BasicDataSourceFactory.createDataSource(props);
104                dataSourceProxy = createProxy(ownDataSource);
105            }
106            catch (Exception ex) {
107                throw new ServiceException(ErrorCode.E0141, ex.getMessage(), ex);
108            }
109        }
110    
111        /**
112         * Destroy the service.
113         */
114        public void destroy() {
115            if (ownDataSource != null) {
116                try {
117                    ownDataSource.close();
118                }
119                catch (SQLException ex) {
120                    XLog log = new XLog(LogFactory.getLog(getClass()));
121                    log.warn("Failed to close datasource, {0}", ex.getMessage(), ex);
122    
123                }
124                ownDataSource = null;
125            }
126            dataSourceProxy = null;
127        }
128    
129        /**
130         * Return the public interface of the service.
131         *
132         * @return {@link DataSourceService}.
133         */
134        public Class<? extends Service> getInterface() {
135            return DataSourceService.class;
136        }
137    
138        /**
139         * Instrument the service.
140         *
141         * @param instr instrumentation instance.
142         */
143        public void instrument(Instrumentation instr) {
144            instr.addSampler(INTRUMENTATION_GROUP, INSTR_ACTIVE_CONNECTIONS_SAMPLER, 60, 1, new Instrumentation.Variable<Long>() {
145                public Long getValue() {
146                    return connectionCount.get();
147                }
148            });
149        }
150    
151        /**
152         * Return a raw direct JDBC connection.
153         * <p/>
154         * The connection is not from the connection pool.
155         * <p/>
156         * The conection is not instrumented.
157         *
158         * @return a raw Direct JDBC connection.
159         * @throws SQLException thrown if the connection could not be obtained.
160         */
161        public Connection getRawConnection() throws SQLException {
162            String driver = Services.get().getConf().get(CONF_DRIVER, "org.hsqldb.jdbcDriver");
163            String url = Services.get().getConf().get(CONF_URL, "jdbc:hsqldb:mem:testdb");
164            String user = Services.get().getConf().get(CONF_USERNAME, "sa");
165            String password = Services.get().getConf().get(CONF_PASSWORD, "").trim();
166            try {
167                Class.forName(driver);
168            }
169            catch (ClassNotFoundException ex) {
170                throw new RuntimeException(ex);
171            }
172            return DriverManager.getConnection(url, user, password);
173        }
174    
175        /**
176         * Return a managed JDBC connection.
177         *
178         * @return a managed JDBC connection.
179         * @throws SQLException thrown if the managed connection could not be obtained.
180         */
181        public Connection getConnection() throws SQLException {
182            return dataSourceProxy.getDataSource().getConnection();
183        }
184    
185        private DataSourceProxy createProxy(DataSource datasource) {
186            DataSourceProxy proxy = new DataSourceProxy(datasource);
187            proxy.dsProxy = (DataSource) Proxy
188                    .newProxyInstance(datasource.getClass().getClassLoader(), new Class[]{DataSource.class}, proxy);
189            return proxy;
190        }
191    
192        private class DataSourceProxy implements InvocationHandler {
193            private final DataSource datasource;
194            private DataSource dsProxy;
195    
196            private DataSourceProxy(DataSource datasource) {
197                this.datasource = datasource;
198            }
199    
200            public Class<? extends DataSource> getTargetClass() {
201                return datasource.getClass();
202            }
203    
204            public DataSource getDataSource() {
205                return dsProxy;
206            }
207    
208            public DataSource getRawDataSource() {
209                return datasource;
210            }
211    
212            @Override
213            public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
214                Object result;
215                try {
216                    result = method.invoke(datasource, args);
217                    if (method.getName().equals("getConnection")) {
218                        connectionCount.incrementAndGet();
219                        result = Proxy.newProxyInstance(result.getClass().getClassLoader(), new Class[]{Connection.class},
220                                                        new ConnectionProxy((Connection) result));
221                    }
222                }
223                catch (InvocationTargetException ite) {
224                    throw ite.getTargetException();
225                }
226                return result;
227            }
228    
229    
230            private class ConnectionProxy implements InvocationHandler {
231    
232                private final Connection connection;
233    
234                private ConnectionProxy(Connection connection) {
235                    this.connection = connection;
236                }
237    
238                @Override
239                public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
240                    Object result;
241                    try {
242                        if (method.getName().equals("close")) {
243                            connectionCount.decrementAndGet();
244                        }
245                        result = method.invoke(connection, args);
246                    }
247                    catch (InvocationTargetException ite) {
248                        throw ite.getTargetException();
249                    }
250                    return result;
251                }
252    
253            }
254    
255        }
