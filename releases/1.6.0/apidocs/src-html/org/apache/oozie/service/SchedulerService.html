018    package org.apache.oozie.service;
019    
020    import org.apache.oozie.util.XLog;
021    
022    import java.util.concurrent.Callable;
023    import java.util.concurrent.ScheduledExecutorService;
024    import java.util.concurrent.ScheduledThreadPoolExecutor;
025    import java.util.concurrent.TimeUnit;
026    
027    /**
028     * This service executes scheduled Runnables and Callables at regular intervals.
029     * <p/>
030     * It uses a java.util.concurrent.ScheduledExecutorService.
031     * <p/>
032     * The {@link #SCHEDULER_THREADS} configuration property indicates how many threads the scheduler will use to run
033     * scheduled commands.
034     */
035    public class SchedulerService implements Service {
036    
037        public static final String CONF_PREFIX = Service.CONF_PREFIX + "SchedulerService.";
038    
039        public static final String SCHEDULER_THREADS = CONF_PREFIX + "threads";
040    
041        private final XLog log = XLog.getLog(getClass());
042    
043        private ScheduledExecutorService scheduler;
044    
045        /**
046         * Initialize the scheduler service.
047         *
048         * @param services services instance.
049         */
050        @Override
051        public void init(Services services) {
052            scheduler = new ScheduledThreadPoolExecutor(services.getConf().getInt(SCHEDULER_THREADS, 5));
053        }
054    
055        /**
056         * Destroy the scheduler service.
057         */
058        @Override
059        public void destroy() {
060            try {
061                long limit = System.currentTimeMillis() + 30 * 1000;// 30 seconds
062                scheduler.shutdownNow();
063                while (!scheduler.awaitTermination(1000, TimeUnit.MILLISECONDS)) {
064                    log.info("Waiting for scheduler to shutdown");
065                    if (System.currentTimeMillis() > limit) {
066                        log.warn("Gave up, continuing without waiting for scheduler to shutdown");
067                        break;
068                    }
069                }
070            }
071            catch (InterruptedException ex) {
072                log.warn(ex);
073            }
074        }
075    
076        /**
077         * Return the public interface for scheduler service.
078         *
079         * @return {@link SchedulerService}.
080         */
081        @Override
082        public Class<? extends Service> getInterface() {
083            return SchedulerService.class;
084        }
085    
086        /**
087         * Return the java.util.concurrent.ScheduledExecutorService instance used by the SchedulerService.
088         * <p/>
089         *
090         * @return the scheduled executor service instance.
091         */
092        public ScheduledExecutorService getScheduler() {
093            return scheduler;
094        }
095    
096        public enum Unit {
097            MILLISEC(1),
098            SEC(1000),
099            MIN(1000 *60),
100            HOUR(1000 * 60 *60);
101    
102            private long millis;
103    
104            private Unit(long millis) {
105                this.millis = millis;
106            }
107    
108            private long getMillis() {
109                return millis;
110            }
111    
112        }
113    
114        /**
115         * Schedule a Callable for execution.
116         *
117         * @param callable callable to schedule for execution.
118         * @param delay    delay for first execution since scheduling.
119         * @param interval interval between executions.
120         * @param unit scheduling unit.
121         */
122        public void schedule(final Callable<Void> callable, long delay, long interval, Unit unit) {
123            log.trace("Scheduling callable [{0}], interval [{1}] seconds, delay [{2}] in [{3}]",
124                      callable.getClass(), delay, interval, unit);
125            Runnable r = new Runnable() {
126                public void run() {
127                    try {
128                        callable.call();
129                    }
130                    catch (Exception ex) {
131                        log.warn("Error executing callable [{0}], {1}", callable.getClass().getSimpleName(),
132                                 ex.getMessage(), ex);
133                    }
134                }
135            };
136            scheduler.scheduleWithFixedDelay(r, delay * unit.getMillis(), interval * unit.getMillis(),
137                                             TimeUnit.MILLISECONDS);
138        }
139    
140        /**
141         * Schedule a Runnable for execution.
142         *
143         * @param runnable Runnable to schedule for execution.
144         * @param delay    delay for first execution since scheduling.
145         * @param interval interval between executions.
146         * @param unit scheduling unit.
147         */
148        public void schedule(final Runnable runnable, long delay, long interval, Unit unit) {
149            log.trace("Scheduling runnable [{0}], interval [{1}], delay [{2}] in [{3}]",
150                      runnable.getClass(), delay, interval, unit);
151            Runnable r = new Runnable() {
152                public void run() {
153                    try {
154                        runnable.run();
155                    }
156                    catch (Exception ex) {
157                        log.warn("Error executing runnable [{0}], {1}", runnable.getClass().getSimpleName(),
158                                 ex.getMessage(), ex);
159                    }
160                }
161            };
162            scheduler.scheduleWithFixedDelay(r, delay * unit.getMillis(), interval * unit.getMillis(),
163                                             TimeUnit.MILLISECONDS);
164        }
165    
166    }
