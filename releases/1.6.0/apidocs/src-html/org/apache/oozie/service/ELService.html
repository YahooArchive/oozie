018    package org.apache.oozie.service;
019    
020    import org.apache.oozie.util.XLog;
021    import org.apache.oozie.util.ELEvaluator;
022    import org.apache.oozie.ErrorCode;
023    import org.apache.hadoop.conf.Configuration;
024    
025    import java.lang.reflect.Field;
026    import java.lang.reflect.Method;
027    import java.lang.reflect.Modifier;
028    import java.util.ArrayList;
029    import java.util.List;
030    
031    /**
032     * The ELService creates {@link ELEvaluator} instances preconfigured with constants and functions defined in the
033     * configuration.
034     * <p/>
035     * The following configuration parameters control the EL service:
036     * <p/>
037     * {@link #CONF_CONSTANTS} list of constant definitions to be available for EL evaluations.
038     * <p/>
039     * {@link #CONF_FUNCTIONS} list of function definitions to be available for EL evalations.
040     * <p/>
041     * Definitions must be separated by a comma, definitions are trimmed.
042     * <p/>
043     * The syntax for a constant definition is <code>PREFIX:NAME=CLASS_NAME#CONSTANT_NAME</code>.
044     * <p/>
045     * The syntax for a constant definition is <code>PREFIX:NAME=CLASS_NAME#METHOD_NAME</code>.
046     */
047    public class ELService implements Service {
048    
049        public static final String CONF_PREFIX = Service.CONF_PREFIX + "ELService.";
050    
051        public static final String CONF_CONSTANTS = CONF_PREFIX + "constants";
052    
053        public static final String CONF_EXT_CONSTANTS = CONF_PREFIX + "ext.constants";
054    
055        public static final String CONF_FUNCTIONS = CONF_PREFIX + "functions";
056    
057        public static final String CONF_EXT_FUNCTIONS = CONF_PREFIX + "ext.functions";
058    
059        private final XLog log = XLog.getLog(getClass());
060    
061        private List<ELConstant> constants;
062        private List<ELFunction> functions;
063    
064        private static class ELConstant {
065            private String name;
066            private Object value;
067    
068            private ELConstant(String prefix, String name, Object value) {
069                if (prefix.length() > 0) {
070                    name = prefix + ":" + name;
071                }
072                this.name = name;
073                this.value = value;
074            }
075        }
076    
077        private static class ELFunction {
078            private String prefix;
079            private String name;
080            private Method method;
081    
082            private ELFunction(String prefix, String name, Method method) {
083                this.prefix = prefix;
084                this.name = name;
085                this.method = method;
086            }
087        }
088    
089        private List<ELService.ELConstant> extractConstants(Configuration conf, String key) throws ServiceException {
090            List<ELService.ELConstant> list = new ArrayList<ELService.ELConstant>();
091            if (conf.get(key, "").trim().length() > 0) {
092                for (String function : conf.getStrings(key)) {
093                    String[] parts = parseDefinition(function);
094                    list.add(new ELConstant(parts[0], parts[1], findConstant(parts[2], parts[3])));
095                    log.trace("Registered prefix:constant[{0}:{1}] for class#field[{2}#{3}]", (Object[]) parts);
096                }
097            }
098            return list;
099        }
100    
101        private List<ELService.ELFunction> extractFunctions(Configuration conf, String key) throws ServiceException {
102            List<ELService.ELFunction> list = new ArrayList<ELService.ELFunction>();
103            if (conf.get(key, "").trim().length() > 0) {
104                for (String function : conf.getStrings(key)) {
105                    String[] parts = parseDefinition(function);
106                    list.add(new ELFunction(parts[0], parts[1], findMethod(parts[2], parts[3])));
107                    log.trace("Registered prefix:constant[{0}:{1}] for class#field[{2}#{3}]", (Object[]) parts);
108                }
109            }
110            return list;
111        }
112    
113        /**
114         * Initialize the EL service.
115         *
116         * @param services services instance.
117         * @throws ServiceException thrown if the EL service could not be initialized.
118         */
119        @Override
120        public synchronized void init(Services services) throws ServiceException {
121            log.trace("Constants and functions registration");
122    
123            constants = new ArrayList<ELConstant>();
124            constants.addAll(extractConstants(services.getConf(), CONF_CONSTANTS));
125            constants.addAll(extractConstants(services.getConf(), CONF_EXT_CONSTANTS));
126    
127            functions = new ArrayList<ELFunction>();
128            functions.addAll(extractFunctions(services.getConf(), CONF_FUNCTIONS));
129            functions.addAll(extractFunctions(services.getConf(), CONF_EXT_FUNCTIONS));
130        }
131    
132        /**
133         * Destroy the EL service.
134         */
135        @Override
136        public void destroy() {
137            constants = null;
138            functions = null;
139        }
140    
141        /**
142         * Return the public interface for EL service.
143         *
144         * @return {@link ELService}.
145         */
146        @Override
147        public Class<? extends Service> getInterface() {
148            return ELService.class;
149        }
150    
151        /**
152         * Return an {@link ELEvaluator} preconfigured with the constants and functions defined in the configuration.
153         *
154         * @return a preconfigured {@link ELEvaluator}.
155         */
156        public ELEvaluator createEvaluator() {
157            ELEvaluator.Context context = new ELEvaluator.Context();
158            for (ELConstant constant : constants) {
159                context.setVariable(constant.name, constant.value);
160            }
161            for (ELFunction function : functions) {
162                context.addFunction(function.prefix, function.name, function.method);
163            }
164            return new ELEvaluator(context);
165        }
166    
167        private static String[] parseDefinition(String str) throws ServiceException {
168            try {
169                str = str.trim();
170                if (!str.contains(":")) {
171                    str = ":" + str;
172                }
173                String[] parts = str.split(":");
174                String prefix = parts[0];
175                parts = parts[1].split("=");
176                String name = parts[0];
177                parts = parts[1].split("#");
178                String klass = parts[0];
179                String method = parts[1];
180                return new String[]{prefix, name, klass, method};
181            }
182            catch (Exception ex) {
183                throw new ServiceException(ErrorCode.E0110, str, ex.getMessage(), ex);
184            }
185        }
186    
187        private static Method findMethod(String className, String methodName) throws ServiceException {
188            Method method = null;
189            try {
190                Class klass = Thread.currentThread().getContextClassLoader().loadClass(className);
191                for (Method m : klass.getMethods()) {
192                    if (m.getName().equals(methodName)) {
193                        method = m;
194                        break;
195                    }
196                }
197                if (method == null) {
198                    throw new ServiceException(ErrorCode.E0111, className, methodName);
199                }
200                if ((method.getModifiers() & (Modifier.PUBLIC | Modifier.STATIC)) != (Modifier.PUBLIC | Modifier.STATIC)) {
201                    throw new ServiceException(ErrorCode.E0112, className, methodName);
202                }
203            }
204            catch (ClassNotFoundException ex) {
205                throw new ServiceException(ErrorCode.E0113, className);
206            }
207            return method;
208        }
209    
210        private static Object findConstant(String className, String constantName) throws ServiceException {
211            try {
212                Class klass = Thread.currentThread().getContextClassLoader().loadClass(className);
213                Field field = klass.getField(constantName);
214                if ((field.getModifiers() & (Modifier.PUBLIC | Modifier.STATIC)) != (Modifier.PUBLIC | Modifier.STATIC)) {
215                    throw new ServiceException(ErrorCode.E0114, className, constantName);
216                }
217                return field.get(null);
218            }
219            catch (IllegalAccessException ex) {
220                throw new IllegalArgumentException(ex);
221            }
222            catch (NoSuchFieldException ex) {
223                throw new ServiceException(ErrorCode.E0115, className, constantName);
224            }
225            catch (ClassNotFoundException ex) {
226                throw new ServiceException(ErrorCode.E0113, className);
227            }
228        }
