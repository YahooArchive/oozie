770        /**
771         * Purge the Workflows Completed older than given days.
772         *
773         * @param olderThanDays number of days for which to preserve the workflows
774         * @throws StoreException
775         */
776        public void purge(final long olderThanDays, final int limit) throws StoreException {
777            doOperation("purge", new Callable<Void>() {
778                public Void call() throws SQLException, StoreException, WorkflowException {
779                    Timestamp maxEndTime = new Timestamp(System.currentTimeMillis() - (olderThanDays * DAY_IN_MS));
780                    Query q = entityManager.createNamedQuery("GET_COMPLETED_WORKFLOWS_OLDER_THAN");
781                    q.setParameter("endTime", maxEndTime);
782                    q.setMaxResults(limit);
783                    List<WorkflowJobBean> workflows = q.getResultList();
784                    int actionDeleted = 0;
785                    if (workflows.size() != 0) {
786                        for (WorkflowJobBean w : workflows) {
787                            String wfId = w.getId();
788                            entityManager.remove(w);
789                            Query g = entityManager.createNamedQuery("DELETE_ACTIONS_FOR_WORKFLOW");
790                            g.setParameter("wfId", wfId);
791                            actionDeleted += g.executeUpdate();
792                        }
793                    }
794                    XLog.getLog(getClass()).debug("ENDED Workflow Purge deleted jobs :" + workflows.size() + " and actions " + actionDeleted);
795                    return null;
796                }
797            });
798        }

800        private <V> V doOperation(String name, Callable<V> command) throws StoreException {
801            try {
802                Instrumentation.Cron cron = new Instrumentation.Cron();
803                cron.start();
804                V retVal;
805                try {
806                    retVal = command.call();
807                }
808                finally {
809                    cron.stop();
810                }
811                Services.get().get(InstrumentationService.class).get().addCron(INSTR_GROUP, name, cron);
812                return retVal;
813            }
814            catch (StoreException ex) {
815                throw ex;
816            }
817            catch (SQLException ex) {
818                throw new StoreException(ErrorCode.E0603, name, ex.getMessage(), ex);
819            }
820            catch (Exception e) {
821                throw new StoreException(ErrorCode.E0607, name, e.getMessage(), e);
822            }
823        }
