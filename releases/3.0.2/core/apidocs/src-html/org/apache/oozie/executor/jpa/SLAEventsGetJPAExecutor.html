015    package org.apache.oozie.executor.jpa;
016    
017    import java.util.List;
018    
019    import javax.persistence.EntityManager;
020    import javax.persistence.Query;
021    
022    import org.apache.oozie.ErrorCode;
023    import org.apache.oozie.SLAEventBean;
024    
025    /**
026     * Load the list of SLAEventBean and return the list.
027     */
028    public class SLAEventsGetJPAExecutor implements JPAExecutor<List<SLAEventBean>> {
029    
030        private int limitLen = 100; // Default
031    
032        public SLAEventsGetJPAExecutor() {
033        }
034    
035        @Override
036        public String getName() {
037            return "SLAEventsGetJPAExecutor";
038        }
039    
040        @Override
041        @SuppressWarnings("unchecked")
042        public List<SLAEventBean> execute(EntityManager em) throws JPAExecutorException {
043            List<SLAEventBean> seBeans;
044            try {
045                Query q = em.createNamedQuery("GET_SLA_EVENTS");
046                q.setMaxResults(limitLen);
047                seBeans = q.getResultList();
048            }
049            catch (Exception e) {
050                throw new JPAExecutorException(ErrorCode.E0603, e.getMessage(), e);
051            }
052            return seBeans;
053        }
054    
055    }
