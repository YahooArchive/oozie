017    /**
018     * Exception thrown by the {@link CoordinatorEngine}.
019     */
020    public class CoordinatorEngineException extends BaseEngineException {
021    
022        /**
023         * Create a coordinator engine exception from a XException.
024         *
025         * @param cause the XException cause.
026         */
027        public CoordinatorEngineException(XException cause) {
028            super(cause);
029        }
030    
031        /**
032         * Create a coordinator engine exception.
033         *
034         * @param errorCode error code.
035         * @param params parameters for the error code message template.
036         */
037        public CoordinatorEngineException(ErrorCode errorCode, Object... params) {
038            super(errorCode, params);
039        }
040    
041    
042    }
