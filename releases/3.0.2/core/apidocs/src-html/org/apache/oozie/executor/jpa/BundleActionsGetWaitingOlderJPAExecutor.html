001    package org.apache.oozie.executor.jpa;
002    
003    import java.sql.Timestamp;
004    import java.util.List;
005    
006    import javax.persistence.EntityManager;
007    import javax.persistence.Query;
008    
009    import org.apache.oozie.BundleActionBean;
010    import org.apache.oozie.ErrorCode;
011    import org.apache.oozie.util.ParamChecker;
012    
013    public class BundleActionsGetWaitingOlderJPAExecutor implements JPAExecutor<List<BundleActionBean>>{
014        private long checkAgeSecs = 0;
015    
016        public BundleActionsGetWaitingOlderJPAExecutor(final long checkAgeSecs) {
017            ParamChecker.notNull(checkAgeSecs, "checkAgeSecs");
018            this.checkAgeSecs = checkAgeSecs;
019        }
020    
021        /* (non-Javadoc)
022         * @see org.apache.oozie.executor.jpa.JPAExecutor#getName()
023         */
024        @Override
025        public String getName() {
026            return "BundleActionsGetWaitingOlderJPAExecutor";
027        }
028    
029        /* (non-Javadoc)
030         * @see org.apache.oozie.executor.jpa.JPAExecutor#execute(javax.persistence.EntityManager)
031         */
032        @Override
033        public List<BundleActionBean> execute(EntityManager em) throws JPAExecutorException {
034            List<BundleActionBean> actions;
035            try {
036                Query q = em.createNamedQuery("GET_BUNDLE_WAITING_ACTIONS_OLDER_THAN");
037                Timestamp ts = new Timestamp(System.currentTimeMillis() - this.checkAgeSecs * 1000);
038                q.setParameter("lastModifiedTime", ts);
039                actions = q.getResultList();
040                return actions;
041            }
042            catch (IllegalStateException e) {
043                throw new JPAExecutorException(ErrorCode.E0601, e.getMessage(), e);
044            }
045        }
046    }
