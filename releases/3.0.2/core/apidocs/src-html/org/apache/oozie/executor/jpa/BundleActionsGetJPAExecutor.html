015    package org.apache.oozie.executor.jpa;
016    
017    import java.util.List;
018    
019    import javax.persistence.EntityManager;
020    import javax.persistence.Query;
021    
022    import org.apache.oozie.BundleActionBean;
023    import org.apache.oozie.ErrorCode;
024    import org.apache.oozie.util.ParamChecker;
025    
026    /**
027     * Load the list of BundleAction return it.
028     */
029    public class BundleActionsGetJPAExecutor implements JPAExecutor<List<BundleActionBean>> {
030    
031        private String bundleId = null;
032    
033        /**
034         * The constructor for class {@link BundleActionsGetJPAExecutor}
035         *
036         * @param bundleId bundle job id
037         */
038        public BundleActionsGetJPAExecutor(String bundleId) {
039            ParamChecker.notNull(bundleId, "bundleId");
040            this.bundleId = bundleId;
041        }
042    
043        /* (non-Javadoc)
044         * @see org.apache.oozie.executor.jpa.JPAExecutor#getName()
045         */
046        @Override
047        public String getName() {
048            return "BundleActionsGetJPAExecutor";
049        }
050    
051        /* (non-Javadoc)
052         * @see org.apache.oozie.executor.jpa.JPAExecutor#execute(javax.persistence.EntityManager)
053         */
054        @Override
055        @SuppressWarnings("unchecked")
056        public List<BundleActionBean> execute(EntityManager em) throws JPAExecutorException {
057            List<BundleActionBean> baBeans;
058            try {
059                Query q = em.createNamedQuery("GET_BUNDLE_ACTIONS_FOR_BUNDLE");
060                q.setParameter("bundleId", bundleId);
061                baBeans = q.getResultList();
062            }
063            catch (Exception e) {
064                throw new JPAExecutorException(ErrorCode.E0603, e);
065            }
066    
067            return baBeans;
068        }
069    }
