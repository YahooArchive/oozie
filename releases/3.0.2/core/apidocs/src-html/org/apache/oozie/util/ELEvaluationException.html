015    package org.apache.oozie.util;
016    
017    /**
018     * Exception thrown by EL functions when they find a transient error.
019     */
020    public class ELEvaluationException extends Exception {
021    
022        /**
023         * Create a EL evaluation exception.
024         *
025         * @param message exception message.
026         * @param cause exception cause.
027         */
028        public ELEvaluationException(String message, Throwable cause) {
029            super(message, cause);
030        }
031    
032        /**
033         * Create a EL evaluation exception.
034         *
035         * @param cause exception cause.
036         */
037        public ELEvaluationException(Throwable cause) {
038            super(cause);
039        }
040    
041    }
