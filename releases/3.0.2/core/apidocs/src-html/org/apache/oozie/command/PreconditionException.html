001    package org.apache.oozie.command;
002    
003    import org.apache.oozie.ErrorCode;
004    import org.apache.oozie.XException;
005    
006    public class PreconditionException extends XException{
007        /**
008         * Create a verifyXCommand exception for verify conditions purposes.
009         *
010         * @param cause the XException cause.
011         */
012        public PreconditionException(XException cause) {
013            super(cause);
014        }
015    
016        /**
017         * Create a verifyXCommand exception for verify conditions purposes.
018         *
019         * @param errorCode error code.
020         * @param params parameters for the error code message template.
021         */
022        public PreconditionException(ErrorCode errorCode, Object... params) {
023            super(errorCode, params);
024        }
025    }
