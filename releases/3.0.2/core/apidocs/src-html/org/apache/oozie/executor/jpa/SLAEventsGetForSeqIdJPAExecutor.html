015    package org.apache.oozie.executor.jpa;
016    
017    import java.util.List;
018    
019    import javax.persistence.EntityManager;
020    import javax.persistence.Query;
021    
022    import org.apache.oozie.ErrorCode;
023    import org.apache.oozie.SLAEventBean;
024    
025    /**
026     * Load the list of SLAEventBean for a seqId and return the list.
027     */
028    public class SLAEventsGetForSeqIdJPAExecutor implements JPAExecutor<List<SLAEventBean>> {
029    
030        private long seqId = 0;
031        private int limitLen = 100; // Default
032        private long[] lastSeqId;
033    
034        public SLAEventsGetForSeqIdJPAExecutor(long seqId, int limitLen, long[] lastSeqId) {
035            this.seqId = seqId;
036            this.limitLen = limitLen;
037            this.lastSeqId = lastSeqId;
038            this.lastSeqId[0] = seqId;
039        }
040    
041        @Override
042        public String getName() {
043            return "SLAEventsGetForSeqIdJPAExecutor";
044        }
045    
046        @Override
047        @SuppressWarnings("unchecked")
048        public List<SLAEventBean> execute(EntityManager em) throws JPAExecutorException {
049            List<SLAEventBean> seBeans;
050            try {
051                Query q = em.createNamedQuery("GET_SLA_EVENT_NEWER_SEQ_LIMITED");
052                q.setParameter("id", seqId);
053                q.setMaxResults(limitLen);
054                seBeans = q.getResultList();
055                for (SLAEventBean j : seBeans) {
056                    lastSeqId[0] = Math.max(lastSeqId[0], j.getEvent_id());
057                }
058            }
059            catch (Exception e) {
060                throw new JPAExecutorException(ErrorCode.E0603, e.getMessage(), e);
061            }
062            return seBeans;
063        }
064    
065    }
