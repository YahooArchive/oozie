015    package org.apache.oozie.action.hadoop;
016    
017    import org.apache.hadoop.mapred.JobConf;
018    import org.apache.oozie.action.ActionExecutor.Context;
019    
020    @SuppressWarnings("deprecation")
021    public abstract class Credentials {
022    
023        /**
024         * This is the interface for all the Credentials implementation. Any new credential implementaion must implement
025         * this function. This function should modify the jobconf which will be used further to pass the credentials to the
026         * tasks while running it. Credentials properties and context is also provided by that user can get all the
027         * necessary configuration.
028         *
029         * @param jobconf job conf where the token will insert into
030         * @param props properties for getting credential token or certificate
031         * @param context workflow context
032         * @throws Exception thrown if failed
033         */
034        public abstract void addtoJobConf(JobConf jobconf, CredentialsProperties props, Context context) throws Exception;
035    }
