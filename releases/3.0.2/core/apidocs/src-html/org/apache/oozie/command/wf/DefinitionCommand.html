015    package org.apache.oozie.command.wf;
016    
017    import org.apache.oozie.WorkflowJobBean;
018    import org.apache.oozie.command.Command;
019    import org.apache.oozie.store.StoreException;
020    import org.apache.oozie.store.WorkflowStore;
021    import org.apache.oozie.store.Store;
022    import org.apache.oozie.util.ParamChecker;
023    import org.apache.oozie.util.XLog;
024    
025    public class DefinitionCommand extends WorkflowCommand<String> {
026        private String id;
027    
028        public DefinitionCommand(String id) {
029            super("definition", "definition", 1, XLog.STD);
030            this.id = ParamChecker.notEmpty(id, "id");
031        }
032    
033        @Override
034        @SuppressWarnings("unchecked")
035        protected String call(WorkflowStore store) throws StoreException {
036            WorkflowJobBean workflow = store.getWorkflow(id, false);
037            return workflow.getWorkflowInstance().getApp().getDefinition();
038        }
039    
040    }
