015    package org.apache.oozie;
016    
017    import java.util.Properties;
018    
019    /**
020     * This class Provide Oozie build information.
021     */
022    public class BuildInfo {
023    
024        public final static String BUILD_USER_NAME = "build.user";
025    
026        public final static String BUILD_TIME = "build.time";
027    
028        public final static String BUILD_VERSION = "build.version";
029    
030        public final static String BUILD_VC_REVISION = "vc.revision";
031    
032        public final static String BUILD_VC_URL = "vc.url";
033    
034        private static final Properties BUILD_INFO;
035    
036        static {
037            BUILD_INFO = new Properties();
038            try {
039                BUILD_INFO.load(BuildInfo.class.getClassLoader().getResourceAsStream("oozie-buildinfo.properties"));
040            }
041            catch (Exception e) {
042                throw new RuntimeException(e);
043            }
044        }
045    
046        private BuildInfo() {
047        }
048    
049        /**
050         * Return the build info properties.
051         *
052         * @return the build info properties.
053         */
054        public static Properties getBuildInfo() {
055            return BUILD_INFO;
056        }
057    
058    }
