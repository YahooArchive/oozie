121        /* (non-Javadoc)
122         * @see org.apache.oozie.command.SubmitTransitionXCommand#submit()
123         */
124        @Override
125        protected String submit() throws CommandException {
126            LOG.info("STARTED Bundle Submit");
127            try {
128                InstrumentUtils.incrJobCounter(getName(), 1, getInstrumentation());
129    
130                XmlUtils.removeComments(this.bundleBean.getOrigJobXml().toString());
131                // Resolving all variables in the job properties.
132                // This ensures the Hadoop Configuration semantics is preserved.
133                XConfiguration resolvedVarsConf = new XConfiguration();
134                for (Map.Entry<String, String> entry : conf) {
135                    resolvedVarsConf.set(entry.getKey(), conf.get(entry.getKey()));
136                }
137                conf = resolvedVarsConf;
138    
139                String resolvedJobXml = resolvedVars(bundleBean.getOrigJobXml(), conf);
140    
141                //verify the uniqueness of coord names
142                verifyCoordNameUnique(resolvedJobXml);
143                this.jobId = storeToDB(bundleBean, resolvedJobXml);
144                LogUtils.setLogInfo(bundleBean, logInfo);
145    
146                if (dryrun) {
147                    Date startTime = bundleBean.getStartTime();
148                    long startTimeMilli = startTime.getTime();
149                    long endTimeMilli = startTimeMilli + (3600 * 1000);
150                    Date jobEndTime = bundleBean.getEndTime();
151                    Date endTime = new Date(endTimeMilli);
152                    if (endTime.compareTo(jobEndTime) > 0) {
153                        endTime = jobEndTime;
154                    }
155                    jobId = bundleBean.getId();
156                    LOG.info("[" + jobId + "]: Update status to PREP");
157                    bundleBean.setStatus(Job.Status.PREP);
158                    try {
159                        new XConfiguration(new StringReader(bundleBean.getConf()));
160                    }
161                    catch (IOException e1) {
162                        LOG.warn("Configuration parse error. read from DB :" + bundleBean.getConf(), e1);
163                    }
164                    String output = bundleBean.getJobXml() + System.getProperty("line.separator");
165                    return output;
166                }
167                else {
168                    if (bundleBean.getKickoffTime() == null) {
169                        // If there is no KickOffTime, default kickoff is NOW.
170                        LOG.debug("Since kickoff time is not defined for job id " + jobId
171                                + ". Queuing and BundleStartXCommand immediately after submission");
172                        queue(new BundleStartXCommand(jobId));
173                    }
174                }
175            }
176            catch (Exception ex) {
177                throw new CommandException(ErrorCode.E1310, ex.getMessage(), ex);
178            }
179            LOG.info("ENDED Bundle Submit");
180            return this.jobId;
181        }

236        protected void eagerVerifyPrecondition() throws CommandException, PreconditionException {
237            try {
238                super.eagerVerifyPrecondition();
239                mergeDefaultConfig();
240                String appXml = readAndValidateXml();
241                bundleBean.setOrigJobXml(appXml);
242                LOG.debug("jobXml after initial validation " + XmlUtils.prettyPrint(appXml).toString());
243            }
244            catch (BundleJobException ex) {
245                LOG.warn("BundleJobException:  ", ex);
246                throw new CommandException(ex);
247            }
248            catch (IllegalArgumentException iex) {
249                LOG.warn("IllegalArgumentException:  ", iex);
250                throw new CommandException(ErrorCode.E1310, iex);
251            }
252            catch (Exception ex) {
253                LOG.warn("Exception:  ", ex);
254                throw new CommandException(ErrorCode.E1310, ex);
255            }

258        /**
259         * Merge default configuration with user-defined configuration.
260         *
261         * @throws CommandException thrown if failed to merge configuration
262         */
263        protected void mergeDefaultConfig() throws CommandException {
264            Path appPath = new Path(conf.get(OozieClient.BUNDLE_APP_PATH));
265            Path configDefault = new Path(appPath.getParent(), CONFIG_DEFAULT);
266            FileSystem fs;
267            try {
268                String user = ParamChecker.notEmpty(conf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
269                String group = ParamChecker.notEmpty(conf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
270                fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, configDefault.toUri(),
271                        new Configuration());
272                if (fs.exists(configDefault)) {
273                    Configuration defaultConf = new XConfiguration(fs.open(configDefault));
274                    PropertiesUtils.checkDisallowedProperties(defaultConf, DISALLOWED_DEFAULT_PROPERTIES);
275                    XConfiguration.injectDefaults(defaultConf, conf);
276                }
277                else {
278                    LOG.info("configDefault Doesn't exist " + configDefault);
279                }
280                PropertiesUtils.checkDisallowedProperties(conf, DISALLOWED_USER_PROPERTIES);
281            }
282            catch (IOException e) {
283                throw new CommandException(ErrorCode.E0702, e.getMessage() + " : Problem reading default config "
284                        + configDefault, e);
285            }
286            catch (HadoopAccessorException e) {
287                throw new CommandException(e);
288            }
289            LOG.debug("Merged CONF :" + XmlUtils.prettyPrint(conf).toString());
290        }

292        /**
293         * Read the application XML and validate against bundle Schema
294         *
295         * @return validated bundle XML
296         * @throws BundleJobException thrown if failed to read or validate xml
297         */
298        private String readAndValidateXml() throws BundleJobException {
299            String appPath = ParamChecker.notEmpty(conf.get(OozieClient.BUNDLE_APP_PATH), OozieClient.BUNDLE_APP_PATH);
300            String bundleXml = readDefinition(appPath);
301            validateXml(bundleXml);
302            return bundleXml;
303        }

305        /**
306         * Read bundle definition.
307         *
308         * @param appPath application path.
309         * @param user user name.
310         * @param group group name.
311         * @param autToken authentication token.
312         * @return bundle definition.
313         * @throws BundleJobException thrown if the definition could not be read.
314         */
315        protected String readDefinition(String appPath) throws BundleJobException {
316            String user = ParamChecker.notEmpty(conf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
317            String group = ParamChecker.notEmpty(conf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
318            //Configuration confHadoop = CoordUtils.getHadoopConf(conf);
319            try {
320                URI uri = new URI(appPath);
321                LOG.debug("user =" + user + " group =" + group);
322                FileSystem fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, uri,
323                        new Configuration());
324                Path p = new Path(uri.getPath());
325    
326                Reader reader = new InputStreamReader(fs.open(p));
327                StringWriter writer = new StringWriter();
328                IOUtils.copyCharStream(reader, writer);
329                return writer.toString();
330            }
331            catch (IOException ex) {
332                LOG.warn("IOException :" + XmlUtils.prettyPrint(conf), ex);
333                throw new BundleJobException(ErrorCode.E1301, ex.getMessage(), ex);
334            }
335            catch (URISyntaxException ex) {
336                LOG.warn("URISyException :" + ex.getMessage());
337                throw new BundleJobException(ErrorCode.E1302, appPath, ex.getMessage(), ex);
338            }
339            catch (HadoopAccessorException ex) {
340                throw new BundleJobException(ex);
341            }
342            catch (Exception ex) {
343                LOG.warn("Exception :", ex);
344                throw new BundleJobException(ErrorCode.E1301, ex.getMessage(), ex);
345            }
346        }

348        /**
349         * Validate against Bundle XSD file
350         *
351         * @param xmlContent input bundle xml
352         * @throws BundleJobException thrown if failed to validate xml
353         */
354        private void validateXml(String xmlContent) throws BundleJobException {
355            javax.xml.validation.Schema schema = Services.get().get(SchemaService.class).getSchema(SchemaName.BUNDLE);
356            Validator validator = schema.newValidator();
357            try {
358                validator.validate(new StreamSource(new StringReader(xmlContent)));
359            }
360            catch (SAXException ex) {
361                LOG.warn("SAXException :", ex);
362                throw new BundleJobException(ErrorCode.E0701, ex.getMessage(), ex);
363            }
364            catch (IOException ex) {
365                LOG.warn("IOException :", ex);
366                throw new BundleJobException(ErrorCode.E0702, ex.getMessage(), ex);
367            }
368        }

370        /**
371         * Write a Bundle Job into database
372         *
373         * @param Bundle job bean
374         * @return job id
375         * @throws CommandException thrown if failed to store bundle job bean to db
376         */
377        private String storeToDB(BundleJobBean bundleJob, String resolvedJobXml) throws CommandException {
378            try {
379                jobId = Services.get().get(UUIDService.class).generateId(ApplicationType.BUNDLE);
380    
381                bundleJob.setId(jobId);
382                bundleJob.setAuthToken(this.authToken);
383                bundleJob.setAppName(XmlUtils.parseXml(bundleBean.getOrigJobXml()).getAttributeValue("name"));
384                bundleJob.setAppName(bundleJob.getAppName());
385                bundleJob.setAppPath(conf.get(OozieClient.BUNDLE_APP_PATH));
386                // bundleJob.setStatus(BundleJob.Status.PREP); //This should be set in parent class.
387                bundleJob.setCreatedTime(new Date());
388                bundleJob.setUser(conf.get(OozieClient.USER_NAME));
389                bundleJob.setGroup(conf.get(OozieClient.GROUP_NAME));
390                bundleJob.setConf(XmlUtils.prettyPrint(conf).toString());
391                bundleJob.setJobXml(resolvedJobXml);
392                Element jobElement = XmlUtils.parseXml(resolvedJobXml);
393                Element controlsElement = jobElement.getChild("controls", jobElement.getNamespace());
394                if (controlsElement != null) {
395                    Element kickoffTimeElement = controlsElement.getChild("kick-off-time", jobElement.getNamespace());
396                    if (kickoffTimeElement != null && !kickoffTimeElement.getValue().isEmpty()) {
397                        Date kickoffTime = DateUtils.parseDateUTC(kickoffTimeElement.getValue());
398                        bundleJob.setKickoffTime(kickoffTime);
399                    }
400                }
401                bundleJob.setLastModifiedTime(new Date());
402    
403                if (!dryrun) {
404                    jpaService.execute(new BundleJobInsertJPAExecutor(bundleJob));
405                }
406            }
407            catch (Exception ex) {
408                throw new CommandException(ErrorCode.E1301, ex.getMessage(), ex);
409            }
410            return jobId;
411        }

421        /**
422         * Resolve job xml with conf
423         *
424         * @param bundleXml bundle job xml
425         * @param conf job configuration
426         * @return resolved job xml
427         * @throws BundleJobException thrown if failed to resolve variables
428         */
429        private String resolvedVars(String bundleXml, Configuration conf) throws BundleJobException {
430            try {
431                ELEvaluator eval = createEvaluator(conf);
432                return eval.evaluate(bundleXml, String.class);
433            }
434            catch (Exception e) {
435                throw new BundleJobException(ErrorCode.E1004, e.getMessage(), e);
436            }
437        }

439        /**
440         * Create ELEvaluator
441         *
442         * @param conf job configuration
443         * @return ELEvaluator the evaluator for el function
444         * @throws BundleJobException thrown if failed to create evaluator
445         */
446        public ELEvaluator createEvaluator(Configuration conf) throws BundleJobException {
447            ELEvaluator eval;
448            ELEvaluator.Context context;
449            try {
450                context = new ELEvaluator.Context();
451                eval = new ELEvaluator(context);
452                for (Map.Entry<String, String> entry : conf) {
453                    eval.setVariable(entry.getKey(), entry.getValue());
454                }
455            }
456            catch (Exception e) {
457                throw new BundleJobException(ErrorCode.E1004, e.getMessage(), e);
458            }
459            return eval;
460        }

462        /**
463         * Verify the uniqueness of coordinator names
464         *
465         * @param resolved job xml
466         * @throws CommandException thrown if failed to verify the uniqueness of coordinator names
467         */
468        @SuppressWarnings("unchecked")
469        private Void verifyCoordNameUnique(String resolvedJobXml) throws CommandException {
470            Set<String> set = new HashSet<String>();
471            try {
472                Element bAppXml = XmlUtils.parseXml(resolvedJobXml);
473                List<Element> coordElems = bAppXml.getChildren("coordinator", bAppXml.getNamespace());
474                for (Element elem : coordElems) {
475                    Attribute name = elem.getAttribute("name");
476                    if (name != null) {
477                        if (set.contains(name.getValue())) {
478                            throw new CommandException(ErrorCode.E1304, name);
479                        }
480                        set.add(name.getValue());
481                    }
482                    else {
483                        throw new CommandException(ErrorCode.E1305);
484                    }
485                }
486            }
487            catch (JDOMException jex) {
488                throw new CommandException(ErrorCode.E1301, jex);
489            }
490    
491            return null;
492        }
