035    /**
036     * Base class for synchronous and asynchronous commands.
037     * <p/>
038     * It enables by API the following pattern:
039     * <p/>
040     * <ul>
041     * <li>single execution: a command instance can be executed only once</li>
042     * <li>eager data loading: loads data for eager precondition check</li>
043     * <li>eager precondition check: verify precondition before obtaining lock</li>
044     * <li>data loading: loads data for precondition check and execution</li>
045     * <li>precondition check: verifies precondition for execution is still met</li>
046     * <li>locking: obtains exclusive lock on key before executing the command</li>
047     * <li>execution: command logic</li>
048     * </ul>
049     * <p/>
050     * It has built in instrumentation and logging.
051     */
052    public abstract class XCommand<T> implements XCallable<T> {
053        public static final String DEFAULT_LOCK_TIMEOUT = "oozie.command.default.lock.timeout";
054    
055        public static final String INSTRUMENTATION_GROUP = "commands";
056    
057        public static final Long DEFAULT_REQUEUE_DELAY = 10L;
058    
059        public XLog LOG = XLog.getLog(getClass());
060    
061        private String key;
062        private String name;
063        private int priority;
064        private String type;
065        private long createdTime;
066        private MemoryLocks.LockToken lock;
067        private boolean used = false;
068    
069        private Map<Long, List<XCommand<?>>> commandQueue;
070        protected boolean dryrun = false;
071        protected Instrumentation instrumentation;
072    
073        protected XLog.Info logInfo;
074    
075        /**
076         * Create a command.
077         *
078         * @param name command name.
079         * @param type command type.
080         * @param priority command priority.
081         */
082        public XCommand(String name, String type, int priority) {
083            this.name = name;
084            this.type = type;
085            this.priority = priority;
086            this.key = name + "_" + UUID.randomUUID();
087            createdTime = System.currentTimeMillis();
088            logInfo = new XLog.Info();
089            instrumentation = Services.get().get(InstrumentationService.class).get();
090        }

098         */
099        public XCommand(String name, String type, int priority, boolean dryrun) {
100            this(name, type, priority);
101            this.dryrun = dryrun;

146        /**
147         * Queue a command for execution after the current command execution completes.
148         * <p/>
149         * All commands queued during the execution of the current command will be queued for a single serial execution.
150         * <p/>
151         * If the command execution throws an exception, no command will be effectively queued.
152         *
153         * @param command command to queue.
154         */
155        protected void queue(XCommand<?> command) {
156            queue(command, 0);
157        }
158    
159        /**
160         * Queue a command for delayed execution after the current command execution completes.
161         * <p/>
162         * All commands queued during the execution of the current command with the same delay will be queued for a single
163         * serial execution.
164         * <p/>
165         * If the command execution throws an exception, no command will be effectively queued.
166         *
167         * @param command command to queue.
168         * @param msDelay delay in milliseconds.
169         */
170        protected void queue(XCommand<?> command, long msDelay) {
171            if (commandQueue == null) {
172                commandQueue = new HashMap<Long, List<XCommand<?>>>();
173            }
174            List<XCommand<?>> list = commandQueue.get(msDelay);
175            if (list == null) {
176                list = new ArrayList<XCommand<?>>();
177                commandQueue.put(msDelay, list);
178            }
179            list.add(command);
180        }
181    
182        /**
183         * Obtain an exclusive lock on the {link #getEntityKey}.
184         * <p/>
185         * A timeout of {link #getLockTimeOut} is used when trying to obtain the lock.
186         *
187         * @throws InterruptedException thrown if an interruption happened while trying to obtain the lock
188         * @throws CommandException thrown i the lock could not be obtained.
189         */
190        private void acquireLock() throws InterruptedException, CommandException {
191            lock = Services.get().get(MemoryLocksService.class).getWriteLock(getEntityKey(), getLockTimeOut());
192            if (lock == null) {
193                Instrumentation instrumentation = Services.get().get(InstrumentationService.class).get();
194                instrumentation.incr(INSTRUMENTATION_GROUP, getName() + ".lockTimeOut", 1);
195                if (isReQueueRequired()) {
196                    //if not acquire the lock, re-queue itself with default delay
197                    resetUsed();
198                    queue(this, getRequeueDelay());
199                    LOG.debug("Could not get lock [{0}], timed out [{1}]ms, and requeue itself [{2}]", this.toString(), getLockTimeOut(), getName());
200                } else {
201                    throw new CommandException(ErrorCode.E0606, this.toString(), getLockTimeOut());
202                }
203            } else {
204                LOG.debug("Acquired lock for [{0}] in [{1}]", getEntityKey(), getName());
205            }
206        }
207    
208        /**
209         * Release the lock on the {link #getEntityKey}.
210         */
211        private void releaseLock() {
212            if (lock != null) {
213                lock.release();
214                LOG.debug("Released lock for [{0}] in [{1}]", getEntityKey(), getName());
215            }
216        }
217    
218        /**
219         * Implements the XCommand life-cycle.
220         *
221         * @return the {link #execute} return value.
222         * @throws Exception thrown if the command could not be executed.
223         */
224        @Override
225        public final T call() throws CommandException {
226            if (used) {
227                throw new IllegalStateException(this.getClass().getSimpleName() + " already used.");
228            }
229            used = true;
230            commandQueue = null;
231            Instrumentation instrumentation = Services.get().get(InstrumentationService.class).get();
232            instrumentation.incr(INSTRUMENTATION_GROUP, getName() + ".executions", 1);
233            Instrumentation.Cron callCron = new Instrumentation.Cron();
234            try {
235                callCron.start();
236                eagerLoadState();
237                eagerVerifyPrecondition();
238                try {
239                    T ret = null;
240                    if (isLockRequired()) {
241                        Instrumentation.Cron acquireLockCron = new Instrumentation.Cron();
242                        acquireLockCron.start();
243                        acquireLock();
244                        acquireLockCron.stop();
245                        instrumentation.addCron(INSTRUMENTATION_GROUP, getName() + ".acquireLock", acquireLockCron);
246                    }
247                    if (!isLockRequired() || (isLockRequired() && lock != null)) {
248                        LOG.debug("Load state for [{0}]", getEntityKey());
249                        loadState();
250                        LOG = XLog.getLog(getClass());
251                        LOG.debug("Precondition check for command [{0}] key [{1}]", getName(), getEntityKey());
252                        verifyPrecondition();
253                        LOG.debug("Execute command [{0}] key [{1}]", getName(), getEntityKey());
254                        Instrumentation.Cron executeCron = new Instrumentation.Cron();
255                        executeCron.start();
256                        ret = execute();
257                        executeCron.stop();
258                        instrumentation.addCron(INSTRUMENTATION_GROUP, getName() + ".execute", executeCron);
259                    }
260                    if (commandQueue != null) {
261                        CallableQueueService callableQueueService = Services.get().get(CallableQueueService.class);
262                        for (Map.Entry<Long, List<XCommand<?>>> entry : commandQueue.entrySet()) {
263                            LOG.debug("Queuing [{0}] commands with delay [{1}]ms", entry.getValue().size(), entry.getKey());
264                            if (!callableQueueService.queueSerial(entry.getValue(), entry.getKey())) {
265                                LOG.warn("Could not queue [{0}] commands with delay [{1}]ms, queue full", entry.getValue()
266                                        .size(), entry.getKey());
267                            }
268                        }
269                    }
270                    return ret;
271                }
272                finally {
273                    if (isLockRequired()) {
274                        releaseLock();
275                    }
276                }
277            }
278            catch(PreconditionException pex){
279                LOG.warn(pex.getMessage().toString() + ", Error Code: " + pex.getErrorCode().toString(), pex);
280                instrumentation.incr(INSTRUMENTATION_GROUP, getName() + ".preconditionfailed", 1);
281                return null;
282            }
283            catch (XException ex) {
284                LOG.error("XException, ", ex);
285                instrumentation.incr(INSTRUMENTATION_GROUP, getName() + ".xexceptions", 1);
286                if (ex instanceof CommandException) {
287                    throw (CommandException) ex;
288                }
289                else {
290                    throw new CommandException(ex);
291                }
292            }
293            catch (Exception ex) {
294                LOG.error("Exception, ", ex);
295                instrumentation.incr(INSTRUMENTATION_GROUP, getName() + ".exceptions", 1);
296                throw new CommandException(ErrorCode.E0607, ex);
297            }
298            finally {
299                FaultInjection.deactivate("org.apache.oozie.command.SkipCommitFaultInjection");
300                callCron.stop();
301                instrumentation.addCron(INSTRUMENTATION_GROUP, getName() + ".call", callCron);
302            }
303        }

305        /**
306         * Return the time out when acquiring a lock.
307         * <p/>
308         * The value is loaded from the Oozie configuration, the property {link #DEFAULT_LOCK_TIMEOUT}.
309         * <p/>
310         * Subclasses should override this method if they want to use a different time out.
311         *
312         * @return the lock time out in milliseconds.
313         */
314        protected long getLockTimeOut() {
315            return Services.get().getConf().getLong(DEFAULT_LOCK_TIMEOUT, 5 * 1000);
316        }
317    
318        /**
319         * Indicate if the the command requires locking.
320         * <p/>
321         * Subclasses should override this method if they require locking.
322         *
323         * @return <code>true/false</code>
324         */
325        protected abstract boolean isLockRequired();
326    
327        /**
328         * Return the entity key for the command.
329         * <p/>
330         *
331         * @return the entity key for the command.
332         */
333        protected abstract String getEntityKey();
334    
335        /**
336         * Indicate if the the command requires to requeue itself if the lock is not acquired.
337         * <p/>
338         * Subclasses should override this method if they don't want to requeue.
339         * <p/>
340         * Default is true.
341         *
342         * @return <code>true/false</code>
343         */
344        protected boolean isReQueueRequired() {
345            return true;
346        }
347    
348        /**
349         * Load the necessary state to perform an eager precondition check.
350         * <p/>
351         * This implementation does a NOP.
352         * <p/>
353         * Subclasses should override this method and load the state needed to do an eager precondition check.
354         * <p/>
355         * A trivial implementation is calling {link #loadState}.
356         */
357        protected void eagerLoadState() throws CommandException{
358        }
359    
360        /**
361         * Verify the precondition for the command before obtaining a lock.
362         * <p/>
363         * This implementation does a NOP.
364         * <p/>
365         * A trivial implementation is calling {link #verifyPrecondition}.
366         *
367         * @throws CommandException thrown if the precondition is not met.
368         */
369        protected void eagerVerifyPrecondition() throws CommandException,PreconditionException {
370        }
371    
372        /**
373         * Load the necessary state to perform the precondition check and to execute the command.
374         * <p/>
375         * Subclasses must implement this method and load the state needed to do the precondition check and execute the
376         * command.
377         */
378        protected abstract void loadState() throws CommandException;
379    
380        /**
381         * Verify the precondition for the command after a lock has been obtain, just before executing the command.
382         * <p/>
383         *
384         * @throws CommandException thrown if the precondition is not met.
385         */
386        protected abstract void verifyPrecondition() throws CommandException,PreconditionException;
387    
388        /**
389         * Command execution body.
390         * <p/>
391         * This method will be invoked after the {link #loadState} and {link #verifyPrecondition} methods.
392         * <p/>
393         * If the command requires locking, this method will be invoked ONLY if the lock has been acquired.
394         *
395         * @return a return value from the execution of the command, only meaningful if the command is executed
396         *         synchronously.
397         * @throws CommandException thrown if the command execution failed.
398         */
399        protected abstract T execute() throws CommandException;
400    
401    
402        /**
403         * Return the {@link Instrumentation} instance in use.
404         *
405         * @return the {@link Instrumentation} instance in use.
406         */
407        protected Instrumentation getInstrumentation() {
408            return instrumentation;
409        }
410    
411        /**
412         * @param used set false to the used
413         */
414        public void resetUsed() {
415            this.used = false;
416        }
417    
418    
419        /**
420         * Return the delay time for requeue
421         *
422         * @return delay time when requeue itself
423         */
424        protected Long getRequeueDelay() {
425            return DEFAULT_REQUEUE_DELAY;
426        }

435            return this.key;
436        }

444            return LOG;
