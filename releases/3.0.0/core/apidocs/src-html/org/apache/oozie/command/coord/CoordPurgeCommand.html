015    package org.apache.oozie.command.coord;
016    
017    import org.apache.oozie.store.CoordinatorStore;
018    import org.apache.oozie.store.StoreException;
019    import org.apache.oozie.util.XLog;
020    import org.apache.oozie.command.CommandException;
021    
022    public class CoordPurgeCommand extends CoordinatorCommand<Void> {
023        private static XLog LOG = XLog.getLog(CoordPurgeCommand.class);
024        private int olderThan;
025        private int limit;
026    
027        public CoordPurgeCommand(int olderThan, int limit) {
028            super("coord_purge", "coord_purge", 0, XLog.OPS);
029            this.olderThan = olderThan;
030            this.limit = limit;
031        }
032    
033        protected Void call(CoordinatorStore store) throws StoreException, CommandException {
034            LOG.debug("STARTED Coord Purge to purge Jobs older than [{0}] days.", olderThan);
035            store.purge(olderThan, limit);
036            LOG.debug("Coord-Purge succeeded");
037            return null;
038        }
039    
040    }
