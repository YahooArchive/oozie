066        /* (non-Javadoc)
067         * @see org.apache.oozie.command.XCommand#execute()
068         */
069        @Override
070        protected Void execute() throws CommandException {
071            LOG.info("[" + actionId + "]::ActionInputCheck:: Action is in WAITING state.");
072    
073            // this action should only get processed if current time > nominal time;
074            // otherwise, requeue this action for delay execution;
075            Date nominalTime = coordAction.getNominalTime();
076            Date currentTime = new Date();
077            if (nominalTime.compareTo(currentTime) > 0) {
078                queue(new CoordActionInputCheckXCommand(coordAction.getId()), Math.max(
079                        (nominalTime.getTime() - currentTime.getTime()), COMMAND_REQUEUE_INTERVAL));
080                // update lastModifiedTime
081                coordAction.setLastModifiedTime(new Date());
082                try {
083                    jpaService.execute(new org.apache.oozie.executor.jpa.CoordActionUpdateJPAExecutor(coordAction));
084                }
085                catch (JPAExecutorException e) {
086                    throw new CommandException(e);
087                }
088                LOG.info("[" + actionId
089                        + "]::ActionInputCheck:: nominal Time is newer than current time, so requeue and wait. Current="
090                        + currentTime + ", nominal=" + nominalTime);
091    
092                return null;
093            }
094    
095            StringBuilder actionXml = new StringBuilder(coordAction.getActionXml());
096            Instrumentation.Cron cron = new Instrumentation.Cron();
097            try {
098                Configuration actionConf = new XConfiguration(new StringReader(coordAction.getRunConf()));
099                cron.start();
100                StringBuilder existList = new StringBuilder();
101                StringBuilder nonExistList = new StringBuilder();
102                StringBuilder nonResolvedList = new StringBuilder();
103                CoordCommandUtils.getResolvedList(coordAction.getMissingDependencies(), nonExistList, nonResolvedList);
104    
105                LOG.info("[" + actionId + "]::CoordActionInputCheck:: Missing deps:" + nonExistList.toString() + " "
106                        + nonResolvedList.toString());
107                Date actualTime = new Date();
108                boolean status = checkInput(actionXml, existList, nonExistList, actionConf, actualTime);
109                coordAction.setLastModifiedTime(actualTime);
110                coordAction.setActionXml(actionXml.toString());
111                if (nonResolvedList.length() > 0 && status == false) {
112                    nonExistList.append(CoordCommandUtils.RESOLVED_UNRESOLVED_SEPARATOR).append(nonResolvedList);
113                }
114                coordAction.setMissingDependencies(nonExistList.toString());
115                if (status == true) {
116                    coordAction.setStatus(CoordinatorAction.Status.READY);
117                    // pass jobID to the CoordActionReadyXCommand
118                    queue(new CoordActionReadyXCommand(coordAction.getJobId()), 100);
119                }
120                else {
121                    long waitingTime = (actualTime.getTime() - Math.max(coordAction.getNominalTime().getTime(), coordAction
122                            .getCreatedTime().getTime()))
123                            / (60 * 1000);
124                    int timeOut = coordAction.getTimeOut();
125                    if ((timeOut >= 0) && (waitingTime > timeOut)) {
126                        queue(new CoordActionTimeOutXCommand(coordAction), 100);
127                        coordAction.setStatus(CoordinatorAction.Status.TIMEDOUT);
128                    }
129                    else {
130                        queue(new CoordActionInputCheckXCommand(coordAction.getId()), COMMAND_REQUEUE_INTERVAL);
131                    }
132                }
133                coordAction.setLastModifiedTime(new Date());
134                jpaService.execute(new org.apache.oozie.executor.jpa.CoordActionUpdateJPAExecutor(coordAction));
135            }
136            catch (Exception e) {
137                throw new CommandException(ErrorCode.E1005, e.getMessage(), e);
138            }
139            cron.stop();
140    
141            return null;
142        }

144        /**
145         * To check the list of input paths if all of them exist
146         *
147         * @param actionXml action xml
148         * @param existList the list of existed paths
149         * @param nonExistList the list of non existed paths
150         * @param conf action configuration
151         * @param actualTime current time
152         * @return true if all input paths are existed
153         * @throws Exception thrown of unable to check input path
154         */
155        protected boolean checkInput(StringBuilder actionXml, StringBuilder existList, StringBuilder nonExistList,
156                Configuration conf, Date actualTime) throws Exception {
157            Element eAction = XmlUtils.parseXml(actionXml.toString());
158            boolean allExist = checkResolvedUris(eAction, existList, nonExistList, conf);
159            if (allExist) {
160                LOG.debug("[" + actionId + "]::ActionInputCheck:: Checking Latest/future");
161                allExist = checkUnresolvedInstances(eAction, conf, actualTime);
162            }
163            if (allExist == true) {
164                materializeDataProperties(eAction, conf);
165                actionXml.replace(0, actionXml.length(), XmlUtils.prettyPrint(eAction).toString());
166            }
167            return allExist;
168        }

170        /**
171         * Materialize data properties defined in <action> tag. it includes dataIn(<DS>) and dataOut(<DS>) it creates a list
172         * of files that will be needed.
173         *
174         * @param eAction action element
175         * @param conf action configuration
176         * @throws Exception thrown if failed to resolve data properties
177         * @update modify 'Action' element with appropriate list of files.
178         */
179        @SuppressWarnings("unchecked")
180        private void materializeDataProperties(Element eAction, Configuration conf) throws Exception {
181            ELEvaluator eval = CoordELEvaluator.createDataEvaluator(eAction, conf, actionId);
182            Element configElem = eAction.getChild("action", eAction.getNamespace()).getChild("workflow",
183                    eAction.getNamespace()).getChild("configuration", eAction.getNamespace());
184            if (configElem != null) {
185                for (Element propElem : (List<Element>) configElem.getChildren("property", configElem.getNamespace())) {
186                    resolveTagContents("value", propElem, eval);
187                }
188            }
189        }
190    
191        /**
192         * To resolve property value which contains el functions
193         *
194         * @param tagName tag name
195         * @param elem the child element of "property" element
196         * @param eval el functions evaluator
197         * @throws Exception thrown if unable to resolve tag value
198         */
199        private void resolveTagContents(String tagName, Element elem, ELEvaluator eval) throws Exception {
200            if (elem == null) {
201                return;
202            }
203            Element tagElem = elem.getChild(tagName, elem.getNamespace());
204            if (tagElem != null) {
205                String updated = CoordELFunctions.evalAndWrap(eval, tagElem.getText());
206                tagElem.removeContent();
207                tagElem.addContent(updated);
208            }
209            else {
210                LOG.warn(" Value NOT FOUND " + tagName);
211            }
212        }

214        /**
215         * Check if any unsolved paths under data output. Resolve the unresolved data input paths.
216         *
217         * @param eAction action element
218         * @param actionConf action configuration
219         * @param actualTime current time
220         * @return true if successful to resolve input and output paths
221         * @throws Exception thrown if failed to resolve data input and output paths
222         */
223        @SuppressWarnings("unchecked")
224        private boolean checkUnresolvedInstances(Element eAction, Configuration actionConf, Date actualTime)
225                throws Exception {
226            String strAction = XmlUtils.prettyPrint(eAction).toString();
227            Date nominalTime = DateUtils.parseDateUTC(eAction.getAttributeValue("action-nominal-time"));
228            StringBuffer resultedXml = new StringBuffer();
229    
230            boolean ret;
231            Element inputList = eAction.getChild("input-events", eAction.getNamespace());
232            if (inputList != null) {
233                ret = materializeUnresolvedEvent(inputList.getChildren("data-in", eAction.getNamespace()), nominalTime,
234                        actualTime, actionConf);
235                if (ret == false) {
236                    resultedXml.append(strAction);
237                    return false;
238                }
239            }
240    
241            // Using latest() or future() in output-event is not intuitive.
242            // We need to make sure, this assumption is correct.
243            Element outputList = eAction.getChild("output-events", eAction.getNamespace());
244            if (outputList != null) {
245                for (Element dEvent : (List<Element>) outputList.getChildren("data-out", eAction.getNamespace())) {
246                    if (dEvent.getChild("unresolved-instances", dEvent.getNamespace()) != null) {
247                        throw new CommandException(ErrorCode.E1006, "coord:latest()/future()",
248                                " not permitted in output-event ");
249                    }
250                }
251            }
252            return true;
253        }

255        /**
256         * Resolve the list of data input paths
257         *
258         * @param eDataEvents the list of data input elements
259         * @param nominalTime action nominal time
260         * @param actualTime current time
261         * @param conf action configuration
262         * @return true if all unresolved URIs can be resolved
263         * @throws Exception thrown if failed to resolve data input paths
264         */
265        @SuppressWarnings("unchecked")
266        private boolean materializeUnresolvedEvent(List<Element> eDataEvents, Date nominalTime, Date actualTime,
267                Configuration conf) throws Exception {
268            for (Element dEvent : eDataEvents) {
269                if (dEvent.getChild("unresolved-instances", dEvent.getNamespace()) == null) {
270                    continue;
271                }
272                ELEvaluator eval = CoordELEvaluator.createLazyEvaluator(actualTime, nominalTime, dEvent, conf);
273                String uresolvedInstance = dEvent.getChild("unresolved-instances", dEvent.getNamespace()).getTextTrim();
274                String unresolvedList[] = uresolvedInstance.split(CoordELFunctions.INSTANCE_SEPARATOR);
275                StringBuffer resolvedTmp = new StringBuffer();
276                for (int i = 0; i < unresolvedList.length; i++) {
277                    String ret = CoordELFunctions.evalAndWrap(eval, unresolvedList[i]);
278                    Boolean isResolved = (Boolean) eval.getVariable("is_resolved");
279                    if (isResolved == false) {
280                        LOG.info("[" + actionId + "]::Cannot resolve: " + ret);
281                        return false;
282                    }
283                    if (resolvedTmp.length() > 0) {
284                        resolvedTmp.append(CoordELFunctions.INSTANCE_SEPARATOR);
285                    }
286                    resolvedTmp.append((String) eval.getVariable("resolved_path"));
287                }
288                if (resolvedTmp.length() > 0) {
289                    if (dEvent.getChild("uris", dEvent.getNamespace()) != null) {
290                        resolvedTmp.append(CoordELFunctions.INSTANCE_SEPARATOR).append(
291                                dEvent.getChild("uris", dEvent.getNamespace()).getTextTrim());
292                        dEvent.removeChild("uris", dEvent.getNamespace());
293                    }
294                    Element uriInstance = new Element("uris", dEvent.getNamespace());
295                    uriInstance.addContent(resolvedTmp.toString());
296                    dEvent.getContent().add(1, uriInstance);
297                }
298                dEvent.removeChild("unresolved-instances", dEvent.getNamespace());
299            }
300    
301            return true;
302        }

304        /**
305         * Check all resolved URIs existence
306         *
307         * @param eAction action element
308         * @param existList the list of existed paths
309         * @param nonExistList the list of paths to check existence
310         * @param conf action configuration
311         * @return true if all nonExistList paths exist
312         * @throws IOException thrown if unable to access the path
313         */
314        private boolean checkResolvedUris(Element eAction, StringBuilder existList, StringBuilder nonExistList,
315                Configuration conf) throws IOException {
316            LOG.info("[" + actionId + "]::ActionInputCheck:: In checkResolvedUris...");
317            Element inputList = eAction.getChild("input-events", eAction.getNamespace());
318            if (inputList != null) {
319                if (nonExistList.length() > 0) {
320                    checkListOfPaths(existList, nonExistList, conf);
321                }
322                return nonExistList.length() == 0;
323            }
324            return true;
325        }

327        /**
328         * Check a list of non existed paths and add to exist list if it exists
329         *
330         * @param existList the list of existed paths
331         * @param nonExistList the list of paths to check existence
332         * @param conf action configuration
333         * @return true if all nonExistList paths exist
334         * @throws IOException thrown if unable to access the path
335         */
336        private boolean checkListOfPaths(StringBuilder existList, StringBuilder nonExistList, Configuration conf)
337                throws IOException {
338    
339            String[] uriList = nonExistList.toString().split(CoordELFunctions.INSTANCE_SEPARATOR);
340            if (uriList[0] != null) {
341                LOG.info("[" + actionId + "]::ActionInputCheck:: In checkListOfPaths: " + uriList[0] + " is Missing.");
342            }
343    
344            nonExistList.delete(0, nonExistList.length());
345            boolean allExists = true;
346            String existSeparator = "", nonExistSeparator = "";
347            for (int i = 0; i < uriList.length; i++) {
348                if (allExists) {
349                    allExists = pathExists(uriList[i], conf);
350                    LOG.info("[" + actionId + "]::ActionInputCheck:: File:" + uriList[i] + ", Exists? :" + allExists);
351                }
352                if (allExists) {
353                    existList.append(existSeparator).append(uriList[i]);
354                    existSeparator = CoordELFunctions.INSTANCE_SEPARATOR;
355                }
356                else {
357                    nonExistList.append(nonExistSeparator).append(uriList[i]);
358                    nonExistSeparator = CoordELFunctions.INSTANCE_SEPARATOR;
359                }
360            }
361            return allExists;
362        }

364        /**
365         * Check if given path exists
366         *
367         * @param sPath uri path
368         * @param actionConf action configuration
369         * @return true if path exists
370         * @throws IOException thrown if unable to access the path
371         */
372        private boolean pathExists(String sPath, Configuration actionConf) throws IOException {
373            LOG.debug("checking for the file " + sPath);
374            Path path = new Path(sPath);
375            String user = ParamChecker.notEmpty(actionConf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
376            String group = ParamChecker.notEmpty(actionConf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
377            try {
378                return Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, path.toUri(),
379                        new Configuration()).exists(path);
380            }
381            catch (HadoopAccessorException e) {
382                throw new IOException(e);
383            }
384        }

386        /**
387         * The function create a list of URIs separated by "," using the instances time stamp and URI-template
388         *
389         * @param event : <data-in> event
390         * @param instances : List of time stamp seprated by ","
391         * @param unresolvedInstances : list of instance with latest/future function
392         * @return : list of URIs separated by ",".
393         * @throws Exception thrown if failed to create URIs from unresolvedInstances
394         */
395        @SuppressWarnings("unused")
396        private String createURIs(Element event, String instances, StringBuilder unresolvedInstances) throws Exception {
397            if (instances == null || instances.length() == 0) {
398                return "";
399            }
400            String[] instanceList = instances.split(CoordELFunctions.INSTANCE_SEPARATOR);
401            StringBuilder uris = new StringBuilder();
402    
403            for (int i = 0; i < instanceList.length; i++) {
404                int funcType = CoordCommandUtils.getFuncType(instanceList[i]);
405                if (funcType == CoordCommandUtils.LATEST || funcType == CoordCommandUtils.FUTURE) {
406                    if (unresolvedInstances.length() > 0) {
407                        unresolvedInstances.append(CoordELFunctions.INSTANCE_SEPARATOR);
408                    }
409                    unresolvedInstances.append(instanceList[i]);
410                    continue;
411                }
412                ELEvaluator eval = CoordELEvaluator.createURIELEvaluator(instanceList[i]);
413                if (uris.length() > 0) {
414                    uris.append(CoordELFunctions.INSTANCE_SEPARATOR);
415                }
416                uris.append(CoordELFunctions.evalAndWrap(eval, event.getChild("dataset", event.getNamespace()).getChild(
417                        "uri-template", event.getNamespace()).getTextTrim()));
418            }
419            return uris.toString();
420        }

449        @Override
450        protected void loadState() throws CommandException {
451            if (jpaService == null) {
452                jpaService = Services.get().get(JPAService.class);
453            }
454            try {
455                coordAction = jpaService.execute(new CoordActionGetJPAExecutor(actionId));
456                coordJob = jpaService.execute(new CoordJobGetJPAExecutor(coordAction.getJobId()));
457            }
458            catch (JPAExecutorException je) {
459                throw new CommandException(je);
460            }
461            LogUtils.setLogInfo(coordAction, logInfo);
462        }

464        /* (non-Javadoc)
465         * @see org.apache.oozie.command.XCommand#verifyPrecondition()
466         */
467        @Override
468        protected void verifyPrecondition() throws CommandException, PreconditionException {
469            if (coordAction.getStatus() != CoordinatorActionBean.Status.WAITING) {
470                throw new PreconditionException(ErrorCode.E1100, "[" + actionId
471                        + "]::CoordActionInputCheck:: Ignoring action. Should be in WAITING state, but state="
472                        + coordAction.getStatus());
473            }
474            if (coordJob.getStatus() != Job.Status.RUNNING && coordJob.getStatus() != Job.Status.PAUSED && coordJob.getStatus() != Job.Status.PAUSEDWITHERROR) {
475                throw new PreconditionException(ErrorCode.E1100, "[" + actionId
476                        + "]::CoordActionInputCheck:: Ignoring action. Coordinator job is not in RUNNING state, but state="
477                        + coordJob.getStatus());
478            }
479        }
