018    package org.apache.oozie.command.wf;
019    
020    import org.apache.oozie.WorkflowJobBean;
021    import org.apache.oozie.command.Command;
022    import org.apache.oozie.store.StoreException;
023    import org.apache.oozie.store.WorkflowStore;
024    import org.apache.oozie.store.Store;
025    import org.apache.oozie.util.ParamChecker;
026    import org.apache.oozie.util.XLog;
027    
028    public class DefinitionCommand extends WorkflowCommand<String> {
029        private String id;
030    
031        public DefinitionCommand(String id) {
032            super("definition", "definition", 0, XLog.STD);
033            this.id = ParamChecker.notEmpty(id, "id");
034        }
035    
036        @Override
037        @SuppressWarnings("unchecked")
038        protected String call(WorkflowStore store) throws StoreException {
039            WorkflowJobBean workflow = store.getWorkflow(id, false);
040            return workflow.getWorkflowInstance().getApp().getDefinition();
041        }
042    
043    }
