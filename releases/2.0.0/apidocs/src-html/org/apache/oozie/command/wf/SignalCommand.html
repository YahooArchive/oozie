064        protected static final String INSTR_SUCCEEDED_JOBS_COUNTER_NAME = "succeeded";

069        protected SignalCommand(String name, int priority, String jobId) {

074        public SignalCommand(String jobId, String actionId) {
075            super("signal", "signal", 0, XLog.STD);
076            this.jobId = ParamChecker.notEmpty(jobId, "jobId");
077            this.actionId = ParamChecker.notEmpty(actionId, "actionId");
078        }
079    
080        @Override
081        protected Void call(WorkflowStore store) throws CommandException, StoreException {
082    
083            WorkflowJobBean workflow = store.getWorkflow(jobId, false);
084            setLogInfo(workflow);
085            WorkflowActionBean action = null;
086            boolean skipAction = false;
087            if (actionId != null) {
088                action = store.getAction(actionId, false);
089                setLogInfo(action);
090            }
091            if ((action == null) || (action.isComplete() && action.isPending())) {
092                try {
093                    if (workflow.getStatus() == WorkflowJob.Status.RUNNING
094                            || workflow.getStatus() == WorkflowJob.Status.PREP) {
095                        WorkflowInstance workflowInstance = workflow.getWorkflowInstance();
096                        workflowInstance.setTransientVar(WorkflowStoreService.WORKFLOW_BEAN, workflow);
097                        boolean completed;
098                        if (action == null) {
099                            if (workflow.getStatus() == WorkflowJob.Status.PREP) {
100                                completed = workflowInstance.start();
101                                workflow.setStatus(WorkflowJob.Status.RUNNING);
102                                workflow.setStartTime(new Date());
103                                workflow.setWorkflowInstance(workflowInstance);
104                                // 1. Add SLA status event for WF-JOB with status
105                                // STARTED
106                                // 2. Add SLA registration events for all WF_ACTIONS
107                                SLADbOperations.writeStausEvent(workflow.getSlaXml(), jobId, store, Status.STARTED,
108                                                                SlaAppType.WORKFLOW_JOB);
109                                writeSLARegistrationForAllActions(workflowInstance.getApp().getDefinition(), workflow
110                                        .getUser(), workflow.getGroup(), workflow.getConf(), store);
111                                queueCallable(new NotificationCommand(workflow));
112                            }
113                            else {
114                                throw new CommandException(ErrorCode.E0801, workflow.getId());
115                            }
116                        }
117                        else {
118                            String skipVar = workflowInstance.getVar(action.getName() + WorkflowInstance.NODE_VAR_SEPARATOR
119                                    + ReRunCommand.TO_SKIP);
120                            if (skipVar != null) {
121                                skipAction = skipVar.equals("true");
122                            }
123                            completed = workflowInstance.signal(action.getExecutionPath(), action.getSignalValue());
124                            workflow.setWorkflowInstance(workflowInstance);
125                            action.resetPending();
126                            if (!skipAction) {
127                                action.setTransition(workflowInstance.getTransition(action.getName()));
128                                store.updateAction(action);
129                            }
130                        }
131    
132                        if (completed) {
133                            for (String actionToKillId : WorkflowStoreService.getActionsToKill(workflowInstance)) {
134                                WorkflowActionBean actionToKill = store.getAction(actionToKillId, false);
135                                actionToKill.setPending();
136                                actionToKill.setStatus(WorkflowActionBean.Status.KILLED);
137                                store.updateAction(actionToKill);
138                                queueCallable(new ActionKillCommand(actionToKill.getId(), actionToKill.getType()));
139                            }
140    
141                            for (String actionToFailId : WorkflowStoreService.getActionsToFail(workflowInstance)) {
142                                WorkflowActionBean actionToFail = store.getAction(actionToFailId, false);
143                                actionToFail.resetPending();
144                                actionToFail.setStatus(WorkflowActionBean.Status.FAILED);
145                                SLADbOperations.writeStausEvent(action.getSlaXml(), action.getId(), store, Status.FAILED,
146                                                                SlaAppType.WORKFLOW_ACTION);
147                                store.updateAction(actionToFail);
148                            }
149    
150                            workflow.setStatus(WorkflowJob.Status.valueOf(workflowInstance.getStatus().toString()));
151                            workflow.setEndTime(new Date());
152                            workflow.setWorkflowInstance(workflowInstance);
153                            Status slaStatus = Status.SUCCEEDED;
154                            switch (workflow.getStatus()) {
155                                case SUCCEEDED:
156                                    slaStatus = Status.SUCCEEDED;
157                                    break;
158                                case KILLED:
159                                    slaStatus = Status.KILLED;
160                                    break;
161                                case FAILED:
162                                    slaStatus = Status.FAILED;
163                                    break;
164                                default: // TODO about SUSPENDED
165    
166                            }
167                            SLADbOperations.writeStausEvent(workflow.getSlaXml(), jobId, store, slaStatus,
168                                                            SlaAppType.WORKFLOW_JOB);
169                            queueCallable(new NotificationCommand(workflow));
170                            if (workflow.getStatus() == WorkflowJob.Status.SUCCEEDED) {
171                                incrJobCounter(INSTR_SUCCEEDED_JOBS_COUNTER_NAME, 1);
172                            }
173                        }
174                        else {
175                            for (WorkflowActionBean newAction : WorkflowStoreService.getStartedActions(workflowInstance)) {
176                                String skipVar = workflowInstance.getVar(newAction.getName()
177                                        + WorkflowInstance.NODE_VAR_SEPARATOR + ReRunCommand.TO_SKIP);
178                                boolean skipNewAction = false;
179                                if (skipVar != null) {
180                                    skipNewAction = skipVar.equals("true");
181                                }
182                                if (skipNewAction) {
183                                    WorkflowActionBean oldAction = store.getAction(newAction.getId(), false);
184                                    oldAction.setPending();
185                                    store.updateAction(oldAction);
186                                    queueCallable(new SignalCommand(jobId, oldAction.getId()));
187                                }
188                                else {
189                                    newAction.setPending();
190                                    String actionSlaXml = getActionSLAXml(newAction.getName(), workflowInstance.getApp()
191                                            .getDefinition(), workflow.getConf());
192                                    // System.out.println("111111 actionXml " +
193                                    // actionSlaXml);
194                                    // newAction.setSlaXml(workflow.getSlaXml());
195                                    newAction.setSlaXml(actionSlaXml);
196                                    store.insertAction(newAction);
197                                    queueCallable(new ActionStartCommand(newAction.getId(), newAction.getType()));
198                                }
199                            }
200                        }
201    
202                        store.updateWorkflow(workflow);
203                        XLog.getLog(getClass()).debug(
204                                "Updated the workflow status to " + workflow.getId() + "  status ="
205                                        + workflow.getStatusStr());
206                        if (workflow.getStatus() != WorkflowJob.Status.RUNNING
207                                && workflow.getStatus() != WorkflowJob.Status.SUSPENDED) {
208                            queueCallable(new CoordActionUpdateCommand(workflow));
209                        }
210                    }
211                    else {
212                        XLog.getLog(getClass()).warn("Workflow not RUNNING, current status [{0}]", workflow.getStatus());
213                    }
214                }
215                catch (WorkflowException ex) {
216                    throw new CommandException(ex);
217                }
218            }
219            else {
220                XLog.getLog(getClass()).warn(
221                        "SignalCommand for action id :" + actionId + " is already processed. status=" + action.getStatus()
222                                + ", Pending=" + action.isPending());
223            }
224            return null;
225        }

227        public static ELEvaluator createELEvaluatorForGroup(Configuration conf, String group) {
228            ELEvaluator eval = Services.get().get(ELService.class).createEvaluator(group);
229            for (Map.Entry<String, String> entry : conf) {
230                eval.setVariable(entry.getKey(), entry.getValue());
231            }
232            return eval;
233        }

235        private String getActionSLAXml(String actionName, String wfXml, String wfConf) throws CommandException {
236            String slaXml = null;
237            // TODO need to fill-out the code
238            // Get the appropriate action:slaXml and resolve that.
239            try {
240                // Configuration conf = new XConfiguration(new
241                // StringReader(wfConf));
242                Element eWfJob = XmlUtils.parseXml(wfXml);
243                // String prefix = XmlUtils.getNamespacePrefix(eWfJob,
244                // SchemaService.SLA_NAME_SPACE_URI);
245                for (Element action : (List<Element>) eWfJob.getChildren("action", eWfJob.getNamespace())) {
246                    if (action.getAttributeValue("name").equals(actionName) == false) {
247                        continue;
248                    }
249                    Element eSla = action.getChild("info", Namespace.getNamespace(SchemaService.SLA_NAME_SPACE_URI));
250                    if (eSla != null) {
251                        // resolveSla(eSla, conf);
252                        slaXml = XmlUtils.prettyPrint(eSla).toString();// Could use
253                        // any
254                        // non-null
255                        // string
256                        break;
257                    }
258                }
259            }
260            catch (Exception e) {
261                throw new CommandException(ErrorCode.E1004, e.getMessage(), e);
262            }
263            return slaXml;
264        }

266        private String resolveSla(Element eSla, Configuration conf) throws CommandException {
267            String slaXml = null;
268            try {
269                ELEvaluator evalSla = SubmitCommand.createELEvaluatorForGroup(conf, "wf-sla-submit");
270                slaXml = SubmitCommand.resolveSla(eSla, evalSla);
271            }
272            catch (Exception e) {
273                throw new CommandException(ErrorCode.E1004, e.getMessage(), e);
274            }
275            return slaXml;
276        }

278        private void writeSLARegistrationForAllActions(String wfXml, String user, String group, String strConf,
279                                                       WorkflowStore store) throws CommandException {
280            try {
281                Element eWfJob = XmlUtils.parseXml(wfXml);
282                // String prefix = XmlUtils.getNamespacePrefix(eWfJob,
283                // SchemaService.SLA_NAME_SPACE_URI);
284                Configuration conf = new XConfiguration(new StringReader(strConf));
285                for (Element action : (List<Element>) eWfJob.getChildren("action", eWfJob.getNamespace())) {
286                    Element eSla = action.getChild("info", Namespace.getNamespace(SchemaService.SLA_NAME_SPACE_URI));
287                    if (eSla != null) {
288                        String slaXml = resolveSla(eSla, conf);
289                        eSla = XmlUtils.parseXml(slaXml);
290                        String actionId = Services.get().get(UUIDService.class).generateChildId(jobId,
291                                                                                                action.getAttributeValue("name") + "");
292                        SLADbOperations.writeSlaRegistrationEvent(eSla, store, actionId, SlaAppType.WORKFLOW_ACTION, user,
293                                                                  group);
294                    }
295                }
296            }
297            catch (Exception e) {
298                throw new CommandException(ErrorCode.E1007, "workflow:Actions " + jobId, e);
299            }

303        @Override
304        protected Void execute(WorkflowStore store) throws CommandException, StoreException {
305            XLog.getLog(getClass()).debug("STARTED SignalCommand for jobid=" + jobId + ", actionId=" + actionId);
306            try {
307                if (lock(jobId)) {
308                    call(store);
309                }
310                else {
311                    queueCallable(new SignalCommand(jobId, actionId), LOCK_FAILURE_REQUEUE_INTERVAL);
312                    XLog.getLog(getClass()).warn("SignalCommand lock was not acquired - failed {0}", jobId);
313                }
314            }
315            catch (InterruptedException e) {
316                queueCallable(new SignalCommand(jobId, actionId), LOCK_FAILURE_REQUEUE_INTERVAL);
317                XLog.getLog(getClass()).warn("SignalCommand lock not acquired - interrupted exception failed {0}", jobId);
318            }
319            XLog.getLog(getClass()).debug("ENDED SignalCommand for jobid=" + jobId + ", actionId=" + actionId);
320            return null;
321        }
