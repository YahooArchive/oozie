072        @Override
073        protected Void call(CoordinatorStore store) throws StoreException, CommandException {
074            //CoordinatorJobBean job = store.getCoordinatorJob(jobId, true);
075            CoordinatorJobBean job = store.getEntityManager().find(CoordinatorJobBean.class, jobId);
076            setLogInfo(job);
077            if (job.getLastActionTime() != null && job.getLastActionTime().compareTo(endTime) >= 0) {
078                log.info("ENDED Coordinator materialization for jobId=" + jobId
079                        + " Action is *already* materialized for time " + startTime + " : " + endTime);
080                return null;
081            }
082    
083            this.user = job.getUser();
084            this.group = job.getGroup();
085    
086            if (job.getStatus().equals(CoordinatorJobBean.Status.PREMATER)) {
087                Configuration jobConf = null;
088                log.debug("start job :" + jobId + " Materialization ");
089                try {
090                    jobConf = new XConfiguration(new StringReader(job.getConf()));
091                }
092                catch (IOException e1) {
093                    log.warn("Configuration parse error. read from DB :" + job.getConf(), e1);
094                    throw new CommandException(ErrorCode.E1005, e1.getMessage(), e1);
095                }
096    
097                Instrumentation.Cron cron = new Instrumentation.Cron();
098                cron.start();
099                try {
100                    materializeJobs(false, job, jobConf, store);
101                    updateJobTable(job, store);
102                }
103                catch (CommandException ex) {
104                    log.warn("Exception occurs:" + ex + " Making the job failed ");
105                    job.setStatus(CoordinatorJobBean.Status.FAILED);
106                    store.updateCoordinatorJob(job);
107                }
108                catch (Exception e) {
109                    log.error("Excepion thrown :", e);
110                    throw new CommandException(ErrorCode.E1001, e.getMessage(), e);
111                }
112                cron.stop();
113            }
114            else {
115                log.info("WARN: action is not in PREMATER state!  It's in state=" + job.getStatus());
116            }
117            return null;
118        }

120        /**
121         * Create action instances starting from "start-time" to end-time" and store them into Action table.
122         *
123         * @param dryrun
124         * @param jobBean
125         * @param conf
126         * @param store
127         * @throws Exception
128         */
129        protected String materializeJobs(boolean dryrun, CoordinatorJobBean jobBean, Configuration conf,
130                                         CoordinatorStore store) throws Exception {
131            String jobXml = jobBean.getJobXml();
132            Element eJob = XmlUtils.parseXml(jobXml);
133            // TODO: always UTC?
134            TimeZone appTz = DateUtils.getTimeZone(jobBean.getTimeZone());
135            // TimeZone appTz = DateUtils.getTimeZone("UTC");
136            int frequency = jobBean.getFrequency();
137            TimeUnit freqTU = TimeUnit.valueOf(eJob.getAttributeValue("freq_timeunit"));
138            TimeUnit endOfFlag = TimeUnit.valueOf(eJob.getAttributeValue("end_of_duration"));
139            Calendar start = Calendar.getInstance(appTz);
140            start.setTime(startTime);
141            DateUtils.moveToEnd(start, endOfFlag);
142            Calendar end = Calendar.getInstance(appTz);
143            end.setTime(endTime);
144            lastActionNumber = jobBean.getLastActionNumber();
145            // DateUtils.moveToEnd(end, endOfFlag);
146            log.info("   *** materialize Actions for tz=" + appTz.getDisplayName() + ",\n start=" + start.getTime()
147                    + ", end=" + end.getTime() + "\n TimeUNIT " + freqTU.getCalendarUnit() + " Frequency :" + frequency
148                    + ":" + freqTU + " lastActionNumber " + lastActionNumber);
149            // Keep the actual start time
150            Calendar origStart = Calendar.getInstance(appTz);
151            origStart.setTime(jobBean.getStartTimestamp());
152            // Move to the End of duration, if needed.
153            DateUtils.moveToEnd(origStart, endOfFlag);
154            // Cloning the start time to be used in loop iteration
155            Calendar effStart = (Calendar) origStart.clone();
156            // Move the time when the previous action finished
157            effStart.add(freqTU.getCalendarUnit(), lastActionNumber * frequency);
158    
159            String action = null;
160            StringBuilder actionStrings = new StringBuilder();
161            while (effStart.compareTo(end) < 0) {
162                CoordinatorActionBean actionBean = new CoordinatorActionBean();
163                lastActionNumber++;
164    
165                actionBean.setTimeOut(jobBean.getTimeout());
166    
167                log.info(origStart.getTime() + " Materializing action for time=" + effStart.getTime()
168                        + ", lastactionnumber=" + lastActionNumber);
169                action = materializeOneInstance(dryrun, (Element) eJob.clone(), effStart.getTime(), lastActionNumber, conf,
170                                                actionBean);
171                if (actionBean.getNominalTimestamp().before(jobBean.getCreatedTimestamp())) {
172                    actionBean.setTimeOut(-1);
173                }
174    
175                if (!dryrun) {
176                    storeToDB(actionBean, action, store); // Storing to table
177                }
178                else {
179                    actionStrings.append("action for new instance");
180                    actionStrings.append(action);
181                }
182                // Restore the original start time
183                effStart = (Calendar) origStart.clone();
184                effStart.add(freqTU.getCalendarUnit(), lastActionNumber * frequency);
185            }
186    
187            endTime = new Date(effStart.getTimeInMillis());
188            if (!dryrun) {
189                return action;
190            }
191            else {
192                return actionStrings.toString();
193            }
194        }

196        /**
197         * materialize one instance for specific nominal time. It includes: 1. Materialize data events (i.e. <data-in> and
198         * <data-out>) 2. Materialize data properties (i.e dataIn(<DS>) and dataOut(<DS>) 3. remove 'start' and 'end' tag 4.
199         * Add 'instance_number' and 'nominal-time' tag
200         *
201         * @param eAction : frequency unexploded-job
202         * @param nominalTime : materialization time
203         * @param instanceCount : instance numbers
204         * @param conf
205         * @return return one materialized job for specific nominal time
206         * @throws Exception
207         */
208        private String materializeOneInstance(boolean dryrun, Element eAction, Date nominalTime, int instanceCount,
209                                              Configuration conf, CoordinatorActionBean actionBean) throws Exception {
210            String actionId = Services.get().get(UUIDService.class).generateChildId(jobId, instanceCount + "");
211            SyncCoordAction appInst = new SyncCoordAction();
212            appInst.setActionId(actionId);
213            appInst.setName(eAction.getAttributeValue("name"));
214            appInst.setNominalTime(nominalTime);
215            int frequency = Integer.parseInt(eAction.getAttributeValue("frequency"));
216            appInst.setFrequency(frequency);
217            appInst.setTimeUnit(TimeUnit.valueOf(eAction.getAttributeValue("freq_timeunit"))); // TODO:
218            appInst.setTimeZone(DateUtils.getTimeZone(eAction.getAttributeValue("timezone")));
219            appInst.setEndOfDuration(TimeUnit.valueOf(eAction.getAttributeValue("end_of_duration")));
220    
221            StringBuffer dependencyList = new StringBuffer();
222    
223            Element inputList = eAction.getChild("input-events", eAction.getNamespace());
224            List<Element> dataInList = null;
225            if (inputList != null) {
226                dataInList = (List<Element>) inputList.getChildren("data-in", eAction.getNamespace());
227                materializeDataEvents(dataInList, appInst, conf, dependencyList);
228            }
229    
230            Element outputList = eAction.getChild("output-events", eAction.getNamespace());
231            List<Element> dataOutList = null;
232            if (outputList != null) {
233                dataOutList = (List<Element>) outputList.getChildren("data-out", eAction.getNamespace());
234                StringBuffer tmp = new StringBuffer();
235                materializeDataEvents(dataOutList, appInst, conf, tmp);// no dependency checks
236            }
237    
238            eAction.removeAttribute("start");
239            eAction.removeAttribute("end");
240            eAction.setAttribute("instance-number", Integer.toString(instanceCount));
241            eAction.setAttribute("action-nominal-time", DateUtils.formatDateUTC(nominalTime));
242    
243            boolean isSla = materializeSLA(eAction.getChild("action", eAction.getNamespace()).getChild("info",
244                                                                                                       eAction.getNamespace("sla")), nominalTime, conf);
245    
246            // Setting up action bean
247            actionBean.setCreatedConf(XmlUtils.prettyPrint(conf).toString());
248            actionBean.setRunConf(XmlUtils.prettyPrint(conf).toString()); // TODO:
249            actionBean.setCreatedTime(new Date());
250            actionBean.setJobId(jobId);
251            // actionBean.setId(jobId + "_" + instanceCount);
252            actionBean.setId(actionId);
253            actionBean.setLastModifiedTime(new Date());
254            actionBean.setStatus(CoordinatorAction.Status.WAITING);
255            actionBean.setActionNumber(instanceCount);
256            actionBean.setMissingDependencies(dependencyList.toString());
257            actionBean.setNominalTime(nominalTime);
258            if (isSla == true) {
259                actionBean.setSlaXml(XmlUtils.prettyPrint(
260                        eAction.getChild("action", eAction.getNamespace()).getChild("info", eAction.getNamespace("sla")))
261                        .toString());
262            }
263    
264            // actionBean.setTrackerUri(trackerUri);//TOOD:
265            // actionBean.setConsoleUrl(consoleUrl); //TODO:
266            // actionBean.setType(type);//TODO:
267            // actionBean.setErrorInfo(errorCode, errorMessage); //TODO:
268            // actionBean.setExternalStatus(externalStatus);//TODO
269            if (!dryrun) {
270                return XmlUtils.prettyPrint(eAction).toString();
271            }
272            else {
273                String action = XmlUtils.prettyPrint(eAction).toString();
274                CoordActionInputCheckCommand coordActionInput = new CoordActionInputCheckCommand(actionBean.getId());
275                StringBuilder actionXml = new StringBuilder(action);
276                StringBuilder existList = new StringBuilder();
277                StringBuilder nonExistList = new StringBuilder();
278                StringBuilder nonResolvedList = new StringBuilder();
279                getResolvedList(actionBean.getMissingDependencies(), nonExistList, nonResolvedList);
280                Date actualTime = new Date();
281                Configuration actionConf = new XConfiguration(new StringReader(actionBean.getRunConf()));
282                coordActionInput.checkInput(actionXml, existList, nonExistList, actionConf, actualTime);
283                return actionXml.toString();
284            }
285    
286            // return XmlUtils.prettyPrint(eAction).toString();
287        }

289        /**
290         * Materialize all <input-events>/<data-in> or <output-events>/<data-out> tags Create uris for resolved instances.
291         * Create unresolved instance for latest().
292         *
293         * @param events
294         * @param appInst
295         * @param conf
296         * @throws Exception
297         */
298        private void materializeDataEvents(List<Element> events, SyncCoordAction appInst, Configuration conf,
299                                           StringBuffer dependencyList) throws Exception {
300    
301            if (events == null) {
302                return;
303            }
304            StringBuffer unresolvedList = new StringBuffer();
305            for (Element event : events) {
306                StringBuilder instances = new StringBuilder();
307                ELEvaluator eval = CoordELEvaluator.createInstancesELEvaluator(event, appInst, conf);
308                // Handle list of instance tag
309                resolveInstances(event, instances, appInst, conf, eval);
310                // Handle start-instance and end-instance
311                resolveInstanceRange(event, instances, appInst, conf, eval);
312                // Separate out the unresolved instances
313                separateResolvedAndUnresolved(event, instances, dependencyList);
314                String tmpUnresolved = event.getChildTextTrim("unresolved-instances", event.getNamespace());
315                if (tmpUnresolved != null) {
316                    if (unresolvedList.length() > 0) {
317                        unresolvedList.append(CoordELFunctions.INSTANCE_SEPARATOR);
318                    }
319                    unresolvedList.append(tmpUnresolved);
320                }
321            }
322            if (unresolvedList.length() > 0) {
323                dependencyList.append(RESOLVED_UNRESOLVED_SEPARATOR);
324                dependencyList.append(unresolvedList);
325            }
326            return;
327        }

330         * Resolve list of <instance> </instance> tags.
331         *
332         * @param event
333         * @param instances
334         * @param actionInst
335         * @param conf
336         * @throws Exception
337         */
338        private void resolveInstances(Element event, StringBuilder instances, SyncCoordAction actionInst,
339                                      Configuration conf, ELEvaluator eval) throws Exception {
340            for (Element eInstance : (List<Element>) event.getChildren("instance", event.getNamespace())) {
341                if (instances.length() > 0) {
342                    instances.append(CoordELFunctions.INSTANCE_SEPARATOR);
343                }
344                instances.append(materializeInstance(event, eInstance.getTextTrim(), actionInst, conf, eval));
345            }
346            event.removeChildren("instance", event.getNamespace());
347        }

349        /**
350         * Resolve <start-instance> <end-insatnce> tag. Don't resolve any latest()
351         *
352         * @param event
353         * @param instances
354         * @param appInst
355         * @param conf
356         * @throws Exception
357         */
358        private void resolveInstanceRange(Element event, StringBuilder instances, SyncCoordAction appInst,
359                                          Configuration conf, ELEvaluator eval) throws Exception {
360            Element eStartInst = event.getChild("start-instance", event.getNamespace());
361            Element eEndInst = event.getChild("end-instance", event.getNamespace());
362            if (eStartInst != null && eEndInst != null) {
363                String strStart = eStartInst.getTextTrim();
364                String strEnd = eEndInst.getTextTrim();
365                checkIfBothSameType(strStart, strEnd);
366                int startIndex = getInstanceNumber(strStart, event, appInst, conf);
367                int endIndex = getInstanceNumber(strEnd, event, appInst, conf);
368                if (startIndex > endIndex) {
369                    throw new CommandException(ErrorCode.E1010,
370                                               " start-instance should be equal or earlier than the end-instance \n"
371                                                       + XmlUtils.prettyPrint(event));
372                }
373                if (strStart.indexOf("latest") < 0 && strEnd.indexOf("latest") < 0) {
374                    // Everything could be resolved NOW. no latest() ELs
375                    for (int i = endIndex; i >= startIndex; i--) {
376                        String matInstance = materializeInstance(event, "${coord:current(" + i + ")}", appInst, conf, eval);
377                        if (matInstance == null || matInstance.length() == 0) {
378                            // Earlier than dataset's initial instance
379                            break;
380                        }
381                        if (instances.length() > 0) {
382                            instances.append(CoordELFunctions.INSTANCE_SEPARATOR);
383                        }
384                        instances.append(matInstance);
385                    }
386                }
387                else { // latest(n) EL is present
388                    for (; startIndex <= endIndex; startIndex++) {
389                        if (instances.length() > 0) {
390                            instances.append(CoordELFunctions.INSTANCE_SEPARATOR);
391                        }
392                        instances.append("${coord:latest(" + startIndex + ")}");
393                    }
394                }
395                // Remove start-instance and end-instances
396                event.removeChild("start-instance", event.getNamespace());
397                event.removeChild("end-instance", event.getNamespace());
398            }
399        }

401        private void checkIfBothSameType(String startInst, String endInst) throws CommandException {
402            if ((startInst.indexOf("current") >= 0 && endInst.indexOf("latest") >= 0)
403                    || (startInst.indexOf("latest") >= 0 && endInst.indexOf("current") >= 0)) {
404                throw new CommandException(ErrorCode.E1010,
405                                           " start-instance and end-instance both should be either latest or current\n" + " start "
406                                                   + startInst + " and end " + endInst);
407            }
408        }

410        /**
411         * Create two new tags with <uris> and <unresolved-instances>.
412         *
413         * @param event
414         * @param instances
415         * @param dependencyList
416         * @throws Exception
417         */
418        private void separateResolvedAndUnresolved(Element event, StringBuilder instances, StringBuffer dependencyList)
419                throws Exception {
420            StringBuilder unresolvedInstances = new StringBuilder();
421            StringBuilder urisWithDoneFlag = new StringBuilder();
422            String uris = createURIs(event, instances.toString(), unresolvedInstances, urisWithDoneFlag);
423            if (uris.length() > 0) {
424                Element uriInstance = new Element("uris", event.getNamespace());
425                uriInstance.addContent(uris);
426                event.getContent().add(1, uriInstance);
427                if (dependencyList.length() > 0) {
428                    dependencyList.append(CoordELFunctions.INSTANCE_SEPARATOR);
429                }
430                dependencyList.append(urisWithDoneFlag);
431            }
432            if (unresolvedInstances.length() > 0) {
433                Element elemInstance = new Element("unresolved-instances", event.getNamespace());
434                elemInstance.addContent(unresolvedInstances.toString());
435                event.getContent().add(1, elemInstance);
436            }
437        }

439        /**
440         * The function create a list of URIs separated by "," using the instances time stamp and URI-template
441         *
442         * @param event : <data-in> event
443         * @param instances : List of time stanmp seprated by ","
444         * @param unresolvedInstances : list of instance with latest function
445         * @param urisWithDoneFlag : list of URIs with the done flag appended
446         * @return : list of URIs separated by ",".
447         * @throws Exception
448         */
449        private String createURIs(Element event, String instances, StringBuilder unresolvedInstances,
450                                  StringBuilder urisWithDoneFlag) throws Exception {
451            if (instances == null || instances.length() == 0) {
452                return "";
453            }
454            String[] instanceList = instances.split(CoordELFunctions.INSTANCE_SEPARATOR);
455            StringBuilder uris = new StringBuilder();
456    
457            Element doneFlagElement = event.getChild("dataset", event.getNamespace()).getChild("done-flag",
458                                                                                               event.getNamespace());
459            String doneFlag = CoordUtils.getDoneFlag(doneFlagElement);
460    
461            for (int i = 0; i < instanceList.length; i++) {
462                if (instanceList[i].indexOf("latest") >= 0) {
463                    if (unresolvedInstances.length() > 0) {
464                        unresolvedInstances.append(CoordELFunctions.INSTANCE_SEPARATOR);
465                    }
466                    unresolvedInstances.append(instanceList[i]);
467                    continue;
468                }
469                ELEvaluator eval = CoordELEvaluator.createURIELEvaluator(instanceList[i]);
470                if (uris.length() > 0) {
471                    uris.append(CoordELFunctions.INSTANCE_SEPARATOR);
472                    urisWithDoneFlag.append(CoordELFunctions.INSTANCE_SEPARATOR);
473                }
474    
475                String uriPath = CoordELFunctions.evalAndWrap(eval, event.getChild("dataset", event.getNamespace())
476                        .getChild("uri-template", event.getNamespace()).getTextTrim());
477                uris.append(uriPath);
478                if (doneFlag.length() > 0) {
479                    uriPath += "/" + doneFlag;
480                }
481                urisWithDoneFlag.append(uriPath);
482            }
483            return uris.toString();
484        }

506        /**
507         * parse a function like coord:latest(n) and return the 'n'.
508         *
509         * @param function
510         * @return parameter of the function
511         * @throws Exception
512         */
513        private int getInstanceNumber(String function, Element event, SyncCoordAction appInst, Configuration conf)
514                throws Exception {
515            ELEvaluator eval = CoordELEvaluator
516                    .createInstancesELEvaluator("coord-action-create-inst", event, appInst, conf);
517            String newFunc = CoordELFunctions.evalAndWrap(eval, function);
518            int firstPos = newFunc.indexOf("(");
519            int lastPos = newFunc.lastIndexOf(")");
520            if (firstPos >= 0 && lastPos > firstPos) {
521                String tmp = newFunc.substring(firstPos + 1, lastPos).trim();
522                if (tmp.length() > 0) {
523                    return Integer.parseInt(tmp);
524                }
525            }
526            // Error handling
527            throw new RuntimeException("Unformatted function :" + newFunc);
528        }

559        private void updateJobTable(CoordinatorJobBean job, CoordinatorStore store) throws StoreException {
560            // TODO: why do we need this? Isn't lastMatTime enough???
561            job.setLastActionTime(endTime);
562            job.setLastActionNumber(lastActionNumber);
563            // if the job endtime == action endtime, then set status of job to
564            // succeeded
565            // we dont need to materialize this job anymore
566            Date jobEndTime = job.getEndTime();
567            if (jobEndTime.compareTo(endTime) <= 0) {
568                job.setStatus(CoordinatorJob.Status.SUCCEEDED);
569                log.info("[" + job.getId() + "]: Update status from PREMATER to SUCCEEDED");
570            }
571            else {
572                job.setStatus(CoordinatorJob.Status.RUNNING);
573                log.info("[" + job.getId() + "]: Update status from PREMATER to RUNNING");
574            }
575            job.setNextMaterializedTime(endTime);
576            store.updateCoordinatorJob(job);
577        }

579        private boolean materializeSLA(Element eSla, Date nominalTime, Configuration conf) throws CoordinatorJobException {
580            if (eSla == null) {
581                // System.out.println("NO SLA presents " +
582                // eAppXml.getNamespace("sla"));
583                return false;
584            }
585            try {
586                ELEvaluator evalSla = CoordELEvaluator.createSLAEvaluator(nominalTime, conf);
587                // System.out.println("SLA presents");
588                List<Element> elemList = eSla.getChildren();
589                for (Element elem : elemList) {
590                    String updated;
591                    try {
592                        updated = CoordELFunctions.evalAndWrap(evalSla, elem.getText().trim());
593                    }
594                    catch (Exception e) {
595                        throw new CoordinatorJobException(ErrorCode.E1004, e.getMessage(), e);
596                    }
597                    elem.removeContent();
598                    elem.addContent(updated);
599                }
600            }
601            catch (Exception e) {
602                throw new CoordinatorJobException(ErrorCode.E1004, e.getMessage(), e);
603            }
604            return true;
605        }

607        @Override
608        protected Void execute(CoordinatorStore store) throws StoreException, CommandException {
609            log.info("STARTED CoordActionMaterializeCommand for jobId=" + jobId + ", startTime=" + startTime + ", endTime="
610                    + endTime);
611            try {
612                if (lock(jobId)) {
613                    call(store);
614                }
615                else {
616                    queueCallable(new CoordActionMaterializeCommand(jobId, startTime, endTime), LOCK_FAILURE_REQUEUE_INTERVAL);
617                    log.warn("CoordActionMaterializeCommand lock was not acquired - failed jobId=" + jobId
618                            + ". Requeing the same.");
619                }
620            }
621            catch (InterruptedException e) {
622                queueCallable(new CoordActionMaterializeCommand(jobId, startTime, endTime), LOCK_FAILURE_REQUEUE_INTERVAL);
623                log.warn("CoordActionMaterializeCommand lock acquiring failed with exception " + e.getMessage()
624                        + " for jobId=" + jobId + " Requeing the same.");
625            }
626            finally {
627                log.info(" ENDED CoordActionMaterializeCommand for jobId=" + jobId + ", startTime=" + startTime
628                        + ", endTime=" + endTime);
629            }
630            return null;
631        }

634            if (missDepList != null) {
635                int index = missDepList.indexOf(RESOLVED_UNRESOLVED_SEPARATOR);
636                if (index < 0) {
637                    resolved.append(missDepList);
638                }
639                else {
640                    resolved.append(missDepList.substring(0, index));
641                    unresolved.append(missDepList.substring(index + 1));
642                }
643            }
644            return resolved.toString();
645        }
