018    package org.apache.oozie.command.wf;
019    
020    import org.apache.oozie.client.WorkflowJob;
021    import org.apache.oozie.WorkflowJobBean;
022    import org.apache.oozie.command.Command;
023    import org.apache.oozie.command.CommandException;
024    import org.apache.oozie.store.StoreException;
025    import org.apache.oozie.store.WorkflowStore;
026    import org.apache.oozie.store.Store;
027    import org.apache.oozie.workflow.WorkflowException;
028    import org.apache.oozie.workflow.WorkflowInstance;
029    import org.apache.oozie.workflow.lite.LiteWorkflowInstance;
030    import org.apache.oozie.util.ParamChecker;
031    import org.apache.oozie.util.XLog;
032    
033    import java.util.Date;
034    
035    public class SuspendCommand extends WorkflowCommand<Void> {
036    
037        private String id;
038    
039        public SuspendCommand(String id) {
040            super("suspend", "suspend", 0, XLog.STD);
041            this.id = ParamChecker.notEmpty(id, "id");
042        }
043    
044        @Override
045        protected Void call(WorkflowStore store) throws StoreException, CommandException {
046            try {
047                WorkflowJobBean workflow = store.getWorkflow(id, false);
048                setLogInfo(workflow);
049                if (workflow.getStatus() == WorkflowJob.Status.RUNNING) {
050                    incrJobCounter(1);
051                    suspendJob(workflow, id);
052                    store.updateWorkflow(workflow);
053                    queueCallable(new NotificationCommand(workflow));
054                }
055                return null;
056            }
057            catch (WorkflowException ex) {
058                throw new CommandException(ex);
059            }
060        }
061    
062        public static void suspendJob(WorkflowJobBean workflow, String id) throws WorkflowException {
063            if (workflow.getStatus() == WorkflowJob.Status.RUNNING) {
064                workflow.getWorkflowInstance().suspend();
065                WorkflowInstance wfInstance = workflow.getWorkflowInstance();
066                ((LiteWorkflowInstance) wfInstance).setStatus(WorkflowInstance.Status.SUSPENDED);
067                workflow.setStatus(WorkflowJob.Status.SUSPENDED);
068                workflow.setWorkflowInstance(wfInstance);
069            }
070        }
071    
072        @Override
073        protected Void execute(WorkflowStore store) throws CommandException, StoreException {
074            XLog.getLog(getClass()).debug("STARTED SuspendCommand for action " + id);
075            try {
076                if (lock(id)) {
077                    call(store);
078                }
079                else {
080                    queueCallable(new SuspendCommand(id), LOCK_FAILURE_REQUEUE_INTERVAL);
081                    XLog.getLog(getClass()).warn("Suspend lock was not acquired - failed {0}", id);
082                }
083            }
084            catch (InterruptedException e) {
085                queueCallable(new SuspendCommand(id), LOCK_FAILURE_REQUEUE_INTERVAL);
086                XLog.getLog(getClass()).warn("SuspendCommand lock was not acquired - interrupted exception failed {0}", id);
087            }
088            finally {
089                XLog.getLog(getClass()).debug("ENDED SuspendCommand for action " + id);
090            }
091            return null;
092        }
093    }
