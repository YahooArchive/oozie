137         * (non-Javadoc)
138         * 
139         * @see org.apache.oozie.command.Command#call(org.apache.oozie.store.Store)
140         */
141        @Override
142        protected String call(CoordinatorStore store) throws StoreException, CommandException {
143            String jobId = null;
144            log.info("STARTED Coordinator Submit");
145            incrJobCounter(1);
146            CoordinatorJobBean coordJob = new CoordinatorJobBean();
147            try {
148                XLog.Info.get().setParameter(DagXLogInfoService.TOKEN, conf.get(OozieClient.LOG_TOKEN));
149                mergeDefaultConfig();
150    
151                String appXml = readAndValidateXml();
152                coordJob.setOrigJobXml(appXml);
153                log.debug("jobXml after initial validation " + XmlUtils.prettyPrint(appXml).toString());
154                appXml = XmlUtils.removeComments(appXml);
155                initEvaluators();
156                Element eJob = basicResolveAndIncludeDS(appXml, conf, coordJob);
157                log.debug("jobXml after all validation " + XmlUtils.prettyPrint(eJob).toString());
158    
159                jobId = storeToDB(eJob, store, coordJob);
160    
161                // log JOB info for coordinator jobs
162                setLogInfo(coordJob);
163    
164                if (!dryrun) {
165                    // submit a command to materialize jobs for the next 1 hour (3600 secs)
166                    // so we don't wait 10 mins for the Service to run.
167                    queueCallable(new CoordJobMatLookupCommand(jobId, 3600), 100);
168                }
169                else {
170                    Date startTime = coordJob.getStartTime();
171                    long startTimeMilli = startTime.getTime();
172                    long endTimeMilli = startTimeMilli + (3600 * 1000);
173                    Date jobEndTime = coordJob.getEndTime();
174                    Date endTime = new Date(endTimeMilli);
175                    if (endTime.compareTo(jobEndTime) > 0) {
176                        endTime = jobEndTime;
177                    }
178                    jobId = coordJob.getId();
179                    log.info("[" + jobId + "]: Update status to PREMATER");
180                    coordJob.setStatus(CoordinatorJob.Status.PREMATER);
181                    CoordActionMaterializeCommand coordActionMatCom = new CoordActionMaterializeCommand(jobId, startTime,
182                                                                                                        endTime);
183                    Configuration jobConf = null;
184                    try {
185                        jobConf = new XConfiguration(new StringReader(coordJob.getConf()));
186                    }
187                    catch (IOException e1) {
188                        log.warn("Configuration parse error. read from DB :" + coordJob.getConf(), e1);
189                    }
190                    String action = coordActionMatCom.materializeJobs(true, coordJob, jobConf, null);
191                    String output = coordJob.getJobXml() + System.getProperty("line.separator")
192                            + "***actions for instance***" + action;
193                    return output;
194                }
195            }
196            catch (CoordinatorJobException ex) {
197                log.warn("ERROR:  ", ex);
198                throw new CommandException(ex);
199            }
200            catch (IllegalArgumentException iex) {
201                log.warn("ERROR:  ", iex);
202                throw new CommandException(ErrorCode.E1003, iex);
203            }
204            catch (Exception ex) {// TODO
205                log.warn("ERROR:  ", ex);
206                throw new CommandException(ErrorCode.E0803, ex);
207            }
208            log.info("ENDED Coordinator Submit jobId=" + jobId);
209            return jobId;
210        }

226        /**
227         * Validate against Coordinator XSD file
228         *
229         * @param xmlContent : Input coordinator xml
230         * @throws CoordinatorJobException
231         */
232        private void validateXml(String xmlContent) throws CoordinatorJobException {
233            javax.xml.validation.Schema schema = Services.get().get(SchemaService.class).getSchema(SchemaName.COORDINATOR);
234            Validator validator = schema.newValidator();
235            // log.warn("XML " + xmlContent);
236            try {
237                validator.validate(new StreamSource(new StringReader(xmlContent)));
238            }
239            catch (SAXException ex) {
240                log.warn("SAXException :", ex);
241                throw new CoordinatorJobException(ErrorCode.E0701, ex.getMessage(), ex);
242            }
243            catch (IOException ex) {
244                // ex.printStackTrace();
245                log.warn("IOException :", ex);
246                throw new CoordinatorJobException(ErrorCode.E0702, ex.getMessage(), ex);
247            }
248        }

250        /**
251         * Merge default configuration with user-defined configuration.
252         *
253         * @throws CommandException
254         */
255        protected void mergeDefaultConfig() throws CommandException {
256            Path configDefault = new Path(conf.get(OozieClient.COORDINATOR_APP_PATH), CONFIG_DEFAULT);
257            // Configuration fsConfig = new Configuration();
258            // log.warn("CONFIG :" + configDefault.toUri());
259            Configuration fsConfig = CoordUtils.getHadoopConf(conf);
260            FileSystem fs;
261            // TODO: which conf?
262            try {
263                String user = ParamChecker.notEmpty(conf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
264                String group = ParamChecker.notEmpty(conf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
265                fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, configDefault.toUri(),
266                                                                                      new Configuration());
267                if (fs.exists(configDefault)) {
268                    Configuration defaultConf = new XConfiguration(fs.open(configDefault));
269                    PropertiesUtils.checkDisallowedProperties(defaultConf, DISALLOWED_DEFAULT_PROPERTIES);
270                    XConfiguration.injectDefaults(defaultConf, conf);
271                }
272                else {
273                    log.info("configDefault Doesn't exist " + configDefault);
274                }
275                PropertiesUtils.checkDisallowedProperties(conf, DISALLOWED_USER_PROPERTIES);
276            }
277            catch (IOException e) {
278                throw new CommandException(ErrorCode.E0702, e.getMessage() + " : Problem reading default config "
279                        + configDefault, e);
280            }
281            log.debug("Merged CONF :" + XmlUtils.prettyPrint(conf).toString());
282        }

301        /**
302         * Insert data set into data-in and data-out tags.
303         *
304         * @param eAppXml : coordinator application XML
305         * @param eDatasets : DataSet XML
306         * @return updated application
307         */
308        private void insertDataSet(Element eAppXml, Element eDatasets) {
309            // Adding DS definition in the coordinator XML
310            Element inputList = eAppXml.getChild("input-events", eAppXml.getNamespace());
311            if (inputList != null) {
312                for (Element dataIn : (List<Element>) inputList.getChildren("data-in", eAppXml.getNamespace())) {
313                    Element eDataset = findDataSet(eDatasets, dataIn.getAttributeValue("dataset"));
314                    dataIn.getContent().add(0, eDataset);
315                }
316            }
317            Element outputList = eAppXml.getChild("output-events", eAppXml.getNamespace());
318            if (outputList != null) {
319                for (Element dataOut : (List<Element>) outputList.getChildren("data-out", eAppXml.getNamespace())) {
320                    Element eDataset = findDataSet(eDatasets, dataOut.getAttributeValue("dataset"));
321                    dataOut.getContent().add(0, eDataset);
322                }
323            }
324        }

326        /**
327         * Find a specific dataset from a list of Datasets.
328         *
329         * @param eDatasets : List of data sets
330         * @param name : queried data set name
331         * @return one Dataset element. otherwise throw Exception
332         */
333        private static Element findDataSet(Element eDatasets, String name) {
334            for (Element eDataset : (List<Element>) eDatasets.getChildren("dataset", eDatasets.getNamespace())) {
335                if (eDataset.getAttributeValue("name").equals(name)) {
336                    eDataset = (Element) eDataset.clone();
337                    eDataset.detach();
338                    return eDataset;
339                }
340            }
341            throw new RuntimeException("undefined dataset: " + name);
342        }

354        /**
355         * Resolve basic entities using job Configuration.
356         *
357         * @param conf :Job configuration
358         * @param appXml : Original job XML
359         * @param coordJob : Coordinator job bean to be populated.
360         * @return Resolved job XML element.
361         * @throws Exception
362         */
363        protected Element resolveInitial(Configuration conf, String appXml, CoordinatorJobBean coordJob)
364                throws CoordinatorJobException, Exception {
365            Element eAppXml = XmlUtils.parseXml(appXml);
366            // job's main attributes
367            // frequency
368            String val = resolveAttribute("frequency", eAppXml, evalFreq);
369            int ival = ParamChecker.checkInteger(val, "frequency");
370            ParamChecker.checkGTZero(ival, "frequency");
371            coordJob.setFrequency(ival);
372            TimeUnit tmp = (evalFreq.getVariable("timeunit") == null) ? TimeUnit.MINUTE : ((TimeUnit) evalFreq
373                    .getVariable("timeunit"));
374            addAnAttribute("freq_timeunit", eAppXml, tmp.toString()); // TODO: Store
375            // TimeUnit
376            coordJob.setTimeUnit(CoordinatorJob.Timeunit.valueOf(tmp.toString()));
377            // End Of Duration
378            tmp = evalFreq.getVariable("endOfDuration") == null ? TimeUnit.NONE : ((TimeUnit) evalFreq
379                    .getVariable("endOfDuration"));
380            addAnAttribute("end_of_duration", eAppXml, tmp.toString());
381            // coordJob.setEndOfDuration(tmp) // TODO: Add new attribute in Job bean
382    
383            // start time
384            val = resolveAttribute("start", eAppXml, evalNofuncs);
385            ParamChecker.checkUTC(val, "start");
386            coordJob.setStartTime(DateUtils.parseDateUTC(val));
387            // end time
388            val = resolveAttribute("end", eAppXml, evalNofuncs);
389            ParamChecker.checkUTC(val, "end");
390            coordJob.setEndTime(DateUtils.parseDateUTC(val));
391            // Time zone
392            val = resolveAttribute("timezone", eAppXml, evalNofuncs);
393            ParamChecker.checkTimeZone(val, "timezone");
394            coordJob.setTimeZone(val);
395    
396            // controls
397            val = resolveTagContents("timeout", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
398            if (val == "") {
399                val = "-1";
400            }
401            ival = ParamChecker.checkInteger(val, "timeout");
402            // ParamChecker.checkGEZero(ival, "timeout");
403            coordJob.setTimeout(ival);
404            val = resolveTagContents("concurrency", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
405            if (val == "") {
406                val = "-1";
407            }
408            ival = ParamChecker.checkInteger(val, "concurrency");
409            // ParamChecker.checkGEZero(ival, "concurrency");
410            coordJob.setConcurrency(ival);
411            val = resolveTagContents("execution", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
412            if (val == "") {
413                val = Execution.FIFO.toString();
414            }
415            coordJob.setExecution(Execution.valueOf(val));
416            String[] acceptedVals = {Execution.LIFO.toString(), Execution.FIFO.toString(), Execution.LAST_ONLY.toString()};
417            ParamChecker.isMember(val, acceptedVals, "execution");
418    
419            // datasets
420            resolveTagContents("include", eAppXml.getChild("datasets", eAppXml.getNamespace()), evalNofuncs);
421            // for each data set
422            resolveDataSets(eAppXml);
423            HashMap<String, String> dataNameList = new HashMap<String, String>();
424            resolveIOEvents(eAppXml, dataNameList);
425    
426            resolveTagContents("app-path", eAppXml.getChild("action", eAppXml.getNamespace()).getChild("workflow",
427                                                                                                       eAppXml.getNamespace()), evalNofuncs);
428            // TODO: If action or workflow tag is missing, NullPointerException will
429            // occur
430            Element configElem = eAppXml.getChild("action", eAppXml.getNamespace()).getChild("workflow",
431                                                                                             eAppXml.getNamespace()).getChild("configuration", eAppXml.getNamespace());
432            evalData = CoordELEvaluator.createELEvaluatorForDataEcho(conf, "coord-job-submit-data", dataNameList);
433            if (configElem != null) {
434                for (Element propElem : (List<Element>) configElem.getChildren("property", configElem.getNamespace())) {
435                    resolveTagContents("name", propElem, evalData);
436                    // log.warn("Value :");
437                    // Want to check the data-integrity but don't want to modify the
438                    // XML
439                    // for properties only
440                    Element tmpProp = (Element) propElem.clone();
441                    resolveTagContents("value", tmpProp, evalData);
442                    // val = resolveTagContents("value", propElem, evalData);
443                    // log.warn("Value OK :" + val);
444                }
445            }
446            resolveSLA(eAppXml, coordJob);
447            return eAppXml;
448        }

450        private void resolveSLA(Element eAppXml, CoordinatorJobBean coordJob) throws CommandException {
451            // String prefix = XmlUtils.getNamespacePrefix(eAppXml,
452            // SchemaService.SLA_NAME_SPACE_URI);
453            Element eSla = eAppXml.getChild("action", eAppXml.getNamespace()).getChild("info",
454                                                                                       Namespace.getNamespace(SchemaService.SLA_NAME_SPACE_URI));
455    
456            if (eSla != null) {
457                String slaXml = XmlUtils.prettyPrint(eSla).toString();
458                try {
459                    // EL evaluation
460                    slaXml = evalSla.evaluate(slaXml, String.class);

463                }
464                catch (Exception e) {
465                    throw new CommandException(ErrorCode.E1004, "Validation ERROR :" + e.getMessage(), e);
466                }
467            }
468        }

470        /**
471         * Resolve input-events/data-in and output-events/data-out tags.
472         *
473         * @param eJob : Job element
474         * @throws CoordinatorJobException
475         */
476        private void resolveIOEvents(Element eJobOrg, HashMap<String, String> dataNameList) throws CoordinatorJobException {
477            // Resolving input-events/data-in
478            // Clone the job and don't update anything in the original
479            Element eJob = (Element) eJobOrg.clone();
480            Element inputList = eJob.getChild("input-events", eJob.getNamespace());
481            if (inputList != null) {
482                TreeSet<String> eventNameSet = new TreeSet<String>();
483                for (Element dataIn : (List<Element>) inputList.getChildren("data-in", eJob.getNamespace())) {
484                    String dataInName = dataIn.getAttributeValue("name");
485                    dataNameList.put(dataInName, "data-in");
486                    // check whether there is any duplicate data-in name
487                    if (eventNameSet.contains(dataInName)) {
488                        throw new RuntimeException("Duplicate dataIn name " + dataInName);
489                    }
490                    else {
491                        eventNameSet.add(dataInName);
492                    }
493                    resolveTagContents("instance", dataIn, evalInst);
494                    resolveTagContents("start-instance", dataIn, evalInst);
495                    resolveTagContents("end-instance", dataIn, evalInst);
496                }
497            }
498            // Resolving output-events/data-out
499            Element outputList = eJob.getChild("output-events", eJob.getNamespace());
500            if (outputList != null) {
501                TreeSet<String> eventNameSet = new TreeSet<String>();
502                for (Element dataOut : (List<Element>) outputList.getChildren("data-out", eJob.getNamespace())) {
503                    String dataOutName = dataOut.getAttributeValue("name");
504                    dataNameList.put(dataOutName, "data-out");
505                    // check whether there is any duplicate data-out name
506                    if (eventNameSet.contains(dataOutName)) {
507                        throw new RuntimeException("Duplicate dataIn name " + dataOutName);
508                    }
509                    else {
510                        eventNameSet.add(dataOutName);
511                    }
512                    resolveTagContents("instance", dataOut, evalInst);
513                }
514            }
515    
516        }

546        /**
547         * Resolve Data set using job configuration.
548         *
549         * @param dsElems : Data set XML element.
550         * @throws CoordinatorJobException
551         * @throws Exception
552         */
553        private void resolveDataSets(List<Element> dsElems) throws CoordinatorJobException /*
554                                                                                            * throws
555                                                                                            * Exception
556                                                                                            */ {
557            for (Element dsElem : dsElems) {
558                // Setting up default TimeUnit and EndOFDuraion
559                evalFreq.setVariable("timeunit", TimeUnit.MINUTE);
560                evalFreq.setVariable("endOfDuration", TimeUnit.NONE);
561    
562                String val = resolveAttribute("frequency", dsElem, evalFreq);
563                int ival = ParamChecker.checkInteger(val, "frequency");
564                ParamChecker.checkGTZero(ival, "frequency");
565                addAnAttribute("freq_timeunit", dsElem, evalFreq.getVariable("timeunit") == null ? TimeUnit.MINUTE
566                        .toString() : ((TimeUnit) evalFreq.getVariable("timeunit")).toString());
567                addAnAttribute("end_of_duration", dsElem, evalFreq.getVariable("endOfDuration") == null ? TimeUnit.NONE
568                        .toString() : ((TimeUnit) evalFreq.getVariable("endOfDuration")).toString());
569                val = resolveAttribute("initial-instance", dsElem, evalNofuncs);
570                ParamChecker.checkUTC(val, "initial-instance");
571                val = resolveAttribute("timezone", dsElem, evalNofuncs);
572                ParamChecker.checkTimeZone(val, "timezone");
573                resolveTagContents("uri-template", dsElem, evalNofuncs);
574                resolveTagContents("done-flag", dsElem, evalNofuncs);
575            }
576        }

640         * Include referred Datasets into XML.
641         *
642         * @param resolvedXml : Job XML element.
643         * @param conf : Job configuration
644         * @throws CoordinatorJobException
645         */
646        protected void includeDataSets(Element resolvedXml, Configuration conf) throws CoordinatorJobException
647            /* throws Exception */ {
648            Element datasets = resolvedXml.getChild("datasets", resolvedXml.getNamespace());
649            Element allDataSets = new Element("all_datasets", resolvedXml.getNamespace());
650            List<String> dsList = new ArrayList<String>();
651            if (datasets != null) {
652                for (Element includeElem : (List<Element>) datasets.getChildren("include", datasets.getNamespace())) {
653                    String incDSFile = includeElem.getTextTrim();
654                    // log.warn(" incDSFile " + incDSFile);
655                    includeOneDSFile(incDSFile, dsList, allDataSets, datasets.getNamespace());
656                }
657                for (Element e : (List<Element>) datasets.getChildren("dataset", datasets.getNamespace())) {
658                    String dsName = (String) e.getAttributeValue("name");
659                    if (dsList.contains(dsName)) {// Override with this DS
660                        // Remove old DS
661                        removeDataSet(allDataSets, dsName);
662                        // throw new RuntimeException("Duplicate Dataset " +
663                        // dsName);
664                    }
665                    else {
666                        dsList.add(dsName);
667                    }
668                    allDataSets.addContent((Element) e.clone());
669                }
670            }
671            insertDataSet(resolvedXml, allDataSets);
672            resolvedXml.removeChild("datasets", resolvedXml.getNamespace());
673        }

675        /**
676         * Include One Dataset file.
677         *
678         * @param incDSFile : Include data set filename.
679         * @param dsList :List of dataset names to verify the duplicate.
680         * @param allDataSets : Element that includes all dataset definitions.
681         * @param dsNameSpace : Data set name space
682         * @throws CoordinatorJobException
683         * @throws Exception
684         */
685        private void includeOneDSFile(String incDSFile, List<String> dsList, Element allDataSets, Namespace dsNameSpace)
686                throws CoordinatorJobException {
687            Element tmpDataSets = null;
688            try {
689                String dsXml = readDefinition(incDSFile, "");
690                log.debug("DSFILE :" + incDSFile + "\n" + dsXml);
691                tmpDataSets = XmlUtils.parseXml(dsXml);
692            }
693            /*
694             * catch (IOException iex) {XLog.getLog(getClass()).warn(
695             * "Error reading included dataset file [{0}].  Message [{1}]",
696             * incDSFile, iex.getMessage()); throw new
697             * CommandException(ErrorCode.E0803, iex.getMessage()); }
698             */
699            catch (JDOMException e) {
700                log.warn("Error parsing included dataset [{0}].  Message [{1}]", incDSFile, e.getMessage());
701                throw new CoordinatorJobException(ErrorCode.E0700, e.getMessage());
702            }
703            resolveDataSets((List<Element>) tmpDataSets.getChildren("dataset"));
704            for (Element e : (List<Element>) tmpDataSets.getChildren("dataset")) {
705                String dsName = (String) e.getAttributeValue("name");
706                if (dsList.contains(dsName)) {
707                    throw new RuntimeException("Duplicate Dataset " + dsName);
708                }
709                dsList.add(dsName);
710                Element tmp = (Element) e.clone();
711                // TODO: Don't like to over-write the external/include DS's
712                // namespace
713                tmp.setNamespace(dsNameSpace);// TODO:
714                tmp.getChild("uri-template").setNamespace(dsNameSpace);
715                if (e.getChild("done-flag") != null) {
716                    tmp.getChild("done-flag").setNamespace(dsNameSpace);
717                }
718                allDataSets.addContent(tmp);
719            }
720            // nested include
721            for (Element includeElem : (List<Element>) tmpDataSets.getChildren("include", tmpDataSets.getNamespace())) {
722                String incFile = includeElem.getTextTrim();
723                // log.warn("incDSFile "+ incDSFile);
724                includeOneDSFile(incFile, dsList, allDataSets, dsNameSpace);
725            }
726        }

745         *
746         * @param appPath application path.
747         * @param user user name.
748         * @param group group name.
749         * @param autToken authentication token.
750         * @return workflow definition.
751         * @throws WorkflowException thrown if the definition could not be read.
752         */
753        protected String readDefinition(String appPath, String fileName) throws CoordinatorJobException {// TODO:
754            String user = ParamChecker.notEmpty(conf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
755            String group = ParamChecker.notEmpty(conf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
756            Configuration confHadoop = CoordUtils.getHadoopConf(conf);
757            try {
758                URI uri = new URI(appPath);
759                log.debug("user =" + user + " group =" + group);
760                FileSystem fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, uri,
761                                                                                                 new Configuration());
762                Path p;
763                if (fileName == null || fileName.length() == 0) {
764                    p = new Path(uri.getPath());
765                }
766                else {
767                    p = new Path(uri.getPath(), fileName);
768                }
769                // Reader reader = new InputStreamReader(fs.open(new Path(uri
770                // .getPath(), fileName)));
771                Reader reader = new InputStreamReader(fs.open(p));// TODO
772                StringWriter writer = new StringWriter();
773                IOUtils.copyCharStream(reader, writer);
774                return writer.toString();
775            }
776            catch (IOException ex) {
777                log.warn("IOException :" + XmlUtils.prettyPrint(confHadoop), ex);
778                throw new CoordinatorJobException(ErrorCode.E1001, ex.getMessage(), ex); // TODO:
779            }
780            catch (URISyntaxException ex) {
781                log.warn("URISyException :" + ex.getMessage());
782                throw new CoordinatorJobException(ErrorCode.E1002, appPath, ex.getMessage(), ex);// TODO:
783            }
784            catch (Exception ex) {
785                log.warn("Exception :", ex);
786                throw new CoordinatorJobException(ErrorCode.E1001, ex.getMessage(), ex);// TODO:
787            }
