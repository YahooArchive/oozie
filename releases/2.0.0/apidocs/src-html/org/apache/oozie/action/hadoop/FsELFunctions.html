018    package org.apache.oozie.action.hadoop;
019    
020    import java.io.IOException;
021    import java.net.URI;
022    import java.net.URISyntaxException;
023    
024    import org.apache.hadoop.fs.FileStatus;
025    import org.apache.hadoop.fs.FileSystem;
026    import org.apache.hadoop.fs.Path;
027    import org.apache.hadoop.conf.Configuration;
028    import org.apache.oozie.DagELFunctions;
029    import org.apache.oozie.client.WorkflowJob;
030    import org.apache.oozie.service.Services;
031    import org.apache.oozie.service.HadoopAccessorService;
032    
033    /**
034     * EL function for fs action executor.
035     */
036    public class FsELFunctions {
037    
038        private static FileSystem getFileSystem(URI uri) throws IOException {
039            WorkflowJob workflow = DagELFunctions.getWorkflow();
040            String user = workflow.getUser();
041            String group = workflow.getGroup();
042            return Services.get().get(HadoopAccessorService.class).
043                    createFileSystem(user, group, uri, new Configuration());
044        }
045    
046        /**
047         * Get file status.
048         *
049         * @param pathUri fs path uri
050         * @return file status
051         * @throws URISyntaxException
052         * @throws IOException
053         * @throws Exception
054         */
055        private static FileStatus getFileStatus(String pathUri) throws Exception {
056            URI uri = new URI(pathUri);
057            String path = uri.getPath();
058            FileSystem fs = getFileSystem(uri);
059            Path p = new Path(path);
060            return fs.exists(p) ? fs.getFileStatus(p) : null;
061        }
062    
063        /**
064         * Return if a path exists.
065         *
066         * @param pathUri file system path uri.
067         * @return <code>true</code> if the path exists, <code>false</code> if it does not.
068         * @throws Exception
069         */
070        public static boolean fs_exists(String pathUri) throws Exception {
071            URI uri = new URI(pathUri);
072            String path = uri.getPath();
073            FileSystem fs = getFileSystem(uri);
074            return fs.exists(new Path(path));
075        }
076    
077        /**
078         * Return if a path is a directory.
079         *
080         * @param pathUri fs path uri.
081         * @return <code>true</code> if the path exists and it is a directory, <code>false</code> otherwise.
082         * @throws Exception
083         */
084        public static boolean fs_isDir(String pathUri) throws Exception {
085            boolean isDir = false;
086            FileStatus fileStatus = getFileStatus(pathUri);
087            if (fileStatus != null) {
088                isDir = fileStatus.isDir();
089            }
090            return isDir;
091        }
092    
093        /**
094         * Return the len of a file.
095         *
096         * @param pathUri file system path uri.
097         * @return the file len in bytes, -1 if the file does not exist or if it is a directory.
098         * @throws Exception
099         */
100        public static long fs_fileSize(String pathUri) throws Exception {
101            long len = -1;
102            FileStatus fileStatus = getFileStatus(pathUri);
103            if (fileStatus != null) {
104                len = fileStatus.getLen();
105            }
106            return len;
107        }
108    
109        /**
110         * Return the size of all files in the directory, it is not recursive.
111         *
112         * @param pathUri file system path uri.
113         * @return the size of all files in the directory, -1 if the directory does not exist or if it is a file.
114         * @throws Exception
115         */
116        public static long fs_dirSize(String pathUri) throws Exception {
117            URI uri = new URI(pathUri);
118            String path = uri.getPath();
119            long size = -1;
120            try {
121                FileSystem fs = getFileSystem(uri);
122                Path p = new Path(path);
123                if (fs.exists(p) && !fs.isFile(p)) {
124                    FileStatus[] stati = fs.listStatus(p);
125                    size = 0;
126                    if (stati != null) {
127                        for (FileStatus status : stati) {
128                            if (!status.isDir()) {
129                                size += status.getLen();
130                            }
131                        }
132                    }
133                }
134            }
135            catch (Exception ex) {
136                throw new RuntimeException(ex);
137            }
138            return size;
139        }
140    
141        /**
142         * Return the file block size in bytes.
143         *
144         * @param pathUri file system path uri.
145         * @return the block size of the file in bytes, -1 if the file does not exist or if it is a directory.
146         * @throws Exception
147         */
148        public static long fs_blockSize(String pathUri) throws Exception {
149            long blockSize = -1;
150            FileStatus fileStatus = getFileStatus(pathUri);
151            if (fileStatus != null) {
152                blockSize = fileStatus.getBlockSize();
153            }
154            return blockSize;
155        }
156    
157    }
