018    package org.apache.oozie.util;
019    
020    import org.apache.hadoop.io.Writable;
021    
022    import java.util.concurrent.Callable;
023    
024    
025    /**
026     * Extends Callable adding the concept of priority. <p/> The priority is useful when queuing callables for later
027     * execution via the {@link org.apache.oozie.service.CallableQueueService}. <p/> A higher number means a higher
028     * priority. <p/>
029     */
030    public interface XCallable<T> extends Callable<T> {
031    
032        /**
033         * Return the callable name.
034         *
035         * @return the callable name.
036         */
037        public String getName();
038    
039        /**
040         * Return the priority of the callable.
041         *
042         * @return the callable priority.
043         */
044        public int getPriority();
045    
046        /**
047         * Return the callable type. <p/> The callable type is used for concurrency throttling in the {@link
048         * org.apache.oozie.service.CallableQueueService}.
049         *
050         * @return the callable type.
051         */
052        public String getType();
053    
054        /**
055         * Returns the createdTime of the callable in milliseconds
056         *
057         * @return the callable createdTime
058         */
059        public long getCreatedTime();
060    
061    }
