055        /**
056         * v1 service implementation to submit a job, either workflow or coordinator
057         */
058        protected JSONObject submitJob(HttpServletRequest request, Configuration conf) throws XServletException,
059                IOException {
060            JSONObject json = null;
061            String wfPath = conf.get(OozieClient.APP_PATH);
062            String coordPath = conf.get(OozieClient.COORDINATOR_APP_PATH);
063    
064            ServletUtilities.ValidateAppPath(wfPath, coordPath);
065    
066            if (wfPath != null) {
067                json = submitWorkflowJob(request, conf);
068            }
069            else {
070                json = submitCoordinatorJob(request, conf);
071            }
072            return json;
073        }

123        /**
124         * v1 service implementation to submit a workflow job
125         */
126        private JSONObject submitWorkflowJob(HttpServletRequest request, Configuration conf) throws XServletException {
127    
128            JSONObject json = new JSONObject();
129    
130            try {
131                String action = request.getParameter(RestConstants.ACTION_PARAM);
132                if (action != null && !action.equals(RestConstants.JOB_ACTION_START)) {
133                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0303,
134                                                RestConstants.ACTION_PARAM, action);
135                }
136                boolean startJob = (action != null);
137                String user = conf.get(OozieClient.USER_NAME);
138                DagEngine dagEngine = Services.get().get(DagEngineService.class).getDagEngine(user, getAuthToken(request));
139                String id = dagEngine.submitJob(conf, startJob);
140                json.put(JsonTags.JOB_ID, id);
141            }
142            catch (DagEngineException ex) {
143                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
144            }
145    
146            return json;
147        }

149        /**
150         * v1 service implementation to submit a coordinator job
151         */
152        private JSONObject submitCoordinatorJob(HttpServletRequest request, Configuration conf) throws XServletException {
153    
154            JSONObject json = new JSONObject();
155            XLog.getLog(getClass()).warn("submitCoordinatorJob " + XmlUtils.prettyPrint(conf).toString());
156            try {
157                String action = request.getParameter(RestConstants.ACTION_PARAM);
158                if (action != null && !action.equals(RestConstants.JOB_ACTION_START)
159                        && !action.equals(RestConstants.JOB_ACTION_DRYRUN)) {
160                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0303,
161                                                RestConstants.ACTION_PARAM, action);
162                }
163                boolean startJob = (action != null);
164                String user = conf.get(OozieClient.USER_NAME);
165                CoordinatorEngine coordEngine = Services.get().get(CoordinatorEngineService.class).getCoordinatorEngine(
166                        user, getAuthToken(request));
167                String id = null;
168                boolean dryrun = false;
169                if (action != null) {
170                    dryrun = (action.equals(RestConstants.JOB_ACTION_DRYRUN));
171                }
172                if (dryrun) {
173                    id = coordEngine.dryrunSubmit(conf, startJob);
174                }
175                else {
176                    id = coordEngine.submitJob(conf, startJob);
177                }
178                json.put(JsonTags.JOB_ID, id);
179            }
180            catch (CoordinatorEngineException ex) {
181                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
182            }
183    
184            return json;
185        }

215        /**
216         * v1 service implementation to get a list of workflows, with filtering or interested windows embedded in the
217         * request object
218         */
219        private JSONObject getWorkflowJobs(HttpServletRequest request) throws XServletException {
220            JSONObject json = new JSONObject();
221            try {
222                String filter = request.getParameter(RestConstants.JOBS_FILTER_PARAM);
223                String startStr = request.getParameter(RestConstants.OFFSET_PARAM);
224                String lenStr = request.getParameter(RestConstants.LEN_PARAM);
225                int start = (startStr != null) ? Integer.parseInt(startStr) : 1;
226                start = (start < 1) ? 1 : start;
227                int len = (lenStr != null) ? Integer.parseInt(lenStr) : 50;
228                len = (len < 1) ? 50 : len;
229                DagEngine dagEngine = Services.get().get(DagEngineService.class).getDagEngine(getUser(request),
230                                                                                              getAuthToken(request));
231                WorkflowsInfo jobs = dagEngine.getJobs(filter, start, len);
232                List<WorkflowJobBean> jsonWorkflows = jobs.getWorkflows();
233                json.put(JsonTags.WORKFLOWS_JOBS, WorkflowJobBean.toJSONArray(jsonWorkflows));
234                json.put(JsonTags.WORKFLOWS_TOTAL, jobs.getTotal());
235                json.put(JsonTags.WORKFLOWS_OFFSET, jobs.getStart());
236                json.put(JsonTags.WORKFLOWS_LEN, jobs.getLen());
237    
238            }
239            catch (DagEngineException ex) {
240                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
241            }
242    
243            return json;
244        }
