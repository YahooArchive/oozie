018    package org.apache.oozie.servlet;
019    
020    import java.util.Arrays;
021    import java.util.Collections;
022    
023    import javax.servlet.http.HttpServletRequest;
024    import javax.servlet.http.HttpServletResponse;
025    
026    import org.apache.oozie.ErrorCode;
027    import org.apache.oozie.client.OozieClient.SYSTEM_MODE;
028    import org.apache.oozie.client.rest.JsonTags;
029    import org.apache.oozie.client.rest.RestConstants;
030    import org.apache.oozie.service.Services;
031    import org.apache.oozie.servlet.JsonRestServlet.ParameterInfo;
032    import org.apache.oozie.servlet.JsonRestServlet.ResourceInfo;
033    import org.json.simple.JSONObject;
034    
035    public class V1AdminServlet extends BaseAdminServlet {
036    
037        private static final String INSTRUMENTATION_NAME = "v1admin";
038        private static final ResourceInfo RESOURCES_INFO[] = new ResourceInfo[6];
039    
040        static {
041            RESOURCES_INFO[0] = new ResourceInfo(RestConstants.ADMIN_STATUS_RESOURCE, Arrays.asList("PUT", "GET"),
042                                                 Arrays.asList(new ParameterInfo(RestConstants.ADMIN_SYSTEM_MODE_PARAM, String.class, true,
043                                                                                 Arrays.asList("PUT"))));
044            RESOURCES_INFO[1] = new ResourceInfo(RestConstants.ADMIN_OS_ENV_RESOURCE, Arrays.asList("GET"),
045                                                 Collections.EMPTY_LIST);
046            RESOURCES_INFO[2] = new ResourceInfo(RestConstants.ADMIN_JAVA_SYS_PROPS_RESOURCE, Arrays.asList("GET"),
047                                                 Collections.EMPTY_LIST);
048            RESOURCES_INFO[3] = new ResourceInfo(RestConstants.ADMIN_CONFIG_RESOURCE, Arrays.asList("GET"),
049                                                 Collections.EMPTY_LIST);
050            RESOURCES_INFO[4] = new ResourceInfo(RestConstants.ADMIN_INSTRUMENTATION_RESOURCE, Arrays.asList("GET"),
051                                                 Collections.EMPTY_LIST);
052            RESOURCES_INFO[5] = new ResourceInfo(RestConstants.ADMIN_BUILD_VERSION_RESOURCE, Arrays.asList("GET"),
053                                                 Collections.EMPTY_LIST);
054        }
055    
056        public V1AdminServlet() {
057            super(INSTRUMENTATION_NAME, RESOURCES_INFO);
058            modeTag = RestConstants.ADMIN_SYSTEM_MODE_PARAM;
059        }
060    
061        protected void populateOozieMode(JSONObject json) {
062            System.out.println(" getOozeMode " + Services.get().getSystemMode());
063            json.put(JsonTags.OOZIE_SYSTEM_MODE, Services.get().getSystemMode().toString());
064        }
065    
066        @Override
067        protected void setOozieMode(HttpServletRequest request,
068                                    HttpServletResponse response, String resourceName)
069                throws XServletException {
070            if (resourceName.equals(RestConstants.ADMIN_STATUS_RESOURCE)) {
071                SYSTEM_MODE sysMode = SYSTEM_MODE.valueOf(request.getParameter(modeTag));
072                System.out.println(modeTag + " CCCC " + sysMode);
073                Services.get().setSystemMode(sysMode);
074                //Services.get().setSafeMode(safeMode);
075                response.setStatus(HttpServletResponse.SC_OK);
076            }
077            else {
078                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST,
079                                            ErrorCode.E0301, resourceName);
080            }
081        }
082    
083    }
