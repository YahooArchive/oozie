066        private Map<String, Integer> statusWindowCounts = new HashMap<String, Integer>();

068        /**
069         * Gets the number of workflows for each status and populates the hash.
070         */
071        class JobStatusCountCallable implements Runnable {
072            @Override
073            public void run() {
074                WorkflowStore store = null;
075                try {
076                    store = Services.get().get(WorkflowStoreService.class).create();
077                    store.beginTrx();
078                    WorkflowJob.Status[] wfStatusArr = WorkflowJob.Status.values();
079                    for (int i = 0; i < wfStatusArr.length; i++) {
080                        statusCounts.put(wfStatusArr[i].name(), store.getWorkflowCountWithStatus(wfStatusArr[i].name()));
081                        statusWindowCounts.put(wfStatusArr[i].name(), store.getWorkflowCountWithStatusInLastNSeconds(
082                                wfStatusArr[i].name(), statusWindow));
083                    }
084                    store.commitTrx();
085                }
086                catch (StoreException e) {
087                    if (store != null) {
088                        store.rollbackTrx();
089                    }
090                    log.warn("Exception while accessing the store", e);
091                }
092                finally {
093                    try {
094                        if (store != null) {
095                            store.closeTrx();
096                        }
097                    }
098                    catch (RuntimeException rex) {
099                        log.warn("Exception while attempting to close store", rex);
100                    }
101                }
102            }
103        }

228        /**
229         * Set up the oozie schema, create the tables and indexes. Also insert the Version number to version table.
230         *
231         * @param conn DB connection
232         * @param dbType DB Type
233         * @throws ServiceException On failure to create the schema
234         */
235        @Deprecated
236        private void setupOozieSchema(Connection conn, DBType dbType) throws ServiceException {
237            String errorMessage = "";
238            try {
239                if (!dbType.equals(DBType.ORACLE)) {
240                    errorMessage = "Failed to create Schema.\n{0}";
241                    createSchema(conn, dbType);
242                }
243                errorMessage = "Failed to create Tables.\n{0}";
244                createTables(conn, dbType);
245                errorMessage = "Failed to create Indexes.\n{0}";
246                createIndexes(conn, dbType);
247                errorMessage = "Failed to insert version data";
248                doUpdate(conn, OozieSchema.generateInsertVersionScript(schemaName));
249                errorMessage = "";
250            }
251            catch (SQLException e) {
252                throw new ServiceException(ErrorCode.E0603, XLog.format(errorMessage, e.getMessage()), e);
253            }
254        }

274        /**
275         * Check for the support for "select for update" by the DB.
276         *
277         * @param conn Connection
278         * @param validation_query query to executed. adds for update to validation_query and will do the check
279         */
280        @Deprecated
281        private void checkAndSetSelectForUpdateSupport(Connection conn, String validation_query) {
282            try {
283                if (validation_query != null) {
284                    Statement statment = conn.createStatement();
285                    ResultSet resultSet = statment.executeQuery(validation_query + " for update");
286                    resultSet.close();
287                    statment.close();
288                    selectForUpdate = true;
289                }
290            }
291            catch (SQLException ex) {
292                selectForUpdate = false;
293            }
294        }

327        @Override
328        public void instrument(Instrumentation instr) {
329            final WorkflowJob.Status[] wfStatusArr = WorkflowJob.Status.values();
330            for (int i = 0; i < wfStatusArr.length; i++) {
331                final String statusName = wfStatusArr[i].name();
332                instr.addVariable(INSTRUMENTATION_GROUP, statusName, new Instrumentation.Variable<Long>() {
333                    public Long getValue() {
334                        return statusCounts.get(statusName).longValue();
335                    }
336                });
337                instr.addVariable(INSTRUMENTATION_GROUP_WINDOW, statusName, new Instrumentation.Variable<Long>() {
338                    public Long getValue() {
339                        return statusWindowCounts.get(statusName).longValue();
340                    }
341                });
342            }
343        }
