018    package org.apache.oozie.service;
019    
020    import org.apache.oozie.util.ParamChecker;
021    import org.apache.oozie.util.XLog;
022    import org.apache.oozie.ErrorCode;
023    
024    import java.text.SimpleDateFormat;
025    import java.util.Date;
026    import java.util.UUID;
027    import java.util.concurrent.atomic.AtomicLong;
028    
029    /**
030     * The UUID service generates unique IDs. <p/> The configuration property {@link #CONF_GENERATOR} specifies the ID
031     * generation type, 'random' or 'counter'. <p/> For 'random' uses the JDK UUID.randomUUID() method. <p/> For 'counter'
032     * uses a counter postfixed wit the system start up time.
033     */
034    public class UUIDService implements Service {
035    
036        public static final String CONF_PREFIX = Service.CONF_PREFIX + "UUIDService.";
037    
038        public static final String CONF_GENERATOR = CONF_PREFIX + "generator";
039    
040        private String startTime;
041        private AtomicLong counter;
042        private String systemId;
043    
044        /**
045         * Initialize the UUID service.
046         *
047         * @param services services instance.
048         * @throws ServiceException thrown if the UUID service could not be initialized.
049         */
050        @Override
051        public void init(Services services) throws ServiceException {
052            String genType = services.getConf().get(CONF_GENERATOR, "counter").trim();
053            if (genType.equals("counter")) {
054                counter = new AtomicLong();
055                startTime = new SimpleDateFormat("yyMMddHHmmssSSS").format(new Date());
056            }
057            else {
058                if (!genType.equals("random")) {
059                    throw new ServiceException(ErrorCode.E0120, genType);
060                }
061            }
062            systemId = services.getSystemId();
063        }
064    
065        /**
066         * Destroy the UUID service.
067         */
068        @Override
069        public void destroy() {
070            counter = null;
071            startTime = null;
072        }
073    
074        /**
075         * Return the public interface for UUID service.
076         *
077         * @return {@link UUIDService}.
078         */
079        @Override
080        public Class<? extends Service> getInterface() {
081            return UUIDService.class;
082        }
083    
084        private String longPadding(long number) {
085            StringBuilder sb = new StringBuilder();
086            sb.append(number);
087            if (sb.length() <= 7) {
088                sb.insert(0, "0000000".substring(sb.length()));
089            }
090            return sb.toString();
091        }
092    
093        /**
094         * Create a unique ID.
095         *
096         * @param type: Type of Id. Generally 'C' for Coordinator and 'W' for Workflow.
097         * @return unique ID.
098         */
099        public String generateId(ApplicationType type) {
100            StringBuilder sb = new StringBuilder();
101    
102            if (counter != null) {
103                sb.append(longPadding(counter.getAndIncrement())).append('-').append(startTime);
104            }
105            else {
106                sb.append(UUID.randomUUID().toString());
107                if (sb.length() > (37 - systemId.length())) {
108                    sb.setLength(37 - systemId.length());
109                }
110            }
111            sb.append('-').append(systemId);
112            sb.append('-').append(type.getType());
113            // limitation due to current DB schema for action ID length (100)
114            if (sb.length() > 40) {
115                throw new RuntimeException(XLog.format("ID exceeds limit of 40 characters, [{0}]", sb));
116            }
117            return sb.toString();
118        }
119    
120        /**
121         * Create a child ID. <p/> If the same child name is given the returned child ID is the same.
122         *
123         * @param id unique ID.
124         * @param childName child name.
125         * @return a child ID.
126         */
127        public String generateChildId(String id, String childName) {
128            id = ParamChecker.notEmpty(id, "id") + "@" + ParamChecker.notEmpty(childName, "childName");
129    
130            // limitation due to current DB schema for action ID length (100)
131            if (id.length() > 95) {
132                throw new RuntimeException(XLog.format("Child ID exceeds limit of 95 characters, [{0}]", id));
133            }
134            return id;
135        }
136    
137        /**
138         * Return the ID from a child ID.
139         *
140         * @param childId child ID.
141         * @return ID of the child ID.
142         */
143        public String getId(String childId) {
144            int index = ParamChecker.notEmpty(childId, "childId").indexOf("@");
145            if (index == -1) {
146                throw new IllegalArgumentException(XLog.format("invalid child id [{0}]", childId));
147            }
148            return childId.substring(0, index);
149        }
150    
151        /**
152         * Return the child name from a child ID.
153         *
154         * @param childId child ID.
155         * @return child name.
156         */
157        public String getChildName(String childId) {
158            int index = ParamChecker.notEmpty(childId, "childId").indexOf("@");
159            if (index == -1) {
160                throw new IllegalArgumentException(XLog.format("invalid child id [{0}]", childId));
161            }
162            return childId.substring(index + 1);
163        }
164    
165        public enum ApplicationType {
166            WORKFLOW('W'), COORDINATOR('C');
167            private char type;
168    
169            private ApplicationType(char type) {
170                this.type = type;
171            }
172    
173            public char getType() {
174                return type;
175            }
176        }
177    }
