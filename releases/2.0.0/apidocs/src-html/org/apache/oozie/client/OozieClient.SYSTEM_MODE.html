059     * Client API to submit and manage Oozie workflow jobs against an Oozie intance. <p/> This class is thread safe. <p/>
060     * Syntax for filter for the {@link #getJobsInfo(String)} {@link #getJobsInfo(String, int, int)} methods:
061     * <code>[NAME=VALUE][;NAME=VALUE]*</code>. <p/> Valid filter names are: <p/> <ul/> <li>name: the workflow application
062     * name from the workflow definition.</li> <li>user: the user that submitted the job.</li> <li>group: the group for the
063     * job.</li> <li>status: the status of the job.</li> </ul> <p/> The query will do an AND among all the filter names. The
064     * query will do an OR among all the filter values for the same name. Multiple values must be specified as different
065     * name value pairs.

171         * Validate that the Oozie client and server instances are protocol compatible.
172         *
173         * @throws OozieClientException thrown in the client and the server are not protocol compatible.
174         */
175        public synchronized void validateWSVersion() throws OozieClientException {
176            if (!validatedVersion) {
177                try {
178                    URL url = new URL(baseUrl + RestConstants.VERSIONS);
179                    HttpURLConnection conn = createConnection(url, "GET");
180                    if (conn.getResponseCode() == HttpURLConnection.HTTP_OK) {
181                        JSONArray array = (JSONArray) JSONValue.parse(new InputStreamReader(conn.getInputStream()));
182                        if (array == null) {
183                            throw new OozieClientException("HTTP error", "no response message");
184                        }
185                        if (!array.contains(WS_PROTOCOL_VERSION) && !array.contains(WS_PROTOCOL_VERSION_0)) {
186                            StringBuilder msg = new StringBuilder();
187                            msg.append("Supported version [").append(WS_PROTOCOL_VERSION).append(
188                                    "] or less, Unsupported versions[");
189                            String separator = "";
190                            for (Object version : array) {
191                                msg.append(separator).append(version);
192                            }
193                            msg.append("]");
194                            throw new OozieClientException(OozieClientException.UNSUPPORTED_VERSION, msg.toString());
195                        }
196                        if (array.contains(WS_PROTOCOL_VERSION)) {
197                            protocolUrl = baseUrl + "v" + WS_PROTOCOL_VERSION + "/";
198                        }
199                        else {
200                            if (array.contains(WS_PROTOCOL_VERSION_0)) {
201                                protocolUrl = baseUrl + "v" + WS_PROTOCOL_VERSION_0 + "/";
202                            }
203                        }
204                    }
205                    else {
206                        handleError(conn);
207                    }
208                }
209                catch (IOException ex) {
210                    throw new OozieClientException(OozieClientException.IO_ERROR, ex);
211                }
212                validatedVersion = true;
213            }
214        }

265        private URL createURL(String collection, String resource, Map<String, String> parameters) throws IOException,
266                OozieClientException {
267            validateWSVersion();
268            StringBuilder sb = new StringBuilder();
269            sb.append(protocolUrl).append(collection);
270            if (resource != null && resource.length() > 0) {
271                sb.append("/").append(resource);
272            }
273            if (parameters.size() > 0) {
274                String separator = "?";
275                for (Map.Entry<String, String> param : parameters.entrySet()) {
276                    if (param.getValue() != null) {
277                        sb.append(separator).append(URLEncoder.encode(param.getKey(), "UTF-8")).append("=").append(
278                                URLEncoder.encode(param.getValue(), "UTF-8"));
279                        separator = "&";
280                    }
281                }
282            }
283            return new URL(sb.toString());
284        }

286        private boolean validateCommand(String url) {
287            {
288                if (protocolUrl.contains(baseUrl + "v0")) {
289                    if (url.contains("dryrun") || url.contains("jobtype=c") || url.contains("systemmode")) {
290                        return false;
291                    }
292                }
293            }
294            return true;
295        }

297        private HttpURLConnection createConnection(URL url, String method) throws IOException, OozieClientException {
298            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
299            conn.setRequestMethod(method);
300            if (method.equals("POST") || method.equals("PUT")) {
301                conn.setDoOutput(true);
302            }
303            for (Map.Entry<String, String> header : headers.entrySet()) {
304                conn.setRequestProperty(header.getKey(), header.getValue());
305            }
306            return conn;
307        }

309        private abstract class ClientCallable<T> implements Callable<T> {
310            private String method;
311            private String collection;
312            private String resource;
313            private Map<String, String> params;
314    
315            public ClientCallable(String method, String collection, String resource, Map<String, String> params) {
316                this.method = method;
317                this.collection = collection;
318                this.resource = resource;
319                this.params = params;
320            }
321    
322            public T call() throws OozieClientException {
323                try {
324                    URL url = createURL(collection, resource, params);
325                    if (validateCommand(url.toString())) {
326                        HttpURLConnection conn = createConnection(url, method);
327                        return (T) call(conn);
328                    }
329                    else {
330                        System.out
331                                .println("Option not supported in target server. Supported only on Oozie-2.0 or greater. Use 'oozie help' for details");
332                        throw new OozieClientException(OozieClientException.UNSUPPORTED_VERSION, new Exception());
333                    }
334                }
335                catch (IOException ex) {
336                    throw new OozieClientException(OozieClientException.IO_ERROR, ex);
337                }
338    
339            }
340    
341            protected abstract T call(HttpURLConnection conn) throws IOException, OozieClientException;
342        }
343    
344        private static void handleError(HttpURLConnection conn) throws IOException, OozieClientException {
345            int status = conn.getResponseCode();
346            String error = conn.getHeaderField(RestConstants.OOZIE_ERROR_CODE);
347            String message = conn.getHeaderField(RestConstants.OOZIE_ERROR_MESSAGE);
348    
349            if (error == null) {
350                error = "HTTP error code: " + status;
351            }
352    
353            if (message == null) {
354                message = conn.getResponseMessage();
355            }
356            throw new OozieClientException(error, message);
357        }

359        private static Map<String, String> prepareParams(String... params) {
360            Map<String, String> map = new HashMap<String, String>();
361            for (int i = 0; i < params.length; i = i + 2) {
362                map.put(params[i], params[i + 1]);
363            }
364            return map;
365        }

367        public void writeToXml(Properties props, OutputStream out) throws IOException {
368            try {
369                Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder().newDocument();
370                Element conf = doc.createElement("configuration");
371                doc.appendChild(conf);
372                conf.appendChild(doc.createTextNode("\n"));
373                for (Enumeration e = props.keys(); e.hasMoreElements();) {
374                    String name = (String) e.nextElement();
375                    Object object = props.get(name);
376                    String value;
377                    if (object instanceof String) {
378                        value = (String) object;
379                    }
380                    else {
381                        continue;
382                    }
383                    Element propNode = doc.createElement("property");
384                    conf.appendChild(propNode);
385    
386                    Element nameNode = doc.createElement("name");
387                    nameNode.appendChild(doc.createTextNode(name.trim()));
388                    propNode.appendChild(nameNode);
389    
390                    Element valueNode = doc.createElement("value");
391                    valueNode.appendChild(doc.createTextNode(value.trim()));
392                    propNode.appendChild(valueNode);
393    
394                    conf.appendChild(doc.createTextNode("\n"));
395                }
396    
397                DOMSource source = new DOMSource(doc);
398                StreamResult result = new StreamResult(out);
399                TransformerFactory transFactory = TransformerFactory.newInstance();
400                Transformer transformer = transFactory.newTransformer();
401                transformer.transform(source, result);
402            }
403            catch (Exception e) {
404                throw new IOException(e);
405            }
406        }

408        private class JobSubmit extends ClientCallable<String> {
409            private Properties conf;
410    
411            JobSubmit(Properties conf, boolean start) {
412                super("POST", RestConstants.JOBS, "", (start) ? prepareParams(RestConstants.ACTION_PARAM,
413                                                                              RestConstants.JOB_ACTION_START) : prepareParams());
414                this.conf = notNull(conf, "conf");
415            }
416    
417            JobSubmit(String jobId, Properties conf) {
418                super("PUT", RestConstants.JOB, notEmpty(jobId, "jobId"), prepareParams(RestConstants.ACTION_PARAM,
419                                                                                        RestConstants.JOB_ACTION_RERUN));
420                this.conf = notNull(conf, "conf");
421            }
422    
423            public JobSubmit(Properties conf, String jobActionDryrun) {
424                super("POST", RestConstants.JOBS, "", prepareParams(RestConstants.ACTION_PARAM,
425                                                                    RestConstants.JOB_ACTION_DRYRUN));
426                this.conf = notNull(conf, "conf");
427                // TODO Auto-generated constructor stub
428            }
429    
430            protected String call(HttpURLConnection conn) throws IOException, OozieClientException {
431                conn.setRequestProperty("content-type", RestConstants.XML_CONTENT_TYPE);
432                writeToXml(conf, conn.getOutputStream());
433                if (conn.getResponseCode() == HttpURLConnection.HTTP_CREATED) {
434                    JSONObject json = (JSONObject) JSONValue.parse(new InputStreamReader(conn.getInputStream()));
435                    return (String) json.get(JsonTags.JOB_ID);
436                }
437                if (conn.getResponseCode() != HttpURLConnection.HTTP_OK) {
438                    handleError(conn);
439                }
440                return null;
441            }
442        }

659                if ((conn.getResponseCode() == HttpURLConnection.HTTP_OK)) {
660    
661                    String output = getReaderAsString(new InputStreamReader(conn.getInputStream()), -1);
662                    return output;
663                }
664                else {

670            /**
671             * Return a reader as string. <p/>
672             *
673             * @param reader reader to read into a string.
674             * @param maxLen max content length allowed, if -1 there is no limit.
675             * @return the reader content.
676             * @throws IOException thrown if the resource could not be read.
677             */
678            private String getReaderAsString(Reader reader, int maxLen) throws IOException {
679                if (reader == null) {
680                    throw new IllegalArgumentException("reader cannot be null");
681                }
682    
683                StringBuffer sb = new StringBuffer();
684                char[] buffer = new char[2048];
685                int read;
686                int count = 0;
687                while ((read = reader.read(buffer)) > -1) {
688                    count += read;
689    
690                    // read up to maxLen chars;
691                    if ((maxLen > -1) && (count > maxLen)) {
692                        break;
693                    }
694                    sb.append(buffer, 0, read);
695                }
696                reader.close();
697                return sb.toString();
698            }
