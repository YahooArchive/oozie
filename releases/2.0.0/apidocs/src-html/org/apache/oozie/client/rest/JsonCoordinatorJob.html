119        public JsonCoordinatorJob(JSONObject json) {
120            appPath = (String) json.get(JsonTags.COORDINATOR_JOB_PATH);
121            appName = (String) json.get(JsonTags.COORDINATOR_JOB_NAME);
122            id = (String) json.get(JsonTags.COORDINATOR_JOB_ID);
123            externalId = (String) json.get(JsonTags.COORDINATOR_JOB_EXTERNAL_ID);
124            conf = (String) json.get(JsonTags.COORDINATOR_JOB_CONF);
125            status = Status.valueOf((String) json.get(JsonTags.COORDINATOR_JOB_STATUS));
126            executionOrder = Execution.valueOf((String) json.get(JsonTags.COORDINATOR_JOB_EXECUTIONPOLICY));
127            startTime = JsonUtils.parseDateRfc822((String) json.get(JsonTags.COORDINATOR_JOB_START_TIME));
128            endTime = JsonUtils.parseDateRfc822((String) json.get(JsonTags.COORDINATOR_JOB_END_TIME));
129            frequency = (int) JsonUtils.getLongValue(json, JsonTags.COORDINATOR_JOB_FREQUENCY);
130            timeUnit = Timeunit.valueOf((String) json.get(JsonTags.COORDINATOR_JOB_TIMEUNIT));
131            timeZone = (String) json.get(JsonTags.COORDINATOR_JOB_TIMEZONE);
132            concurrency = (int) JsonUtils.getLongValue(json, JsonTags.COORDINATOR_JOB_CONCURRENCY);
133            timeOut = (int) JsonUtils.getLongValue(json, JsonTags.COORDINATOR_JOB_TIMEOUT);
134            lastAction = JsonUtils.parseDateRfc822((String) json.get(JsonTags.COORDINATOR_JOB_LAST_ACTION_TIME));
135            nextMaterializedTime = JsonUtils.parseDateRfc822((String) json
136                    .get(JsonTags.COORDINATOR_JOB_NEXT_MATERIALIZED_TIME));
137            user = (String) json.get(JsonTags.COORDINATOR_JOB_USER);
138            group = (String) json.get(JsonTags.COORDINATOR_JOB_GROUP);
139            consoleUrl = (String) json.get(JsonTags.COORDINATOR_JOB_CONSOLE_URL);
140            actions = JsonCoordinatorAction.fromJSONArray((JSONArray) json.get(JsonTags.COORDINATOR_ACTIONS));
141        }
142    
143        @SuppressWarnings("unchecked")
144        public JSONObject toJSONObject() {
145            JSONObject json = new JSONObject();
146            json.put(JsonTags.COORDINATOR_JOB_PATH, appPath);
147            json.put(JsonTags.COORDINATOR_JOB_NAME, appName);
148            json.put(JsonTags.COORDINATOR_JOB_ID, id);
149            json.put(JsonTags.COORDINATOR_JOB_EXTERNAL_ID, externalId);
150            json.put(JsonTags.COORDINATOR_JOB_CONF, conf);
151            json.put(JsonTags.COORDINATOR_JOB_STATUS, status.toString());
152            json.put(JsonTags.COORDINATOR_JOB_EXECUTIONPOLICY, executionOrder.toString());
153            json.put(JsonTags.COORDINATOR_JOB_FREQUENCY, frequency);
154            json.put(JsonTags.COORDINATOR_JOB_TIMEUNIT, timeUnit.toString());
155            json.put(JsonTags.COORDINATOR_JOB_TIMEZONE, timeZone);
156            json.put(JsonTags.COORDINATOR_JOB_CONCURRENCY, concurrency);
157            json.put(JsonTags.COORDINATOR_JOB_TIMEOUT, timeOut);
158            json.put(JsonTags.COORDINATOR_JOB_LAST_ACTION_TIME, JsonUtils.formatDateRfc822(lastAction));
159            json.put(JsonTags.COORDINATOR_JOB_NEXT_MATERIALIZED_TIME, JsonUtils.formatDateRfc822(nextMaterializedTime));
160            json.put(JsonTags.COORDINATOR_JOB_START_TIME, JsonUtils.formatDateRfc822(startTime));
161            json.put(JsonTags.COORDINATOR_JOB_END_TIME, JsonUtils.formatDateRfc822(endTime));
162            json.put(JsonTags.COORDINATOR_JOB_USER, user);
163            json.put(JsonTags.COORDINATOR_JOB_GROUP, group);
164            json.put(JsonTags.COORDINATOR_JOB_CONSOLE_URL, consoleUrl);
165            json.put(JsonTags.COORDINATOR_ACTIONS, JsonCoordinatorAction.toJSONArray(actions));
166    
167            return json;
168        }

353        /**
354         * Convert a coordinator application list into a JSONArray.
355         *
356         * @param application list.
357         * @return the corresponding JSON array.
358         */
359        @SuppressWarnings("unchecked")
360        public static JSONArray toJSONArray(List<? extends JsonCoordinatorJob> applications) {
361            JSONArray array = new JSONArray();
362            if (applications != null) {
363                for (JsonCoordinatorJob application : applications) {
364                    array.add(application.toJSONObject());
365                }
366            }
367            return array;
368        }
369    
370        /**
371         * Convert a JSONArray into a application list.
372         *
373         * @param array JSON array.
374         * @return the corresponding application list.
375         */
376        @SuppressWarnings("unchecked")
377        public static List<CoordinatorJob> fromJSONArray(JSONArray applications) {
378            List<CoordinatorJob> list = new ArrayList<CoordinatorJob>();
379            for (Object obj : applications) {
380                list.add(new JsonCoordinatorJob((JSONObject) obj));
381            }
382            return list;
383        }
