060    public class SignalXCommand extends WorkflowXCommand<Void> {
061    
062        protected static final String INSTR_SUCCEEDED_JOBS_COUNTER_NAME = "succeeded";
063    
064        private JPAService jpaService = null;
065        private String jobId;
066        private String actionId;
067        private WorkflowJobBean wfJob;
068        private WorkflowActionBean wfAction;

070        public SignalXCommand(String name, int priority, String jobId) {
071            super(name, name, priority);
072            this.jobId = ParamChecker.notEmpty(jobId, "jobId");
073        }

075        public SignalXCommand(String jobId, String actionId) {
076            this("signal", 1, jobId);
077            this.actionId = ParamChecker.notEmpty(actionId, "actionId");
078        }

080        @Override
081        protected boolean isLockRequired() {
082            return true;
083        }
084    
085        @Override
086        protected String getEntityKey() {
087            return this.jobId;
088        }

090        @Override
091        protected void loadState() throws CommandException {
092            try {
093                jpaService = Services.get().get(JPAService.class);
094                if (jpaService != null) {
095                    this.wfJob = jpaService.execute(new WorkflowJobGetJPAExecutor(jobId));
096                    LogUtils.setLogInfo(wfJob, logInfo);
097                    if (actionId != null) {
098                        this.wfAction = jpaService.execute(new WorkflowActionGetJPAExecutor(actionId));
099                        LogUtils.setLogInfo(wfAction, logInfo);
100                    }
101                }
102                else {
103                    throw new CommandException(ErrorCode.E0610);
104                }
105            }
106            catch (XException ex) {
107                throw new CommandException(ex);
108            }
109        }
110    
111        @Override
112        protected void verifyPrecondition() throws CommandException, PreconditionException {
113            if ((wfAction == null) || (wfAction.isComplete() && wfAction.isPending())) {
114                if (wfJob.getStatus() != WorkflowJob.Status.RUNNING && wfJob.getStatus() != WorkflowJob.Status.PREP) {
115                    throw new PreconditionException(ErrorCode.E0813, wfJob.getStatusStr());
116                }
117            }
118            else {
119                throw new PreconditionException(ErrorCode.E0814, actionId, wfAction.getStatusStr(), wfAction.isPending());
120            }
121        }
122    
123        @Override
124        protected Void execute() throws CommandException {
125            LOG.debug("STARTED SignalCommand for jobid=" + jobId + ", actionId=" + actionId);
126            WorkflowInstance workflowInstance = wfJob.getWorkflowInstance();
127            workflowInstance.setTransientVar(WorkflowStoreService.WORKFLOW_BEAN, wfJob);
128            boolean completed = false;
129            boolean skipAction = false;
130            if (wfAction == null) {
131                if (wfJob.getStatus() == WorkflowJob.Status.PREP) {
132                    try {
133                        completed = workflowInstance.start();
134                    }
135                    catch (WorkflowException e) {
136                        throw new CommandException(e);
137                    }
138                    wfJob.setStatus(WorkflowJob.Status.RUNNING);
139                    wfJob.setStartTime(new Date());
140                    wfJob.setWorkflowInstance(workflowInstance);
141                    // 1. Add SLA status event for WF-JOB with status STARTED
142                    // 2. Add SLA registration events for all WF_ACTIONS
143                    SLADbXOperations.writeStausEvent(wfJob.getSlaXml(), jobId, Status.STARTED, SlaAppType.WORKFLOW_JOB);
144                    writeSLARegistrationForAllActions(workflowInstance.getApp().getDefinition(), wfJob.getUser(), wfJob
145                            .getGroup(), wfJob.getConf());
146                    queue(new NotificationXCommand(wfJob));
147                }
148                else {
149                    throw new CommandException(ErrorCode.E0801, wfJob.getId());
150                }
151            }
152            else {
153                String skipVar = workflowInstance.getVar(wfAction.getName() + WorkflowInstance.NODE_VAR_SEPARATOR
154                        + ReRunCommand.TO_SKIP);
155                if (skipVar != null) {
156                    skipAction = skipVar.equals("true");
157                }
158                try {
159                    completed = workflowInstance.signal(wfAction.getExecutionPath(), wfAction.getSignalValue());
160                }
161                catch (WorkflowException e) {
162                    throw new CommandException(e);
163                }
164                wfJob.setWorkflowInstance(workflowInstance);
165                wfAction.resetPending();
166                if (!skipAction) {
167                    wfAction.setTransition(workflowInstance.getTransition(wfAction.getName()));
168                }
169                try {
170                    jpaService.execute(new WorkflowActionUpdateJPAExecutor(wfAction));
171                }
172                catch (JPAExecutorException je) {
173                    throw new CommandException(je);
174                }
175            }
176    
177            if (completed) {
178                try {
179                    for (String actionToKillId : WorkflowStoreService.getActionsToKill(workflowInstance)) {
180                        WorkflowActionBean actionToKill;
181    
182                        actionToKill = jpaService.execute(new WorkflowActionGetJPAExecutor(actionToKillId));
183    
184                        actionToKill.setPending();
185                        actionToKill.setStatus(WorkflowActionBean.Status.KILLED);
186                        jpaService.execute(new WorkflowActionUpdateJPAExecutor(actionToKill));
187                        queue(new ActionKillXCommand(actionToKill.getId(), actionToKill.getType()));
188                    }
189    
190                    for (String actionToFailId : WorkflowStoreService.getActionsToFail(workflowInstance)) {
191                        WorkflowActionBean actionToFail = jpaService.execute(new WorkflowActionGetJPAExecutor(
192                                actionToFailId));
193                        actionToFail.resetPending();
194                        actionToFail.setStatus(WorkflowActionBean.Status.FAILED);
195                        SLADbXOperations.writeStausEvent(wfAction.getSlaXml(), wfAction.getId(), Status.FAILED,
196                                SlaAppType.WORKFLOW_ACTION);
197                        jpaService.execute(new WorkflowActionUpdateJPAExecutor(actionToFail));
198                    }
199                }
200                catch (JPAExecutorException je) {
201                    throw new CommandException(je);
202                }
203    
204                wfJob.setStatus(WorkflowJob.Status.valueOf(workflowInstance.getStatus().toString()));
205                wfJob.setEndTime(new Date());
206                wfJob.setWorkflowInstance(workflowInstance);
207                Status slaStatus = Status.SUCCEEDED;
208                switch (wfJob.getStatus()) {
209                    case SUCCEEDED:
210                        slaStatus = Status.SUCCEEDED;
211                        break;
212                    case KILLED:
213                        slaStatus = Status.KILLED;
214                        break;
215                    case FAILED:
216                        slaStatus = Status.FAILED;
217                        break;
218                    default: // TODO SUSPENDED
219                        break;
220                }
221                SLADbXOperations.writeStausEvent(wfJob.getSlaXml(), jobId, slaStatus, SlaAppType.WORKFLOW_JOB);
222                queue(new NotificationXCommand(wfJob));
223                if (wfJob.getStatus() == WorkflowJob.Status.SUCCEEDED) {
224                    InstrumentUtils.incrJobCounter(INSTR_SUCCEEDED_JOBS_COUNTER_NAME, 1, getInstrumentation());
225                }
226    
227                // output message for Kill node
228                if (wfAction != null) { // wfAction could be a no-op job
229                    NodeDef nodeDef = workflowInstance.getNodeDef(wfAction.getExecutionPath());
230                    if (nodeDef != null && nodeDef instanceof KillNodeDef) {
231                        boolean isRetry = false;
232                        boolean isUserRetry = false;
233                        ActionExecutorContext context = new ActionXCommand.ActionExecutorContext(wfJob, wfAction, isRetry,
234                                isUserRetry);
235                        try {
236                            String tmpNodeConf = nodeDef.getConf();
237                            String actionConf = context.getELEvaluator().evaluate(tmpNodeConf, String.class);
238                            LOG.debug("Try to resolve KillNode message for jobid [{0}], actionId [{1}], before resolve [{2}], after resolve [{3}]",
239                                            jobId, actionId, tmpNodeConf, actionConf);
240                            if (wfAction.getErrorCode() != null) {
241                                wfAction.setErrorInfo(wfAction.getErrorCode(), actionConf);
242                            }
243                            else {
244                                wfAction.setErrorInfo(ErrorCode.E0729.toString(), actionConf);
245                            }
246                            jpaService.execute(new WorkflowActionUpdateJPAExecutor(wfAction));
247                        }
248                        catch (JPAExecutorException je) {
249                            throw new CommandException(je);
250                        }
251                        catch (Exception ex) {
252                            LOG.warn("Exception in SignalXCommand ", ex.getMessage(), ex);
253                            throw new CommandException(ErrorCode.E0729, wfAction.getName(), ex);
254                        }
255                    }
256                }
257    
258            }
259            else {
260                for (WorkflowActionBean newAction : WorkflowStoreService.getStartedActions(workflowInstance)) {
261                    String skipVar = workflowInstance.getVar(newAction.getName() + WorkflowInstance.NODE_VAR_SEPARATOR
262                            + ReRunCommand.TO_SKIP);
263                    boolean skipNewAction = false;
264                    if (skipVar != null) {
265                        skipNewAction = skipVar.equals("true");
266                    }
267                    try {
268                        if (skipNewAction) {
269                            WorkflowActionBean oldAction;
270    
271                            oldAction = jpaService.execute(new WorkflowActionGetJPAExecutor(newAction.getId()));
272    
273                            oldAction.setPending();
274                            jpaService.execute(new WorkflowActionUpdateJPAExecutor(oldAction));
275    
276                            queue(new SignalXCommand(jobId, oldAction.getId()));
277                        }
278                        else {
279                            newAction.setPending();
280                            String actionSlaXml = getActionSLAXml(newAction.getName(), workflowInstance.getApp()
281                                    .getDefinition(), wfJob.getConf());
282                            newAction.setSlaXml(actionSlaXml);
283                            jpaService.execute(new WorkflowActionInsertJPAExecutor(newAction));
284                            LOG.debug("SignalXCommand: Name: "+ newAction.getName() + ", Id: " +newAction.getId() + ", Authcode:" + newAction.getCred());
285                            queue(new ActionStartXCommand(newAction.getId(), newAction.getType()));
286                        }
287                    }
288                    catch (JPAExecutorException je) {
289                        throw new CommandException(je);
290                    }
291                }
292            }
293    
294            try {
295                jpaService.execute(new WorkflowJobUpdateJPAExecutor(wfJob));
296            }
297            catch (JPAExecutorException je) {
298                throw new CommandException(je);
299            }
300            LOG.debug(
301                    "Updated the workflow status to " + wfJob.getId() + "  status =" + wfJob.getStatusStr());
302            if (wfJob.getStatus() != WorkflowJob.Status.RUNNING && wfJob.getStatus() != WorkflowJob.Status.SUSPENDED) {
303                // update coordinator action
304                new CoordActionUpdateXCommand(wfJob).call();    //Note: Called even if wf is not necessarily instantiated by coordinator
305                new WfEndXCommand(wfJob).call(); //To delete the WF temp dir
306            }
307            LOG.debug("ENDED SignalCommand for jobid=" + jobId + ", actionId=" + actionId);
308            return null;
309        }

311        public static ELEvaluator createELEvaluatorForGroup(Configuration conf, String group) {
312            ELEvaluator eval = Services.get().get(ELService.class).createEvaluator(group);
313            for (Map.Entry<String, String> entry : conf) {
314                eval.setVariable(entry.getKey(), entry.getValue());
315            }
316            return eval;
317        }

319        @SuppressWarnings("unchecked")
320        private String getActionSLAXml(String actionName, String wfXml, String wfConf) throws CommandException {
321            String slaXml = null;
322            try {
323                Element eWfJob = XmlUtils.parseXml(wfXml);
324                for (Element action : (List<Element>) eWfJob.getChildren("action", eWfJob.getNamespace())) {
325                    if (action.getAttributeValue("name").equals(actionName) == false) {
326                        continue;
327                    }
328                    Element eSla = action.getChild("info", Namespace.getNamespace(SchemaService.SLA_NAME_SPACE_URI));
329                    if (eSla != null) {
330                        slaXml = XmlUtils.prettyPrint(eSla).toString();
331                        break;
332                    }
333                }
334            }
335            catch (Exception e) {
336                throw new CommandException(ErrorCode.E1004, e.getMessage(), e);
337            }
338            return slaXml;
339        }
340    
341        private String resolveSla(Element eSla, Configuration conf) throws CommandException {
342            String slaXml = null;
343            try {
344                ELEvaluator evalSla = SubmitCommand.createELEvaluatorForGroup(conf, "wf-sla-submit");
345                slaXml = SubmitCommand.resolveSla(eSla, evalSla);
346            }
347            catch (Exception e) {
348                throw new CommandException(ErrorCode.E1004, e.getMessage(), e);
349            }
350            return slaXml;
351        }
352    
353        @SuppressWarnings("unchecked")
354        private void writeSLARegistrationForAllActions(String wfXml, String user, String group, String strConf)
355                throws CommandException {
356            try {
357                Element eWfJob = XmlUtils.parseXml(wfXml);
358                Configuration conf = new XConfiguration(new StringReader(strConf));
359                for (Element action : (List<Element>) eWfJob.getChildren("action", eWfJob.getNamespace())) {
360                    Element eSla = action.getChild("info", Namespace.getNamespace(SchemaService.SLA_NAME_SPACE_URI));
361                    if (eSla != null) {
362                        String slaXml = resolveSla(eSla, conf);
363                        eSla = XmlUtils.parseXml(slaXml);
364                        String actionId = Services.get().get(UUIDService.class).generateChildId(jobId,
365                                action.getAttributeValue("name") + "");
366                        SLADbXOperations.writeSlaRegistrationEvent(eSla, actionId, SlaAppType.WORKFLOW_ACTION, user, group);
367                    }
368                }
369            }
370            catch (Exception e) {
371                throw new CommandException(ErrorCode.E1007, "workflow:Actions " + jobId, e);
372            }
373    
374        }
