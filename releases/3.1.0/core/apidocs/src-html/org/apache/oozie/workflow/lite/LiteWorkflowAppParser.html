041    /**
042     * Class to parse and validate workflow xml
043     */
044    public class LiteWorkflowAppParser {
045    
046        private static final String DECISION_E = "decision";
047        private static final String ACTION_E = "action";
048        private static final String END_E = "end";
049        private static final String START_E = "start";
050        private static final String JOIN_E = "join";
051        private static final String FORK_E = "fork";
052        private static final Object KILL_E = "kill";
053    
054        private static final String SLA_INFO = "info";
055        private static final String CREDENTIALS = "credentials";
056    
057        private static final String NAME_A = "name";
058        private static final String CRED_A = "cred";
059        private static final String USER_RETRY_MAX_A = "retry-max";
060        private static final String USER_RETRY_INTERVAL_A = "retry-interval";
061        private static final String TO_A = "to";
062    
063        private static final String FORK_PATH_E = "path";
064        private static final String FORK_START_A = "start";
065    
066        private static final String ACTION_OK_E = "ok";
067        private static final String ACTION_ERROR_E = "error";
068    
069        private static final String DECISION_SWITCH_E = "switch";
070        private static final String DECISION_CASE_E = "case";
071        private static final String DECISION_DEFAULT_E = "default";
072    
073        private static final String KILL_MESSAGE_E = "message";
074    
075        private Schema schema;
076        private Class<? extends DecisionNodeHandler> decisionHandlerClass;
077        private Class<? extends ActionNodeHandler> actionHandlerClass;
078    
079        private static enum VisitStatus {
080            VISITING, VISITED
081        }
082    
083        ;
084    
085    
086        public LiteWorkflowAppParser(Schema schema, Class<? extends DecisionNodeHandler> decisionHandlerClass,
087                                     Class<? extends ActionNodeHandler> actionHandlerClass) throws WorkflowException {
088            this.schema = schema;
089            this.decisionHandlerClass = decisionHandlerClass;
090            this.actionHandlerClass = actionHandlerClass;
091        }
092    
093        /**
094         * Parse and validate xml to {@link LiteWorkflowApp}
095         *
096         * @param reader
097         * @return LiteWorkflowApp
098         * @throws WorkflowException
099         */
100        public LiteWorkflowApp validateAndParse(Reader reader) throws WorkflowException {
101            try {
102                StringWriter writer = new StringWriter();
103                IOUtils.copyCharStream(reader, writer);
104                String strDef = writer.toString();
105    
106                if (schema != null) {
107                    Validator validator = schema.newValidator();
108                    validator.validate(new StreamSource(new StringReader(strDef)));
109                }
110    
111                Element wfDefElement = XmlUtils.parseXml(strDef);
112                LiteWorkflowApp app = parse(strDef, wfDefElement);
113                Map<String, VisitStatus> traversed = new HashMap<String, VisitStatus>();
114                traversed.put(app.getNode(StartNodeDef.START).getName(), VisitStatus.VISITING);
115                validate(app, app.getNode(StartNodeDef.START), traversed);
116                return app;
117            }
118            catch (JDOMException ex) {
119                throw new WorkflowException(ErrorCode.E0700, ex.getMessage(), ex);
120            }
121            catch (SAXException ex) {
122                throw new WorkflowException(ErrorCode.E0701, ex.getMessage(), ex);
123            }
124            catch (IOException ex) {
125                throw new WorkflowException(ErrorCode.E0702, ex.getMessage(), ex);
126            }
127        }
128    
129        /**
130         * Parse xml to {@link LiteWorkflowApp}
131         *
132         * @param strDef
133         * @param root
134         * @return LiteWorkflowApp
135         * @throws WorkflowException
136         */
137        @SuppressWarnings({"unchecked", "ConstantConditions"})
138        private LiteWorkflowApp parse(String strDef, Element root) throws WorkflowException {
139            Namespace ns = root.getNamespace();
140            LiteWorkflowApp def = null;
141            for (Element eNode : (List<Element>) root.getChildren()) {
142                if (eNode.getName().equals(START_E)) {
143                    def = new LiteWorkflowApp(root.getAttributeValue(NAME_A), strDef,
144                                              new StartNodeDef(eNode.getAttributeValue(TO_A)));
145                }
146                else {
147                    if (eNode.getName().equals(END_E)) {
148                        def.addNode(new EndNodeDef(eNode.getAttributeValue(NAME_A)));
149                    }
150                    else {
151                        if (eNode.getName().equals(KILL_E)) {
152                            def.addNode(new KillNodeDef(eNode.getAttributeValue(NAME_A), eNode.getChildText(KILL_MESSAGE_E, ns)));
153                        }
154                        else {
155                            if (eNode.getName().equals(FORK_E)) {
156                                List<String> paths = new ArrayList<String>();
157                                for (Element tran : (List<Element>) eNode.getChildren(FORK_PATH_E, ns)) {
158                                    paths.add(tran.getAttributeValue(FORK_START_A));
159                                }
160                                def.addNode(new ForkNodeDef(eNode.getAttributeValue(NAME_A), paths));
161                            }
162                            else {
163                                if (eNode.getName().equals(JOIN_E)) {
164                                    def.addNode(new JoinNodeDef(eNode.getAttributeValue(NAME_A), eNode.getAttributeValue(TO_A)));
165                                }
166                                else {
167                                    if (eNode.getName().equals(DECISION_E)) {
168                                        Element eSwitch = eNode.getChild(DECISION_SWITCH_E, ns);
169                                        List<String> transitions = new ArrayList<String>();
170                                        for (Element e : (List<Element>) eSwitch.getChildren(DECISION_CASE_E, ns)) {
171                                            transitions.add(e.getAttributeValue(TO_A));
172                                        }
173                                        transitions.add(eSwitch.getChild(DECISION_DEFAULT_E, ns).getAttributeValue(TO_A));
174    
175                                        String switchStatement = XmlUtils.prettyPrint(eSwitch).toString();
176                                        def.addNode(new DecisionNodeDef(eNode.getAttributeValue(NAME_A), switchStatement, decisionHandlerClass,
177                                                                        transitions));
178                                    }
179                                    else {
180                                        if (ACTION_E.equals(eNode.getName())) {
181                                            String[] transitions = new String[2];
182                                            Element eActionConf = null;
183                                            for (Element elem : (List<Element>) eNode.getChildren()) {
184                                                if (ACTION_OK_E.equals(elem.getName())) {
185                                                    transitions[0] = elem.getAttributeValue(TO_A);
186                                                }
187                                                else {
188                                                    if (ACTION_ERROR_E.equals(elem.getName())) {
189                                                        transitions[1] = elem.getAttributeValue(TO_A);
190                                                    }
191                                                    else {
192                                                        if (SLA_INFO.equals(elem.getName()) || CREDENTIALS.equals(elem.getName())) {
193                                                            continue;
194                                                        }
195                                                        else {
196                                                            eActionConf = elem;
197                                                        }
198                                                    }
199                                                }
200                                            }
201                                            
202                                            String credStr = eNode.getAttributeValue(CRED_A);
203                                            String userRetryMaxStr = eNode.getAttributeValue(USER_RETRY_MAX_A);
204                                            String userRetryIntervalStr = eNode.getAttributeValue(USER_RETRY_INTERVAL_A);
205                                            
206                                            String actionConf = XmlUtils.prettyPrint(eActionConf).toString();
207                                            def.addNode(new ActionNodeDef(eNode.getAttributeValue(NAME_A), actionConf, actionHandlerClass,
208                                                                          transitions[0], transitions[1], credStr,
209                                                                          userRetryMaxStr, userRetryIntervalStr));
210                                        }
211                                        else {
212                                            if (SLA_INFO.equals(eNode.getName()) || CREDENTIALS.equals(eNode.getName())) {
213                                                // No operation is required
214                                            }
215                                            else {
216                                                throw new WorkflowException(ErrorCode.E0703, eNode.getName());
217                                            }
218                                        }
219                                    }
220                                }
221                            }
222                        }
223                    }
224                }
225            }
226            return def;
227        }

229        /**
230         * Validate workflow xml
231         *
232         * @param app
233         * @param node
234         * @param traversed
235         * @throws WorkflowException
236         */
237        private void validate(LiteWorkflowApp app, NodeDef node, Map<String, VisitStatus> traversed) throws WorkflowException {
238            if (!(node instanceof StartNodeDef)) {
239                try {
240                    ParamChecker.validateActionName(node.getName());
241                }
242                catch (IllegalArgumentException ex) {
243                    throw new WorkflowException(ErrorCode.E0724, ex.getMessage());
244                }
245            }
246            if (node instanceof ActionNodeDef) {
247                try {
248                    Element action = XmlUtils.parseXml(node.getConf());
249                    boolean supportedAction = Services.get().get(ActionService.class).getExecutor(action.getName()) != null;
250                    if (!supportedAction) {
251                        throw new WorkflowException(ErrorCode.E0723, node.getName(), action.getName());
252                    }
253                }
254                catch (JDOMException ex) {
255                    throw new RuntimeException("It should never happen, " + ex.getMessage(), ex);
256                }
257            }
258    
259            if (node instanceof EndNodeDef) {
260                traversed.put(node.getName(), VisitStatus.VISITED);
261                return;
262            }
263            if (node instanceof KillNodeDef) {
264                traversed.put(node.getName(), VisitStatus.VISITED);
265                return;
266            }
267            for (String transition : node.getTransitions()) {
268    
269                if (app.getNode(transition) == null) {
270                    throw new WorkflowException(ErrorCode.E0708, node.getName(), transition);
271                }
272    
273                //check if it is a cycle
274                if (traversed.get(app.getNode(transition).getName()) == VisitStatus.VISITING) {
275                    throw new WorkflowException(ErrorCode.E0707, app.getNode(transition).getName());
276                }
277                //ignore validated one
278                if (traversed.get(app.getNode(transition).getName()) == VisitStatus.VISITED) {
279                    continue;
280                }
281    
282                traversed.put(app.getNode(transition).getName(), VisitStatus.VISITING);
283                validate(app, app.getNode(transition), traversed);
284            }
285            traversed.put(node.getName(), VisitStatus.VISITED);
286        }
