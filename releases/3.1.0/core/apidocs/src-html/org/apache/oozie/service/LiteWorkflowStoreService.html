053        /**
054         * Delegation method used by the Action and Decision {@link NodeHandler} on start. <p/> This method provides the
055         * necessary information to create ActionExecutors.
056         *
057         * @param context NodeHandler context.
058         * @throws WorkflowException thrown if there was an error parsing the action configuration.
059         */
060        @SuppressWarnings("unchecked")
061        protected static void liteExecute(NodeHandler.Context context) throws WorkflowException {
062            XLog log = XLog.getLog(LiteWorkflowStoreService.class);
063            String jobId = context.getProcessInstance().getId();
064            String nodeName = context.getNodeDef().getName();
065            String skipVar = context.getProcessInstance().getVar(context.getNodeDef().getName()
066                    + WorkflowInstance.NODE_VAR_SEPARATOR + ReRunXCommand.TO_SKIP);
067            boolean skipAction = false;
068            if (skipVar != null) {
069                skipAction = skipVar.equals("true");
070            }
071            WorkflowActionBean action = new WorkflowActionBean();
072            String actionId = Services.get().get(UUIDService.class).generateChildId(jobId, nodeName);
073    
074            if (!skipAction) {
075                String nodeConf = context.getNodeDef().getConf();
076                String executionPath = context.getExecutionPath();
077    
078                String actionType;
079                try {
080                    Element element = XmlUtils.parseXml(nodeConf);
081                    actionType = element.getName();
082                    nodeConf = XmlUtils.prettyPrint(element).toString();
083                }
084                catch (JDOMException ex) {
085                    throw new WorkflowException(ErrorCode.E0700, ex.getMessage(), ex);
086                }
087    
088                log.debug(" Creating action for node [{0}]", nodeName);
089                action.setType(actionType);
090                action.setExecutionPath(executionPath);
091                action.setConf(nodeConf);
092                action.setLogToken(((WorkflowJobBean) context.getTransientVar(WORKFLOW_BEAN)).getLogToken());
093                action.setStatus(WorkflowAction.Status.PREP);
094                action.setJobId(jobId);
095            }
096            action.setCred(context.getNodeDef().getCred());
097            log.debug("Setting action for cred: '"+context.getNodeDef().getCred() +
098                            "', name: '"+ context.getNodeDef().getName() + "'");
099    
100            action.setUserRetryCount(0);
101            int userRetryMax = getUserRetryMax(context);
102            int userRetryInterval = getUserRetryInterval(context);
103            action.setUserRetryMax(userRetryMax);
104            action.setUserRetryInterval(userRetryInterval);
105            log.debug("Setting action for userRetryMax: '"+ userRetryMax +
106                            "', userRetryInterval: '" + userRetryInterval +
107                            "', name: '"+ context.getNodeDef().getName() + "'");
108    
109            action.setName(nodeName);
110            action.setId(actionId);
111            context.setVar(nodeName + WorkflowInstance.NODE_VAR_SEPARATOR + ACTION_ID, actionId);
112            List list = (List) context.getTransientVar(ACTIONS_TO_START);
113            if (list == null) {
114                list = new ArrayList();
115                context.setTransientVar(ACTIONS_TO_START, list);
116            }
117            list.add(action);
118        }
119    
120        private static int getUserRetryInterval(NodeHandler.Context context) throws WorkflowException {
121            Configuration conf = Services.get().get(ConfigurationService.class).getConf();
122            int ret = conf.getInt(CONF_USER_RETRY_INTEVAL, 5);
123            String userRetryInterval = context.getNodeDef().getUserRetryInterval();
124    
125            if (!userRetryInterval.equals("null")) {
126                try {
127                    ret = Integer.parseInt(userRetryInterval);
128                }
129                catch (NumberFormatException nfe) {
130                    throw new WorkflowException(ErrorCode.E0700, nfe.getMessage(), nfe);
131                }
132            }
133            return ret;
134        }
135    
136        private static int getUserRetryMax(NodeHandler.Context context) throws WorkflowException {
137            XLog log = XLog.getLog(LiteWorkflowStoreService.class);
138            Configuration conf = Services.get().get(ConfigurationService.class).getConf();
139            int ret = conf.getInt(CONF_USER_RETRY_MAX, 0);
140            int max = ret;
141            String userRetryMax = context.getNodeDef().getUserRetryMax();
142    
143            if (!userRetryMax.equals("null")) {
144                try {
145                    ret = Integer.parseInt(userRetryMax);
146                    if (ret > max) {
147                        ret = max;
148                        log.warn(ErrorCode.E0820.getTemplate(), ret, max);
149                    }
150                }
151                catch (NumberFormatException nfe) {
152                    throw new WorkflowException(ErrorCode.E0700, nfe.getMessage(), nfe);
153                }
154            }
155            else {
156                ret = 0;
157            }
158            return ret;
159        }

161        /**
162         * Get system defined and instance defined error codes for which USER_RETRY is allowed
163         *
164         * @return set of error code user-retry is allowed for
165         */
166        public static Set<String> getUserRetryErrorCode() {
167            Configuration conf = Services.get().get(ConfigurationService.class).getConf();
168            Collection<String> strings = (Collection<String>) conf.getStringCollection(CONF_USER_RETRY_ERROR_CODE);
169            Collection<String> extra = (Collection<String>) conf.getStringCollection(CONF_USER_RETRY_ERROR_CODE_EXT);
170            Set<String> set = new HashSet<String>();
171            set.addAll(strings);
172            set.addAll(extra);
173            return set;
174        }

176        /**
177         * Get NodeDef default version, _oozie_inst_v_0 or _oozie_inst_v_1
178         *
179         * @return nodedef default version
180         * @throws WorkflowException thrown if there was an error parsing the action configuration.
181        */
182        public static String getNodeDefDefaultVersion() throws WorkflowException {
183            Configuration conf = Services.get().get(ConfigurationService.class).getConf();
184            String ret = conf.get(CONF_NODE_DEF_VERSION);
185            if (ret == null) {
186                ret = NODE_DEF_VERSION_1;
187            }
188            return ret;
189        }

191        /**
192         * Delegation method used when failing actions. <p/>
193         *
194         * @param context NodeHandler context.
195         */
196        @SuppressWarnings("unchecked")
197        protected static void liteFail(NodeHandler.Context context) {
198            liteTerminate(context, ACTIONS_TO_FAIL);
199        }
200    
201        /**
202         * Delegation method used when killing actions. <p/>
203         *
204         * @param context NodeHandler context.
205         */
206        @SuppressWarnings("unchecked")
207        protected static void liteKill(NodeHandler.Context context) {
208            liteTerminate(context, ACTIONS_TO_KILL);
209        }
210    
211        /**
212         * Used to terminate jobs - FAIL or KILL. <p/>
213         *
214         * @param context NodeHandler context.
215         * @param transientVar The transient variable name.
216         */
217        @SuppressWarnings("unchecked")
218        private static void liteTerminate(NodeHandler.Context context, String transientVar) {
219            List<String> list = (List<String>) context.getTransientVar(transientVar);
220            if (list == null) {
221                list = new ArrayList<String>();
222                context.setTransientVar(transientVar, list);
223            }
224            list.add(context.getVar(context.getNodeDef().getName() + WorkflowInstance.NODE_VAR_SEPARATOR + ACTION_ID));
225        }
