164        public synchronized boolean start() throws WorkflowException {
165            if (status != Status.PREP) {
166                throw new WorkflowException(ErrorCode.E0719);
167            }
168            log.debug(XLog.STD, "Starting job");
169            status = Status.RUNNING;
170            executionPaths.put(ROOT, new NodeInstance(StartNodeDef.START));
171            return signal(ROOT, StartNodeDef.START);
172        }

176        public synchronized boolean signal(String executionPath, String signalValue) throws WorkflowException {
177            ParamChecker.notEmpty(executionPath, "executionPath");
178            ParamChecker.notNull(signalValue, "signalValue");
179            log.debug(XLog.STD, "Signaling job execution path [{0}] signal value [{1}]", executionPath, signalValue);
180            if (status != Status.RUNNING) {
181                throw new WorkflowException(ErrorCode.E0716);
182            }
183            NodeInstance nodeJob = executionPaths.get(executionPath);
184            if (nodeJob == null) {
185                status = Status.FAILED;
186                log.error("invalid execution path [{0}]", executionPath);
187            }
188            NodeDef nodeDef = null;
189            if (!status.isEndState()) {
190                nodeDef = def.getNode(nodeJob.nodeName);
191                if (nodeDef == null) {
192                    status = Status.FAILED;
193                    log.error("invalid transition [{0}]", nodeJob.nodeName);
194                }
195            }
196            if (!status.isEndState()) {
197                NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
198                boolean exiting = true;
199    
200                Context context = new Context(nodeDef, executionPath, signalValue);
201                if (!nodeJob.started) {
202                    try {
203                        nodeHandler.loopDetection(context);
204                        exiting = nodeHandler.enter(context);
205                        nodeJob.started = true;
206                    }
207                    catch (WorkflowException ex) {
208                        status = Status.FAILED;
209                        throw ex;
210                    }
211                }
212    
213                if (exiting) {
214                    List<String> pathsToStart = new ArrayList<String>();
215                    List<String> fullTransitions;
216                    try {
217                        fullTransitions = nodeHandler.multiExit(context);
218                        int last = fullTransitions.size() - 1;
219                        // TEST THIS
220                        if (last >= 0) {
221                            String transitionTo = getTransitionNode(fullTransitions.get(last));
222    
223                            persistentVars.put(nodeDef.getName() + WorkflowInstance.NODE_VAR_SEPARATOR + TRANSITION_TO,
224                                               transitionTo);
225                        }
226                    }
227                    catch (WorkflowException ex) {
228                        status = Status.FAILED;
229                        throw ex;
230                    }
231    
232                    if (context.status == Status.KILLED) {
233                        status = Status.KILLED;
234                        log.debug(XLog.STD, "Completing job, kill node [{0}]", nodeJob.nodeName);
235                    }
236                    else {
237                        if (context.status == Status.FAILED) {
238                            status = Status.FAILED;
239                            log.debug(XLog.STD, "Completing job, fail node [{0}]", nodeJob.nodeName);
240                        }
241                        else {
242                            if (context.status == Status.SUCCEEDED) {
243                                status = Status.SUCCEEDED;
244                                log.debug(XLog.STD, "Completing job, end node [{0}]", nodeJob.nodeName);
245                            }
246    /*
247                    else if (context.status == Status.SUSPENDED) {
248                        status = Status.SUSPENDED;
249                        log.debug(XLog.STD, "Completing job, end node [{0}]", nodeJob.nodeName);
250                    }
251    */
252                            else {
253                                for (String fullTransition : fullTransitions) {
254                                    // this is the whole trick for forking, we need the
255                                    // executionpath and the transition
256                                    // in the case of no forking last element of
257                                    // executionpath is different from transition
258                                    // in the case of forking they are the same
259    
260                                    log.debug(XLog.STD, "Exiting node [{0}] with transition[{1}]", nodeJob.nodeName,
261                                              fullTransition);
262    
263                                    String execPathFromTransition = getExecutionPath(fullTransition);
264                                    String transition = getTransitionNode(fullTransition);
265                                    def.validateTransition(nodeJob.nodeName, transition);
266    
267                                    NodeInstance nodeJobInPath = executionPaths.get(execPathFromTransition);
268                                    if ((nodeJobInPath == null) || (!transition.equals(nodeJobInPath.nodeName))) {
269                                        // TODO explain this IF better
270                                        // If the WfJob is signaled with the parent
271                                        // execution executionPath again
272                                        // The Fork node will execute again.. and replace
273                                        // the Node WorkflowJobBean
274                                        // so this is required to prevent that..
275                                        // Question : Should we throw an error in this case
276                                        // ??
277                                        executionPaths.put(execPathFromTransition, new NodeInstance(transition));
278                                        pathsToStart.add(execPathFromTransition);
279                                    }
280    
281                                }
282                                // signal all new synch transitions
283                                for (String pathToStart : pathsToStart) {
284                                    signal(pathToStart, "::synch::");
285                                }
286                            }
287                        }
288                    }
289                }
290            }
291            if (status.isEndState()) {
292                if (status == Status.FAILED) {
293                    List<String> failedNodes = terminateNodes(status);
294                    log.warn(XLog.STD, "Workflow completed [{0}], failing [{1}] running nodes", status, failedNodes
295                            .size());
296                }
297                else {
298                    List<String> killedNodes = terminateNodes(Status.KILLED);
299                    if (killedNodes.size() > 1) {
300                        log.warn(XLog.STD, "Workflow completed [{0}], killing [{1}] running nodes", status, killedNodes
301                                .size());
302                    }
303                }
304            }
305            return status.isEndState();
306        }

309         * Get NodeDef from workflow instance
310         * @param executionPath execution path

312         */
313        public NodeDef getNodeDef(String executionPath) {
314            NodeInstance nodeJob = executionPaths.get(executionPath);
315            NodeDef nodeDef = null;
316            if (nodeJob == null) {
317                log.error("invalid execution path [{0}]", executionPath);
318            }
319            else {
320                nodeDef = def.getNode(nodeJob.nodeName);
321                if (nodeDef == null) {
322                    log.error("invalid transition [{0}]", nodeJob.nodeName);
323                }
324            }
325            return nodeDef;
326        }

328        public synchronized void fail(String nodeName) throws WorkflowException {
329            if (status.isEndState()) {
330                throw new WorkflowException(ErrorCode.E0718);
331            }
332            String failedNode = failNode(nodeName);
333            if (failedNode != null) {
334                log.warn(XLog.STD, "Workflow Failed. Failing node [{0}]", failedNode);
335            }
336            else {
337                //TODO failed attempting to fail the action. EXCEPTION
338            }
339            List<String> killedNodes = killNodes();
340            if (killedNodes.size() > 1) {
341                log.warn(XLog.STD, "Workflow Failed, killing [{0}] nodes", killedNodes.size());
342            }
343            status = Status.FAILED;
344        }

346        public synchronized void kill() throws WorkflowException {
347            if (status.isEndState()) {
348                throw new WorkflowException(ErrorCode.E0718);
349            }
350            log.debug(XLog.STD, "Killing job");
351            List<String> killedNodes = killNodes();
352            if (killedNodes.size() > 1) {
353                log.warn(XLog.STD, "workflow killed, killing [{0}] nodes", killedNodes.size());
354            }
355            status = Status.KILLED;
356        }

423        private List<String> terminateNodes(Status endStatus) {
424            List<String> endNodes = new ArrayList<String>();
425            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
426                if (entry.getValue().started) {
427                    NodeDef nodeDef = def.getNode(entry.getValue().nodeName);
428                    NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
429                    try {
430                        if (endStatus == Status.KILLED) {
431                            nodeHandler.kill(new Context(nodeDef, entry.getKey(), null));
432                        }
433                        else {
434                            if (endStatus == Status.FAILED) {
435                                nodeHandler.fail(new Context(nodeDef, entry.getKey(), null));
436                            }
437                        }
438                        endNodes.add(nodeDef.getName());
439                    }
440                    catch (Exception ex) {
441                        log.warn(XLog.STD, "Error Changing node state to [{0}] for Node [{1}]", endStatus.toString(),
442                                 nodeDef.getName(), ex);
443                    }
444                }
445            }
446            return endNodes;
447        }

449        private String failNode(String nodeName) {
450            String failedNode = null;
451            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
452                String node = entry.getKey();
453                NodeInstance nodeInstance = entry.getValue();
454                if (nodeInstance.started && nodeInstance.nodeName.equals(nodeName)) {
455                    NodeDef nodeDef = def.getNode(nodeInstance.nodeName);
456                    NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
457                    try {
458                        nodeHandler.fail(new Context(nodeDef, node, null));
459                        failedNode = nodeDef.getName();
460                        nodeInstance.started = false;
461                    }
462                    catch (Exception ex) {
463                        log.warn(XLog.STD, "Error failing node [{0}]", nodeDef.getName(), ex);
464                    }
465                    return failedNode;
466                }
467            }
468            return failedNode;
469        }

471        private List<String> killNodes() {
472            List<String> killedNodes = new ArrayList<String>();
473            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
474                String node = entry.getKey();
475                NodeInstance nodeInstance = entry.getValue();
476                if (nodeInstance.started) {
477                    NodeDef nodeDef = def.getNode(nodeInstance.nodeName);
478                    NodeHandler nodeHandler = newInstance(nodeDef.getHandlerClass());
479                    try {
480                        nodeHandler.kill(new Context(nodeDef, node, null));
481                        killedNodes.add(nodeDef.getName());
482                    }
483                    catch (Exception ex) {
484                        log.warn(XLog.STD, "Error killing node [{0}]", nodeDef.getName(), ex);
485                    }
486                }
487            }
488            return killedNodes;
489        }

496            return path + child + PATH_SEPARATOR;

499        private static String getParentPath(String path) {
500            path = path.substring(0, path.length() - 1);
501            return (path.length() == 0) ? null : path.substring(0, path.lastIndexOf(PATH_SEPARATOR) + 1);
502        }

504        private static String createFullTransition(String executionPath, String transition) {
505            return executionPath + TRANSITION_SEPARATOR + transition;
506        }
507    
508        private static String getExecutionPath(String fullTransition) {
509            int index = fullTransition.indexOf(TRANSITION_SEPARATOR);
510            if (index == -1) {
511                throw new IllegalArgumentException("Invalid fullTransition");
512            }
513            return fullTransition.substring(0, index);
514        }
515    
516        private static String getTransitionNode(String fullTransition) {
517            int index = fullTransition.indexOf(TRANSITION_SEPARATOR);
518            if (index == -1) {
519                throw new IllegalArgumentException("Invalid fullTransition");
520            }
521            return fullTransition.substring(index + 1);
522        }

528        private void refreshLog() {
529            XLog.Info.get().setParameter(XLogService.USER, conf.get(OozieClient.USER_NAME));
530            XLog.Info.get().setParameter(XLogService.GROUP, conf.get(OozieClient.GROUP_NAME));
531            XLog.Info.get().setParameter(DagXLogInfoService.APP, def.getName());
532            XLog.Info.get().setParameter(DagXLogInfoService.TOKEN, conf.get(OozieClient.LOG_TOKEN, ""));
533            XLog.Info.get().setParameter(DagXLogInfoService.JOB, instanceId);
534            log = XLog.getLog(getClass());
535        }

545        @Override
546        public void write(DataOutput dOut) throws IOException {
547            dOut.writeUTF(instanceId);
548    
549            //Hadoop Configuration has to get its act right
550            ByteArrayOutputStream baos = new ByteArrayOutputStream();
551            conf.writeXml(baos);
552            baos.close();
553            byte[] array = baos.toByteArray();
554            dOut.writeInt(array.length);
555            dOut.write(array);
556    
557            def.write(dOut);
558            dOut.writeUTF(status.toString());
559            dOut.writeInt(executionPaths.size());
560            for (Map.Entry<String, NodeInstance> entry : executionPaths.entrySet()) {
561                dOut.writeUTF(entry.getKey());
562                dOut.writeUTF(entry.getValue().nodeName);
563                dOut.writeBoolean(entry.getValue().started);
564            }
565            dOut.writeInt(persistentVars.size());
566            for (Map.Entry<String, String> entry : persistentVars.entrySet()) {
567                dOut.writeUTF(entry.getKey());
568                dOut.writeUTF(entry.getValue());
569            }
570        }

572        @Override
573        public void readFields(DataInput dIn) throws IOException {
574            instanceId = dIn.readUTF();
575    
576            //Hadoop Configuration has to get its act right
577            int len = dIn.readInt();
578            byte[] array = new byte[len];
579            dIn.readFully(array);
580            ByteArrayInputStream bais = new ByteArrayInputStream(array);
581            conf = new XConfiguration(bais);
582    
583            def = new LiteWorkflowApp();
584            def.readFields(dIn);
585            status = Status.valueOf(dIn.readUTF());
586            int numExPaths = dIn.readInt();
587            for (int x = 0; x < numExPaths; x++) {
588                String path = dIn.readUTF();
589                String nodeName = dIn.readUTF();
590                boolean isStarted = dIn.readBoolean();
591                NodeInstance nodeInstance = new NodeInstance(nodeName);
592                nodeInstance.started = isStarted;
593                executionPaths.put(path, nodeInstance);
594            }
595            int numVars = dIn.readInt();
596            for (int x = 0; x < numVars; x++) {
597                String vName = dIn.readUTF();
598                String vVal = dIn.readUTF();
599                persistentVars.put(vName, vVal);
600            }
601            refreshLog();
602        }
