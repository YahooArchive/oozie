043    /**
044     * The callable queue service queues {@link XCallable}s for asynchronous execution.
045     * <p/>
046     * Callables can be queued for immediate execution or for delayed execution (some time in the future).
047     * <p/>
048     * Callables are consumed from the queue for execution based on their priority.
049     * <p/>
050     * When the queues (for immediate execution and for delayed execution) are full, the callable queue service stops
051     * queuing callables.
052     * <p/>
053     * A thread-pool is used to execute the callables asynchronously.
054     * <p/>
055     * The following configuration parameters control the callable queue service:
056     * <p/>
057     * {@link #CONF_QUEUE_SIZE} size of the immediate execution queue. Defaulf value is 10000.
058     * <p/>
059     * {@link #CONF_THREADS} number of threads in the thread-pool used for asynchronous command execution. When this number
060     * of threads is reached, commands remain the queue until threads become available. Sets up a priority queue for the
061     * execution of Commands via a ThreadPool. Sets up a Delayed Queue to handle actions which will be ready for execution
062     * sometime in the future.
063     */
064    public class CallableQueueService implements Service, Instrumentable {
065        private static final String INSTRUMENTATION_GROUP = "callablequeue";
066        private static final String INSTR_IN_QUEUE_TIME_TIMER = "time.in.queue";
067        private static final String INSTR_EXECUTED_COUNTER = "executed";
068        private static final String INSTR_FAILED_COUNTER = "failed";
069        private static final String INSTR_QUEUED_COUNTER = "queued";
070        private static final String INSTR_QUEUE_SIZE_SAMPLER = "queue.size";
071        private static final String INSTR_THREADS_ACTIVE_SAMPLER = "threads.active";
072    
073        public static final String CONF_PREFIX = Service.CONF_PREFIX + "CallableQueueService.";
074    
075        public static final String CONF_QUEUE_SIZE = CONF_PREFIX + "queue.size";
076        public static final String CONF_THREADS = CONF_PREFIX + "threads";
077        public static final String CONF_CALLABLE_CONCURRENCY = CONF_PREFIX + "callable.concurrency";
078        public static final String CONF_CALLABLE_NEXT_ELIGIBLE = CONF_PREFIX + "callable.next.eligible";
079    
080        public static final int CONCURRENCY_DELAY = 500;
081    
082        public static final int SAFE_MODE_DELAY = 60000;
083    
084        private final Map<String, AtomicInteger> activeCallables = new HashMap<String, AtomicInteger>();
085    
086        private final Map<String, Date> uniqueCallables = new ConcurrentHashMap<String, Date>();
087    
088        private int maxCallableConcurrency;
089    
090        private boolean callableBegin(XCallable<?> callable) {
091            synchronized (activeCallables) {
092                AtomicInteger counter = activeCallables.get(callable.getType());
093                if (counter == null) {
094                    counter = new AtomicInteger(1);
095                    activeCallables.put(callable.getType(), counter);
096                    return true;
097                }
098                else {
099                    int i = counter.incrementAndGet();
100                    return i <= maxCallableConcurrency;
101                }
102            }
103        }
104    
105        private void callableEnd(XCallable<?> callable) {
106            synchronized (activeCallables) {
107                AtomicInteger counter = activeCallables.get(callable.getType());
108                if (counter == null) {
109                    throw new IllegalStateException("It should not happen");
110                }
111                else {
112                    counter.decrementAndGet();
113                }
114            }
115        }
116    
117        private boolean callableReachMaxConcurrency(XCallable<?> callable) {
118            synchronized (activeCallables) {
119                AtomicInteger counter = activeCallables.get(callable.getType());
120                if (counter == null) {
121                    return true;
122                }
123                else {
124                    int i = counter.get();
125                    return i < maxCallableConcurrency;
126                }
127            }
128        }

130        // Callables are wrapped with the this wrapper for execution, for logging
131        // and instrumentation.
132        // The wrapper implements Runnable and Comparable to be able to work with an
133        // executor and a priority queue.
134        class CallableWrapper extends PriorityDelayQueue.QueueElement<XCallable<?>> implements Runnable {
135            private Instrumentation.Cron cron;
136    
137            public CallableWrapper(XCallable<?> callable, long delay) {
138                super(callable, callable.getPriority(), delay, TimeUnit.MILLISECONDS);
139                cron = new Instrumentation.Cron();
140                cron.start();
141            }
142    
143            public void run() {
144                if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
145                    log.info("Oozie is in SAFEMODE, requeuing callable [{0}] with [{1}]ms delay", getElement().getType(),
146                            SAFE_MODE_DELAY);
147                    setDelay(SAFE_MODE_DELAY, TimeUnit.MILLISECONDS);
148                    removeFromUniqueCallables();
149                    queue(this, true);
150                    return;
151                }
152                XCallable<?> callable = getElement();
153                try {
154                    if (callableBegin(callable)) {
155                        cron.stop();
156                        addInQueueCron(cron);
157                        XLog.Info.get().clear();
158                        XLog log = XLog.getLog(getClass());
159                        log.trace("executing callable [{0}]", callable.getName());
160    
161                        removeFromUniqueCallables();
162                        try {
163                            callable.call();
164                            incrCounter(INSTR_EXECUTED_COUNTER, 1);
165                            log.trace("executed callable [{0}]", callable.getName());
166                        }
167                        catch (Exception ex) {
168                            incrCounter(INSTR_FAILED_COUNTER, 1);
169                            log.warn("exception callable [{0}], {1}", callable.getName(), ex.getMessage(), ex);
170                        }
171                        finally {
172                            XLog.Info.get().clear();
173                        }
174                    }
175                    else {
176                        log.warn("max concurrency for callable [{0}] exceeded, requeueing with [{1}]ms delay", callable
177                                .getType(), CONCURRENCY_DELAY);
178                        setDelay(CONCURRENCY_DELAY, TimeUnit.MILLISECONDS);
179                        removeFromUniqueCallables();
180                        queue(this, true);
181                        incrCounter(callable.getType() + "#exceeded.concurrency", 1);
182                    }
183                }
184                finally {
185                    callableEnd(callable);
186                }
187            }
188    
189            /**
190             * @return String the queue dump
191             */
192            @Override
193            public String toString() {
194                return "delay=" + getDelay(TimeUnit.MILLISECONDS) + ", elements=" + getElement().toString();
195            }
196    
197            /**
198             * Filter the duplicate callables from the list before queue this.
199             * <p/>
200             * If it is single callable, checking if key is in unique map or not.
201             * <p/>
202             * If it is composite callable, remove duplicates callables from the composite.
203             *
204             * @return true if this callable should be queued
205             */
206            public boolean filterDuplicates() {
207                XCallable<?> callable = getElement();
208                if (callable instanceof CompositeCallable) {
209                    return ((CompositeCallable) callable).removeDuplicates();
210                }
211                else {
212                    return uniqueCallables.containsKey(callable.getKey()) == false;
213                }
214            }
215    
216            /**
217             * Add the keys to the set
218             */
219            public void addToUniqueCallables() {
220                XCallable<?> callable = getElement();
221                if (callable instanceof CompositeCallable) {
222                    ((CompositeCallable) callable).addToUniqueCallables();
223                }
224                else {
225                    ((ConcurrentHashMap<String, Date>) uniqueCallables).putIfAbsent(callable.getKey(), new Date());
226                }
227            }
228    
229            /**
230             * Remove the keys from the set
231             */
232            public void removeFromUniqueCallables() {
233                XCallable<?> callable = getElement();
234                if (callable instanceof CompositeCallable) {
235                    ((CompositeCallable) callable).removeFromUniqueCallables();
236                }
237                else {
238                    uniqueCallables.remove(callable.getKey());
239                }
240            }
241    
242        }

244        class CompositeCallable implements XCallable<Void> {
245            private List<XCallable<?>> callables;
246            private String name;
247            private int priority;
248            private long createdTime;
249    
250            public CompositeCallable(List<? extends XCallable<?>> callables) {
251                this.callables = new ArrayList<XCallable<?>>(callables);
252                priority = 0;
253                createdTime = Long.MAX_VALUE;
254                StringBuilder sb = new StringBuilder();
255                String separator = "[";
256                for (XCallable<?> callable : callables) {
257                    priority = Math.max(priority, callable.getPriority());
258                    createdTime = Math.min(createdTime, callable.getCreatedTime());
259                    sb.append(separator).append(callable.getName());
260                    separator = ",";
261                }
262                sb.append("]");
263                name = sb.toString();
264            }
265    
266            @Override
267            public String getName() {
268                return name;
269            }
270    
271            @Override
272            public String getType() {
273                return "#composite#" + callables.get(0).getType();
274            }
275    
276            @Override
277            public String getKey() {
278                return "#composite#" + callables.get(0).getKey();
279            }
280    
281            @Override
282            public int getPriority() {
283                return priority;
284            }
285    
286            @Override
287            public long getCreatedTime() {
288                return createdTime;
289            }
290    
291            public Void call() throws Exception {
292                XLog log = XLog.getLog(getClass());
293    
294                for (XCallable<?> callable : callables) {
295                    log.trace("executing callable [{0}]", callable.getName());
296                    try {
297                        callable.call();
298                        incrCounter(INSTR_EXECUTED_COUNTER, 1);
299                        log.trace("executed callable [{0}]", callable.getName());
300                    }
301                    catch (Exception ex) {
302                        incrCounter(INSTR_FAILED_COUNTER, 1);
303                        log.warn("exception callable [{0}], {1}", callable.getName(), ex.getMessage(), ex);
304                    }
305                }
306    
307                // ticking -1 not to count the call to the composite callable
308                incrCounter(INSTR_EXECUTED_COUNTER, -1);
309                return null;
310            }
311    
312            /*
313             * (non-Javadoc)
314             *
315             * @see java.lang.Object#toString()
316             */
317            @Override
318            public String toString() {
319                if (callables.size() == 0) {
320                    return null;
321                }
322                StringBuilder sb = new StringBuilder();
323                int size = callables.size();
324                for (int i = 0; i < size; i++) {
325                    XCallable<?> callable = callables.get(i);
326                    sb.append("(");
327                    sb.append(callable.toString());
328                    if (i + 1 == size) {
329                        sb.append(")");
330                    }
331                    else {
332                        sb.append("),");
333                    }
334                }
335                return sb.toString();
336            }
337    
338            /**
339             * Remove the duplicate callables from the list before queue them
340             *
341             * @return true if callables should be queued
342             */
343            public boolean removeDuplicates() {
344                Set<String> set = new HashSet<String>();
345                List<XCallable<?>> filteredCallables = new ArrayList<XCallable<?>>();
346                if (callables.size() == 0) {
347                    return false;
348                }
349                for (XCallable<?> callable : callables) {
350                    if (!uniqueCallables.containsKey(callable.getKey()) && !set.contains(callable.getKey())) {
351                        filteredCallables.add(callable);
352                        set.add(callable.getKey());
353                    }
354                }
355                callables = filteredCallables;
356                if (callables.size() == 0) {
357                    return false;
358                }
359                return true;
360            }
361    
362            /**
363             * Add the keys to the set
364             */
365            public void addToUniqueCallables() {
366                for (XCallable<?> callable : callables) {
367                    ((ConcurrentHashMap<String, Date>) uniqueCallables).putIfAbsent(callable.getKey(), new Date());
368                }
369            }
370    
371            /**
372             * Remove the keys from the set
373             */
374            public void removeFromUniqueCallables() {
375                for (XCallable<?> callable : callables) {
376                    uniqueCallables.remove(callable.getKey());
377                }
378            }
379    
380        }

391         * Convenience method for instrumentation counters.
392         *

397            if (instrumentation != null) {
398                instrumentation.incr(INSTRUMENTATION_GROUP, name, count);
399            }
400        }

405            }

408        /**
409         * Initialize the command queue service.
410         *
411         * @param services services instance.
412         */
413        @Override
414        @SuppressWarnings("unchecked")
415        public void init(Services services) {
416            Configuration conf = services.getConf();
417    
418            queueSize = conf.getInt(CONF_QUEUE_SIZE, 10000);
419            int threads = conf.getInt(CONF_THREADS, 10);
420            boolean callableNextEligible = conf.getBoolean(CONF_CALLABLE_NEXT_ELIGIBLE, true);
421    
422            if (!callableNextEligible) {
423                queue = new PriorityDelayQueue<CallableWrapper>(3, 1000 * 30, TimeUnit.MILLISECONDS, queueSize) {
424                    @Override
425                    protected void debug(String msgTemplate, Object... msgArgs) {
426                        log.trace(msgTemplate, msgArgs);
427                    }
428                };
429            }
430            else {
431                // If the head of this queue has already reached max concurrency, continuously find next one
432                // which has not yet reach max concurrency.Overrided method 'eligibleToPoll' to check if the
433                // element of this queue has reached the maximum concurrency.
434                queue = new PollablePriorityDelayQueue<CallableWrapper>(3, 1000 * 30, TimeUnit.MILLISECONDS,
435                        queueSize) {
436                    @Override
437                    protected void debug(String msgTemplate, Object... msgArgs) {
438                        log.trace(msgTemplate, msgArgs);
439                    }
440    
441                    @Override
442                    protected boolean eligibleToPoll(QueueElement<?> element) {
443                        if (element != null) {
444                            CallableWrapper wrapper = (CallableWrapper) element;
445                            if (element.getElement() != null) {
446                                return callableReachMaxConcurrency(wrapper.getElement());
447                            }
448                        }
449                        return false;
450                    }
451    
452                };
453            }
454    
455            // IMPORTANT: The ThreadPoolExecutor does not always the execute
456            // commands out of the queue, there are
457            // certain conditions where commands are pushed directly to a thread.
458            // As we are using a queue with DELAYED semantics (i.e. execute the
459            // command in 5 mins) we need to make
460            // sure that the commands are always pushed to the queue.
461            // To achieve this (by looking a the ThreadPoolExecutor.execute()
462            // implementation, we are making the pool
463            // minimum size equals to the maximum size (thus threads are keep always
464            // running) and we are warming up
465            // all those threads (the for loop that runs dummy runnables).
466            executor = new ThreadPoolExecutor(threads, threads, 10, TimeUnit.SECONDS, (BlockingQueue) queue);
467    
468            for (int i = 0; i < threads; i++) {
469                executor.execute(new Runnable() {
470                    public void run() {
471                        try {
472                            Thread.sleep(100);
473                        }
474                        catch (InterruptedException ex) {
475                            log.warn("Could not warm up threadpool {0}", ex.getMessage(), ex);
476                        }
477                    }
478                });
479            }
480    
481            maxCallableConcurrency = conf.getInt(CONF_CALLABLE_CONCURRENCY, 3);
482        }

484        /**
485         * Destroy the command queue service.
486         */
487        @Override
488        public void destroy() {
489            try {
490                long limit = System.currentTimeMillis() + 30 * 1000;// 30 seconds
491                executor.shutdown();
492                queue.clear();
493                while (!executor.awaitTermination(1000, TimeUnit.MILLISECONDS)) {
494                    log.info("Waiting for executor to shutdown");
495                    if (System.currentTimeMillis() > limit) {
496                        log.warn("Gave up, continuing without waiting for executor to shutdown");
497                        break;
498                    }
499                }
500            }
501            catch (InterruptedException ex) {
502                log.warn(ex);
503            }
504        }

523        private synchronized boolean queue(CallableWrapper wrapper, boolean ignoreQueueSize) {
524            if (!ignoreQueueSize && queue.size() >= queueSize) {
525                log.warn("queue if full, ignoring queuing for [{0}]", wrapper.getElement());
526                return false;
527            }
528            if (!executor.isShutdown()) {
529                if (wrapper.filterDuplicates()) {
530                    wrapper.addToUniqueCallables();
531                    try {
532                        executor.execute(wrapper);
533                    }
534                    catch (RejectedExecutionException ree) {
535                        wrapper.removeFromUniqueCallables();
536                        throw ree;
537                    }
538                }
539            }
540            else {
541                log.warn("Executor shutting down, ignoring queueing of [{0}]", wrapper.getElement());
542            }
543            return true;
544        }

563         *
564         * @param callables callables to be executed by the composite callable.
565         * @return <code>true</code> if the callables were queued, <code>false</code> if the queue is full and the callables
566         *         were not queued.

568        @SuppressWarnings("unchecked")
569        public boolean queueSerial(List<? extends XCallable<?>> callables) {
570            return queueSerial(callables, 0);

573        /**
574         * Queue a callable for asynchronous execution sometime in the future.
575         *
576         * @param callable callable to queue for delayed execution
577         * @param delay time, in milliseconds, that the callable should be delayed.
578         * @return <code>true</code> if the callable was queued, <code>false</code> if the queue is full and the callable
579         *         was not queued.
580         */
581        public synchronized boolean queue(XCallable<?> callable, long delay) {
582            if (callable == null) {
583                return true;
584            }
585            boolean queued = false;
586            if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
587                log.warn("[queue] System is in SAFEMODE. Hence no callable is queued. current queue size " + queue.size());
588            }
589            else {
590                queued = queue(new CallableWrapper(callable, delay), false);
591                if (queued) {
592                    incrCounter(INSTR_QUEUED_COUNTER, 1);
593                }
594                else {
595                    log.warn("Could not queue callable");
596                }
597            }
598            return queued;
599        }
600    
601        /**
602         * Queue a list of callables for serial execution sometime in the future.
603         * <p/>
604         * Useful to serialize callables that may compete with each other for resources.
605         * <p/>
606         * All callables will be processed with the priority of the highest priority of all callables.
607         *
608         * @param callables callables to be executed by the composite callable.
609         * @param delay time, in milliseconds, that the callable should be delayed.
610         * @return <code>true</code> if the callables were queued, <code>false</code> if the queue is full and the callables
611         *         were not queued.
612         */
613        @SuppressWarnings("unchecked")
614        public synchronized boolean queueSerial(List<? extends XCallable<?>> callables, long delay) {
615            boolean queued;
616            if (callables == null || callables.size() == 0) {
617                queued = true;
618            }
619            else if (callables.size() == 1) {
620                queued = queue(callables.get(0), delay);
621            }
622            else {
623                XCallable<?> callable = new CompositeCallable(callables);
624                queued = queue(callable, delay);
625                if (queued) {
626                    incrCounter(INSTR_QUEUED_COUNTER, callables.size());
627                }
628            }
629            return queued;
630        }

633         * Instruments the callable queue service.
634         *
635         * @param instr instance to instrument the callable queue service to.
636         */
637        public void instrument(Instrumentation instr) {
638            instrumentation = instr;
639            instr.addSampler(INSTRUMENTATION_GROUP, INSTR_QUEUE_SIZE_SAMPLER, 60, 1, new Instrumentation.Variable<Long>() {
640                public Long getValue() {
641                    return (long) queue.size();
642                }
643            });
644            instr.addSampler(INSTRUMENTATION_GROUP, INSTR_THREADS_ACTIVE_SAMPLER, 60, 1,
645                    new Instrumentation.Variable<Long>() {
646                        public Long getValue() {
647                            return (long) executor.getActiveCount();
648                        }
649                    });
650        }
