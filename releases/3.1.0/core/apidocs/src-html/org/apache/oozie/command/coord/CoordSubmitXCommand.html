191        @Override
192        protected String submit() throws CommandException {
193            String jobId = null;
194            LOG.info("STARTED Coordinator Submit");
195            InstrumentUtils.incrJobCounter(getName(), 1, getInstrumentation());
196    
197            boolean exceptionOccured = false;
198            try {
199                mergeDefaultConfig();
200    
201                String appXml = readAndValidateXml();
202                coordJob.setOrigJobXml(appXml);
203                LOG.debug("jobXml after initial validation " + XmlUtils.prettyPrint(appXml).toString());
204    
205                String appNamespace = readAppNamespace(appXml);
206                coordJob.setAppNamespace(appNamespace);
207    
208                appXml = XmlUtils.removeComments(appXml);
209                initEvaluators();
210                Element eJob = basicResolveAndIncludeDS(appXml, conf, coordJob);
211                LOG.debug("jobXml after all validation " + XmlUtils.prettyPrint(eJob).toString());
212    
213                jobId = storeToDB(eJob, coordJob);
214                // log job info for coordinator job
215                LogUtils.setLogInfo(coordJob, logInfo);
216                LOG = XLog.resetPrefix(LOG);
217    
218                if (!dryrun) {
219                    // submit a command to materialize jobs for the next 1 hour (3600 secs)
220                    // so we don't wait 10 mins for the Service to run.
221                    queue(new CoordMaterializeTransitionXCommand(jobId, 3600), 100);
222                }
223                else {
224                    Date startTime = coordJob.getStartTime();
225                    long startTimeMilli = startTime.getTime();
226                    long endTimeMilli = startTimeMilli + (3600 * 1000);
227                    Date jobEndTime = coordJob.getEndTime();
228                    Date endTime = new Date(endTimeMilli);
229                    if (endTime.compareTo(jobEndTime) > 0) {
230                        endTime = jobEndTime;
231                    }
232                    jobId = coordJob.getId();
233                    LOG.info("[" + jobId + "]: Update status to RUNNING");
234                    coordJob.setStatus(Job.Status.RUNNING);
235                    coordJob.setPending();
236                    CoordActionMaterializeCommand coordActionMatCom = new CoordActionMaterializeCommand(jobId, startTime,
237                            endTime);
238                    Configuration jobConf = null;
239                    try {
240                        jobConf = new XConfiguration(new StringReader(coordJob.getConf()));
241                    }
242                    catch (IOException e1) {
243                        LOG.warn("Configuration parse error. read from DB :" + coordJob.getConf(), e1);
244                    }
245                    String action = coordActionMatCom.materializeJobs(true, coordJob, jobConf, null);
246                    String output = coordJob.getJobXml() + System.getProperty("line.separator")
247                    + "***actions for instance***" + action;
248                    return output;
249                }
250            }
251            catch (CoordinatorJobException cex) {
252                exceptionOccured = true;
253                LOG.warn("ERROR:  ", cex);
254                throw new CommandException(cex);
255            }
256            catch (IllegalArgumentException iex) {
257                exceptionOccured = true;
258                LOG.warn("ERROR:  ", iex);
259                throw new CommandException(ErrorCode.E1003, iex);
260            }
261            catch (Exception ex) {
262                exceptionOccured = true;
263                LOG.warn("ERROR:  ", ex);
264                throw new CommandException(ErrorCode.E0803, ex);
265            }
266            finally {
267                if (exceptionOccured) {
268                    if(coordJob.getId() == null || coordJob.getId().equalsIgnoreCase("")){
269                        coordJob.setStatus(CoordinatorJob.Status.FAILED);
270                        coordJob.resetPending();
271                    }
272                }
273            }
274    
275            LOG.info("ENDED Coordinator Submit jobId=" + jobId);
276            return jobId;
277        }

283         * @throws CoordinatorJobException thrown if unable to read or validate coordinator xml

285        private String readAndValidateXml() throws CoordinatorJobException {
286            String appPath = ParamChecker.notEmpty(conf.get(OozieClient.COORDINATOR_APP_PATH),
287                    OozieClient.COORDINATOR_APP_PATH);
288            String coordXml = readDefinition(appPath);

293        /**
294         * Validate against Coordinator XSD file
295         *
296         * @param xmlContent : Input coordinator xml
297         * @throws CoordinatorJobException thrown if unable to validate coordinator xml
298         */
299        private void validateXml(String xmlContent) throws CoordinatorJobException {
300            javax.xml.validation.Schema schema = Services.get().get(SchemaService.class).getSchema(SchemaName.COORDINATOR);
301            Validator validator = schema.newValidator();
302            try {
303                validator.validate(new StreamSource(new StringReader(xmlContent)));
304            }
305            catch (SAXException ex) {
306                LOG.warn("SAXException :", ex);
307                throw new CoordinatorJobException(ErrorCode.E0701, ex.getMessage(), ex);
308            }
309            catch (IOException ex) {
310                LOG.warn("IOException :", ex);
311                throw new CoordinatorJobException(ErrorCode.E0702, ex.getMessage(), ex);
312            }
313        }

315        /**
316         * Read the application XML schema namespace
317         *
318         * @param xmlContent input coordinator xml
319         * @return app xml namespace
320         * @throws CoordinatorJobException
321         */
322        private String readAppNamespace(String xmlContent) throws CoordinatorJobException {
323            try {
324                Element coordXmlElement = XmlUtils.parseXml(xmlContent);
325                Namespace ns = coordXmlElement.getNamespace();
326                if (ns != null && bundleId != null && ns.getURI().equals(SchemaService.COORDINATOR_NAMESPACE_URI_1)) {
327                    throw new CoordinatorJobException(ErrorCode.E1319, "bundle app can not submit coordinator namespace "
328                            + SchemaService.COORDINATOR_NAMESPACE_URI_1 + ", please use 0.2 or later");
329                }
330                if (ns != null) {
331                    return ns.getURI();
332                }
333                else {
334                    throw new CoordinatorJobException(ErrorCode.E0700, "the application xml namespace is not given");
335                }
336            }
337            catch (JDOMException ex) {
338                LOG.warn("JDOMException :", ex);
339                throw new CoordinatorJobException(ErrorCode.E0700, ex.getMessage(), ex);
340            }
341        }

343        /**
344         * Merge default configuration with user-defined configuration.
345         *
346         * @throws CommandException thrown if failed to read or merge configurations
347         */
348        protected void mergeDefaultConfig() throws CommandException {
349            Path configDefault = null;
350            try {
351                String coordAppPathStr = conf.get(OozieClient.COORDINATOR_APP_PATH);
352                Path coordAppPath = new Path(coordAppPathStr);
353                String user = ParamChecker.notEmpty(conf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
354                String group = ParamChecker.notEmpty(conf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
355                FileSystem fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group,
356                        coordAppPath.toUri(), new Configuration());
357    
358                // app path could be a directory
359                if (!fs.isFile(coordAppPath)) {
360                    configDefault = new Path(coordAppPath, CONFIG_DEFAULT);
361                } else {
362                    configDefault = new Path(coordAppPath.getParent(), CONFIG_DEFAULT);
363                }
364    
365                if (fs.exists(configDefault)) {
366                    Configuration defaultConf = new XConfiguration(fs.open(configDefault));
367                    PropertiesUtils.checkDisallowedProperties(defaultConf, DISALLOWED_DEFAULT_PROPERTIES);
368                    XConfiguration.injectDefaults(defaultConf, conf);
369                }
370                else {
371                    LOG.info("configDefault Doesn't exist " + configDefault);
372                }
373                PropertiesUtils.checkDisallowedProperties(conf, DISALLOWED_USER_PROPERTIES);
374    
375                // Resolving all variables in the job properties.
376                // This ensures the Hadoop Configuration semantics is preserved.
377                XConfiguration resolvedVarsConf = new XConfiguration();
378                for (Map.Entry<String, String> entry : conf) {
379                    resolvedVarsConf.set(entry.getKey(), conf.get(entry.getKey()));
380                }
381                conf = resolvedVarsConf;
382            }
383            catch (IOException e) {
384                throw new CommandException(ErrorCode.E0702, e.getMessage() + " : Problem reading default config "
385                        + configDefault, e);
386            }
387            catch (HadoopAccessorException e) {
388                throw new CommandException(e);
389            }
390            LOG.debug("Merged CONF :" + XmlUtils.prettyPrint(conf).toString());
391        }

411        /**
412         * Insert data set into data-in and data-out tags.
413         *
414         * @param eAppXml : coordinator application XML
415         * @param eDatasets : DataSet XML
416         */
417        @SuppressWarnings("unchecked")
418        private void insertDataSet(Element eAppXml, Element eDatasets) {
419            // Adding DS definition in the coordinator XML
420            Element inputList = eAppXml.getChild("input-events", eAppXml.getNamespace());
421            if (inputList != null) {
422                for (Element dataIn : (List<Element>) inputList.getChildren("data-in", eAppXml.getNamespace())) {
423                    Element eDataset = findDataSet(eDatasets, dataIn.getAttributeValue("dataset"));
424                    dataIn.getContent().add(0, eDataset);
425                }
426            }
427            Element outputList = eAppXml.getChild("output-events", eAppXml.getNamespace());
428            if (outputList != null) {
429                for (Element dataOut : (List<Element>) outputList.getChildren("data-out", eAppXml.getNamespace())) {
430                    Element eDataset = findDataSet(eDatasets, dataOut.getAttributeValue("dataset"));
431                    dataOut.getContent().add(0, eDataset);
432                }
433            }
434        }

436        /**
437         * Find a specific dataset from a list of Datasets.
438         *
439         * @param eDatasets : List of data sets
440         * @param name : queried data set name
441         * @return one Dataset element. otherwise throw Exception
442         */
443        @SuppressWarnings("unchecked")
444        private static Element findDataSet(Element eDatasets, String name) {
445            for (Element eDataset : (List<Element>) eDatasets.getChildren("dataset", eDatasets.getNamespace())) {
446                if (eDataset.getAttributeValue("name").equals(name)) {
447                    eDataset = (Element) eDataset.clone();
448                    eDataset.detach();
449                    return eDataset;
450                }
451            }
452            throw new RuntimeException("undefined dataset: " + name);
453        }

465        /**
466         * Resolve basic entities using job Configuration.
467         *
468         * @param conf :Job configuration
469         * @param appXml : Original job XML
470         * @param coordJob : Coordinator job bean to be populated.
471         * @return Resolved job XML element.
472         * @throws CoordinatorJobException thrown if failed to resolve basic entities
473         * @throws Exception thrown if failed to resolve basic entities
474         */
475        @SuppressWarnings("unchecked")
476        protected Element resolveInitial(Configuration conf, String appXml, CoordinatorJobBean coordJob)
477        throws CoordinatorJobException, Exception {
478            Element eAppXml = XmlUtils.parseXml(appXml);
479            // job's main attributes
480            // frequency
481            String val = resolveAttribute("frequency", eAppXml, evalFreq);
482            int ival = ParamChecker.checkInteger(val, "frequency");
483            ParamChecker.checkGTZero(ival, "frequency");
484            coordJob.setFrequency(ival);
485            TimeUnit tmp = (evalFreq.getVariable("timeunit") == null) ? TimeUnit.MINUTE : ((TimeUnit) evalFreq
486                    .getVariable("timeunit"));
487            addAnAttribute("freq_timeunit", eAppXml, tmp.toString());
488            // TimeUnit
489            coordJob.setTimeUnit(CoordinatorJob.Timeunit.valueOf(tmp.toString()));
490            // End Of Duration
491            tmp = evalFreq.getVariable("endOfDuration") == null ? TimeUnit.NONE : ((TimeUnit) evalFreq
492                    .getVariable("endOfDuration"));
493            addAnAttribute("end_of_duration", eAppXml, tmp.toString());
494            // coordJob.setEndOfDuration(tmp) // TODO: Add new attribute in Job bean
495    
496            // Application name
497            if (this.coordName == null) {
498                val = resolveAttribute("name", eAppXml, evalNofuncs);
499                coordJob.setAppName(val);
500            }
501            else {
502                // this coord job is created from bundle
503                coordJob.setAppName(this.coordName);
504            }
505    
506            // start time
507            val = resolveAttribute("start", eAppXml, evalNofuncs);
508            ParamChecker.checkUTC(val, "start");
509            coordJob.setStartTime(DateUtils.parseDateUTC(val));
510            // end time
511            val = resolveAttribute("end", eAppXml, evalNofuncs);
512            ParamChecker.checkUTC(val, "end");
513            coordJob.setEndTime(DateUtils.parseDateUTC(val));
514            // Time zone
515            val = resolveAttribute("timezone", eAppXml, evalNofuncs);
516            ParamChecker.checkTimeZone(val, "timezone");
517            coordJob.setTimeZone(val);
518    
519            // controls
520            val = resolveTagContents("timeout", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
521            if (val == "") {
522                val = Services.get().getConf().get(CONF_DEFAULT_TIMEOUT_NORMAL);
523            }
524    
525            ival = ParamChecker.checkInteger(val, "timeout");
526            if (ival < 0 || ival > Services.get().getConf().getInt(CONF_DEFAULT_MAX_TIMEOUT, 129600)) {
527                ival = Services.get().getConf().getInt(CONF_DEFAULT_MAX_TIMEOUT, 129600);
528            }
529            coordJob.setTimeout(ival);
530    
531            val = resolveTagContents("concurrency", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
532            if (val == null || val.isEmpty()) {
533                val = Services.get().getConf().get(CONF_DEFAULT_CONCURRENCY, "-1");
534            }
535            ival = ParamChecker.checkInteger(val, "concurrency");
536            coordJob.setConcurrency(ival);
537    
538            val = resolveTagContents("throttle", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
539            if (val == null || val.isEmpty()) {
540                int defaultThrottle = Services.get().getConf().getInt(CONF_DEFAULT_THROTTLE, 12);
541                ival = defaultThrottle;
542            }
543            else {
544                ival = ParamChecker.checkInteger(val, "throttle");
545            }
546            int maxQueue = Services.get().getConf().getInt(CONF_QUEUE_SIZE, 10000);
547            float factor = Services.get().getConf().getFloat(CONF_MAT_THROTTLING_FACTOR, 0.10f);
548            int maxThrottle = (int) (maxQueue * factor);
549            if (ival > maxThrottle || ival < 1) {
550                ival = maxThrottle;
551            }
552            LOG.debug("max throttle " + ival);
553            coordJob.setMatThrottling(ival);
554    
555            val = resolveTagContents("execution", eAppXml.getChild("controls", eAppXml.getNamespace()), evalNofuncs);
556            if (val == "") {
557                val = Execution.FIFO.toString();
558            }
559            coordJob.setExecution(Execution.valueOf(val));
560            String[] acceptedVals = { Execution.LIFO.toString(), Execution.FIFO.toString(), Execution.LAST_ONLY.toString() };
561            ParamChecker.isMember(val, acceptedVals, "execution");
562    
563            // datasets
564            resolveTagContents("include", eAppXml.getChild("datasets", eAppXml.getNamespace()), evalNofuncs);
565            // for each data set
566            resolveDataSets(eAppXml);
567            HashMap<String, String> dataNameList = new HashMap<String, String>();
568            resolveIOEvents(eAppXml, dataNameList);
569    
570            resolveTagContents("app-path", eAppXml.getChild("action", eAppXml.getNamespace()).getChild("workflow",
571                    eAppXml.getNamespace()), evalNofuncs);
572            // TODO: If action or workflow tag is missing, NullPointerException will
573            // occur
574            Element configElem = eAppXml.getChild("action", eAppXml.getNamespace()).getChild("workflow",
575                    eAppXml.getNamespace()).getChild("configuration", eAppXml.getNamespace());
576            evalData = CoordELEvaluator.createELEvaluatorForDataEcho(conf, "coord-job-submit-data", dataNameList);
577            if (configElem != null) {
578                for (Element propElem : (List<Element>) configElem.getChildren("property", configElem.getNamespace())) {
579                    resolveTagContents("name", propElem, evalData);
580                    // Want to check the data-integrity but don't want to modify the
581                    // XML
582                    // for properties only
583                    Element tmpProp = (Element) propElem.clone();
584                    resolveTagContents("value", tmpProp, evalData);
585                }
586            }
587            resolveSLA(eAppXml, coordJob);
588            return eAppXml;
589        }

616        /**
617         * Resolve input-events/data-in and output-events/data-out tags.
618         *
619         * @param eJob : Job element
620         * @throws CoordinatorJobException thrown if failed to resolve input and output events
621         */
622        @SuppressWarnings("unchecked")
623        private void resolveIOEvents(Element eJobOrg, HashMap<String, String> dataNameList) throws CoordinatorJobException {
624            // Resolving input-events/data-in
625            // Clone the job and don't update anything in the original
626            Element eJob = (Element) eJobOrg.clone();
627            Element inputList = eJob.getChild("input-events", eJob.getNamespace());
628            if (inputList != null) {
629                TreeSet<String> eventNameSet = new TreeSet<String>();
630                for (Element dataIn : (List<Element>) inputList.getChildren("data-in", eJob.getNamespace())) {
631                    String dataInName = dataIn.getAttributeValue("name");
632                    dataNameList.put(dataInName, "data-in");
633                    // check whether there is any duplicate data-in name
634                    if (eventNameSet.contains(dataInName)) {
635                        throw new RuntimeException("Duplicate dataIn name " + dataInName);
636                    }
637                    else {
638                        eventNameSet.add(dataInName);
639                    }
640                    resolveTagContents("instance", dataIn, evalInst);
641                    resolveTagContents("start-instance", dataIn, evalInst);
642                    resolveTagContents("end-instance", dataIn, evalInst);
643                }
644            }
645            // Resolving output-events/data-out
646            Element outputList = eJob.getChild("output-events", eJob.getNamespace());
647            if (outputList != null) {
648                TreeSet<String> eventNameSet = new TreeSet<String>();
649                for (Element dataOut : (List<Element>) outputList.getChildren("data-out", eJob.getNamespace())) {
650                    String dataOutName = dataOut.getAttributeValue("name");
651                    dataNameList.put(dataOutName, "data-out");
652                    // check whether there is any duplicate data-out name
653                    if (eventNameSet.contains(dataOutName)) {
654                        throw new RuntimeException("Duplicate dataIn name " + dataOutName);
655                    }
656                    else {
657                        eventNameSet.add(dataOutName);
658                    }
659                    resolveTagContents("instance", dataOut, evalInst);
660                }
661            }
662    
663        }

694        /**
695         * Resolve datasets using job configuration.
696         *
697         * @param dsElems : Data set XML element.
698         * @throws CoordinatorJobException thrown if failed to resolve datasets
699         */
700        private void resolveDataSets(List<Element> dsElems) throws CoordinatorJobException {
701            for (Element dsElem : dsElems) {
702                // Setting up default TimeUnit and EndOFDuraion
703                evalFreq.setVariable("timeunit", TimeUnit.MINUTE);
704                evalFreq.setVariable("endOfDuration", TimeUnit.NONE);
705    
706                String val = resolveAttribute("frequency", dsElem, evalFreq);
707                int ival = ParamChecker.checkInteger(val, "frequency");
708                ParamChecker.checkGTZero(ival, "frequency");
709                addAnAttribute("freq_timeunit", dsElem, evalFreq.getVariable("timeunit") == null ? TimeUnit.MINUTE
710                        .toString() : ((TimeUnit) evalFreq.getVariable("timeunit")).toString());
711                addAnAttribute("end_of_duration", dsElem, evalFreq.getVariable("endOfDuration") == null ? TimeUnit.NONE
712                        .toString() : ((TimeUnit) evalFreq.getVariable("endOfDuration")).toString());
713                val = resolveAttribute("initial-instance", dsElem, evalNofuncs);
714                ParamChecker.checkUTC(val, "initial-instance");
715                val = resolveAttribute("timezone", dsElem, evalNofuncs);
716                ParamChecker.checkTimeZone(val, "timezone");
717                resolveTagContents("uri-template", dsElem, evalNofuncs);
718                resolveTagContents("done-flag", dsElem, evalNofuncs);
719            }
720        }

722        /**
723         * Resolve the content of a tag.
724         *
725         * @param tagName : Tag name of job XML i.e. <timeout> 10 </timeout>
726         * @param elem : Element where the tag exists.
727         * @param eval : EL evealuator
728         * @return Resolved tag content.
729         * @throws CoordinatorJobException thrown if failed to resolve tag content
730         */
731        @SuppressWarnings("unchecked")
732        private String resolveTagContents(String tagName, Element elem, ELEvaluator eval) throws CoordinatorJobException {
733            String ret = "";
734            if (elem != null) {
735                for (Element tagElem : (List<Element>) elem.getChildren(tagName, elem.getNamespace())) {
736                    if (tagElem != null) {
737                        String updated;
738                        try {
739                            updated = CoordELFunctions.evalAndWrap(eval, tagElem.getText().trim());
740    
741                        }
742                        catch (Exception e) {
743                            throw new CoordinatorJobException(ErrorCode.E1004, e.getMessage(), e);
744                        }
745                        tagElem.removeContent();
746                        tagElem.addContent(updated);
747                        ret += updated;
748                    }
749                }
750            }
751            return ret;
752        }

756         *
757         * @param attrName : Attribute name.
758         * @param elem : XML Element where attribute is defiend
759         * @param eval : ELEvaluator used to resolve
760         * @return Resolved attribute value
761         * @throws CoordinatorJobException thrown if failed to resolve an attribute value

763        private String resolveAttribute(String attrName, Element elem, ELEvaluator eval) throws CoordinatorJobException {

779         * Include referred datasets into XML.
780         *
781         * @param resolvedXml : Job XML element.
782         * @param conf : Job configuration
783         * @throws CoordinatorJobException thrown if failed to include referred datasets into XML
784         */
785        @SuppressWarnings("unchecked")
786        protected void includeDataSets(Element resolvedXml, Configuration conf) throws CoordinatorJobException {
787            Element datasets = resolvedXml.getChild("datasets", resolvedXml.getNamespace());
788            Element allDataSets = new Element("all_datasets", resolvedXml.getNamespace());
789            List<String> dsList = new ArrayList<String>();
790            if (datasets != null) {
791                for (Element includeElem : (List<Element>) datasets.getChildren("include", datasets.getNamespace())) {
792                    String incDSFile = includeElem.getTextTrim();
793                    includeOneDSFile(incDSFile, dsList, allDataSets, datasets.getNamespace());
794                }
795                for (Element e : (List<Element>) datasets.getChildren("dataset", datasets.getNamespace())) {
796                    String dsName = e.getAttributeValue("name");
797                    if (dsList.contains(dsName)) {// Override with this DS
798                        // Remove old DS
799                        removeDataSet(allDataSets, dsName);
800                    }
801                    else {
802                        dsList.add(dsName);
803                    }
804                    allDataSets.addContent((Element) e.clone());
805                }
806            }
807            insertDataSet(resolvedXml, allDataSets);
808            resolvedXml.removeChild("datasets", resolvedXml.getNamespace());
809        }

811        /**
812         * Include one dataset file.
813         *
814         * @param incDSFile : Include data set filename.
815         * @param dsList :List of dataset names to verify the duplicate.
816         * @param allDataSets : Element that includes all dataset definitions.
817         * @param dsNameSpace : Data set name space
818         * @throws CoordinatorJobException thrown if failed to include one dataset file
819         */
820        @SuppressWarnings("unchecked")
821        private void includeOneDSFile(String incDSFile, List<String> dsList, Element allDataSets, Namespace dsNameSpace)
822        throws CoordinatorJobException {
823            Element tmpDataSets = null;
824            try {
825                String dsXml = readDefinition(incDSFile);
826                LOG.debug("DSFILE :" + incDSFile + "\n" + dsXml);
827                tmpDataSets = XmlUtils.parseXml(dsXml);
828            }
829            catch (JDOMException e) {
830                LOG.warn("Error parsing included dataset [{0}].  Message [{1}]", incDSFile, e.getMessage());
831                throw new CoordinatorJobException(ErrorCode.E0700, e.getMessage());
832            }
833            resolveDataSets(tmpDataSets.getChildren("dataset"));
834            for (Element e : (List<Element>) tmpDataSets.getChildren("dataset")) {
835                String dsName = e.getAttributeValue("name");
836                if (dsList.contains(dsName)) {
837                    throw new RuntimeException("Duplicate Dataset " + dsName);
838                }
839                dsList.add(dsName);
840                Element tmp = (Element) e.clone();
841                // TODO: Don't like to over-write the external/include DS's namespace
842                tmp.setNamespace(dsNameSpace);
843                tmp.getChild("uri-template").setNamespace(dsNameSpace);
844                if (e.getChild("done-flag") != null) {
845                    tmp.getChild("done-flag").setNamespace(dsNameSpace);
846                }
847                allDataSets.addContent(tmp);
848            }
849            // nested include
850            for (Element includeElem : (List<Element>) tmpDataSets.getChildren("include", tmpDataSets.getNamespace())) {
851                String incFile = includeElem.getTextTrim();
852                includeOneDSFile(incFile, dsList, allDataSets, dsNameSpace);
853            }
854        }

872        /**
873         * Read coordinator definition.
874         *
875         * @param appPath application path.
876         * @return coordinator definition.
877         * @throws CoordinatorJobException thrown if the definition could not be read.
878         */
879        protected String readDefinition(String appPath) throws CoordinatorJobException {
880            String user = ParamChecker.notEmpty(conf.get(OozieClient.USER_NAME), OozieClient.USER_NAME);
881            String group = ParamChecker.notEmpty(conf.get(OozieClient.GROUP_NAME), OozieClient.GROUP_NAME);
882            // Configuration confHadoop = CoordUtils.getHadoopConf(conf);
883            try {
884                URI uri = new URI(appPath);
885                LOG.debug("user =" + user + " group =" + group);
886                FileSystem fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, uri,
887                        new Configuration());
888                Path appDefPath = null;
889    
890                // app path could be a directory
891                Path path = new Path(uri.getPath());
892                // check file exists for dataset include file, app xml already checked
893                if (!fs.exists(path)) {
894                    throw new URISyntaxException(path.toString(), "path not existed : " + path.toString());
895                }
896                if (!fs.isFile(path)) {
897                    appDefPath = new Path(path, COORDINATOR_XML_FILE);
898                } else {
899                    appDefPath = path;
900                }
901    
902                Reader reader = new InputStreamReader(fs.open(appDefPath));
903                StringWriter writer = new StringWriter();
904                IOUtils.copyCharStream(reader, writer);
905                return writer.toString();
906            }
907            catch (IOException ex) {
908                LOG.warn("IOException :" + XmlUtils.prettyPrint(conf), ex);
909                throw new CoordinatorJobException(ErrorCode.E1001, ex.getMessage(), ex);
910            }
911            catch (URISyntaxException ex) {
912                LOG.warn("URISyException :" + ex.getMessage());
913                throw new CoordinatorJobException(ErrorCode.E1002, appPath, ex.getMessage(), ex);
914            }
915            catch (HadoopAccessorException ex) {
916                throw new CoordinatorJobException(ex);
917            }
918            catch (Exception ex) {
919                LOG.warn("Exception :", ex);
920                throw new CoordinatorJobException(ErrorCode.E1001, ex.getMessage(), ex);
921            }
922        }
